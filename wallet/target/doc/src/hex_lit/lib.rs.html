1//! Hex literals without proc macros.
2//! 
3//! This crate implements minimalistic hex literal macros without use of proc macros.
4//! The advantages are much faster compile times, ability to work with non-literal const values and
5//! easier auditing.
6//! However, because of the use of `const fn` the crate has some limitations depending on the Rust
7//! version.
8//!
9//! Either way, the resulting type is a byte array (`[u8; N]`) that doesn't force you to write down
10//! its length. This is already very useful since the compiler can prove the length and you avoid
11//! runtime allocations.
12//!
13//! The crate is `no_std` and does **not** require an allocator.
14//!
15//! ## Usage
16//!
17//! Just pass a `&str` *constant* (usually a literal) into the [`hex`] macro.
18//!
19//! Example
20//!
21//! ```rust
22//! use hex_lit::hex;
23//!
24//! let array = hex!("2a15ff");
25//! assert_eq!(&array, &[42, 21, 255]);
26//!
27//! ```
28//!
29//! The input MUST NOT contain any spaces or other separators and it MUST have even length.
30//! Note that you can still separate long strings into chunks using the [`concat`] macro:
31//!
32//! ```rust
33//! use hex_lit::hex;
34//!
35//! let array = hex!(concat!(
36//!     "0000002a000000",
37//!     "ffffffffffffff",
38//! ));
39//! assert_eq!(&array, &[0, 0, 0, 42, 0, 0, 0, 255, 255, 255, 255, 255, 255, 255]);
40//!
41//! ```
42//!
43//! ## Features depending on Rust version
44//!
45//! * 1.41.1+ - the MSRV, use in const contexts is impossible, only the [`hex!`] macro is available.
46//! * 1.46.0+ - usage in const contexts is available and (regardless of cargo features) correctness
47//!             of input is checked at compile time. 
48//! * 1.57+ - nicer error messages for bad inputs (regardless of cargo features)
49//!
50//! ## Cargo features
51//!
52//! * `rust_v_1_46` - acknowledges bumping MSRV to 1.46+ and enables usage in const context.
53//!
54//! Bumping MSRV is intentionally explicit.
55//!
56//! Because of improved input checking it is recommended to use Rust 1.46+, prefereably 1.57+ in CI
57//! even if your targeted MSRV is lower.
58//!
59//! [`concat`]: core::concat

94// Invoked when there's an invalid digit in the string to cause an error/panic
95// This version is not const because const isn't supported anyway, so we just panic
96#[cfg(not(rust_v_1_46))]
97fn invalid_digit(digit: u8) {
98    panic!("invalid hex digit: ASCII {}", digit);
99}

101// This version makes an out-of-bounds accesses to an array to trigger compilation failure.
102#[cfg(all(not(rust_v_1_57), rust_v_1_46))]
103#[track_caller]
104const fn invalid_digit(digit: u8) {
105    let digit = digit as usize;
106    #[allow(unknown_lints)]
107    #[allow(unconditional_panic)]
108    // We add 10000 to ensure it will panic for any digit, we want to use digit in the expression
109    // to display the number even though the message is messy.
110    let _invalid_digit = [(); 0][digit + 10000];
111}
112
113// This version panics with a nice custom message
114#[cfg(rust_v_1_57)]
115#[track_caller]
116const fn invalid_digit(digit: u8) {
117    // custom message formatting because of rust limiation
118    let mut buf = [b'i', b'n', b'v', b'a', b'l', b'i', b'd', b' ', b'h', b'e', b'x', b' ', b'd', b'i', b'g', b'i', b't', b':', b' ', b'A', b'S', b'C', b'I', b'I', b' ', b' ', b' ', b' '];
119    if digit >= 100 {
120        buf[buf.len() - 3] = digit / 100 + b'0';
121    }
122    if digit >= 10 {
123        buf[buf.len() - 2] = (digit % 100) / 10 + b'0';
124    }
125    buf[buf.len() - 1] = digit % 10 + b'0';
126    let message = unsafe { core::str::from_utf8_unchecked(&buf) };
127    panic!("{}", message);
128}

130// Decodes a single hex digit (char) into an integer 0-15
131maybe_const! {
132    fn decode_digit(digit: u8) -> u8 {
133        match digit {
134            b'0'..=b'9' => digit - b'0',
135            b'a'..=b'f' => digit - b'a' + 10,
136            b'A'..=b'F' => digit - b'A' + 10,
137            _ => {
138                invalid_digit(digit);
139                0
140            }
141        }
142    }
143}

145// Decodes a single byte in the string at given **output** position.
146pub_maybe_const! {
147    fn decode_byte(hex: &str, pos: usize) -> u8 {
148        let c1 = decode_digit(hex.as_bytes()[pos * 2]);
149        let c2 = decode_digit(hex.as_bytes()[pos* 2 + 1]);
150        c1 << 4 | c2
151    }
152}

154// Internal macro implementation simplifies dispatch
155// DO NOT CALL THIS OUTSIDE OF THIS CRATE!!!
156#[doc(hidden)]
157#[macro_export]
158macro_rules! hex_impl {
159    ($hex:expr) => {
160        {
161            const HEX: &str = $hex;
162            // Fails if the length is not even
163            const _HEX_LENGTH_MUST_BE_EVEN: () = [()][$hex.len() % 2];
164            let mut out = [0u8; HEX.len() / 2];
165            // mut refs are not allowed in const fns so we must have the decoding loop here
166            // we minimized its size using a separate decoding function
167            let mut pos = 0;
168            loop {
169                if pos >= out.len() {
170                    break;
171                }
172                out[pos] = $crate::decode_byte(HEX, pos);
173                pos += 1
174            }
175            out
176        }
177    }
178}

192/// Creates a byte array const value from hex &str const value.

194/// Accepts const `&str` as an input. Refer to the crate documentation to learn more.

203/// Creates a byte array const value from hex &str const value.
204///
205/// Accepts const `&str` as an input. Refer to the crate documentation to learn more.
206#[cfg(rust_v_1_46)]
207#[macro_export]
208macro_rules! hex {
209    ($hex:expr) => {
210        {
211            // pass through const to force const eval
212            const TMP: [u8; $hex.len() / 2] = $crate::hex_impl!($hex);
213            // block usage in const context unless the feature is explicitly turned on
214            $crate::msrv_opt_in();
215            TMP
216        }
217    }
218}

220/// Creates a constant byte array of given name.
221///
222/// This is a convenience macro so that you don't have to write out the type. However it is
223/// intentionally not public - public constants must be explicitly typed to avoid accidental change
224/// of the type.
225///
226/// ## Example
227///
228/// ```
229/// use hex_lit::hex_const;
230/// // same as writing `const FOO: [u8; 2] = [0x00, 0xff];`
231/// hex_const!(FOO = "00ff");
232///
233/// assert_eq!(FOO, [0, 255]);
234/// ```
235#[cfg(feature = "rust_v_1_46")]
236#[cfg_attr(docsrs, doc(cfg(feature = "rust_v_1_46")))]
237#[macro_export]
238macro_rules! hex_const {
239    ($name:ident = $hex:expr) => {
240        const $name: [u8; $hex.len() / 2] = $crate::hex_impl!($hex);
241    }
242}
