9//! Merkle Block and Partial Merkle Tree.
10//!
11//! Support proofs that transaction(s) belong to a block.
12//!
13//! # Examples
14//!
15//! ```rust
16//! use bitcoin::hash_types::Txid;
17//! use bitcoin::hashes::hex::FromHex;
18//! use bitcoin::{Block, MerkleBlock};
19//!
20//! // Get the proof from a bitcoind by running in the terminal:
21//! // $ TXID="5a4ebf66822b0b2d56bd9dc64ece0bc38ee7844a23ff1d7320a88c5fdb2ad3e2"
22//! // $ bitcoin-cli gettxoutproof [\"$TXID\"]
23//! let mb_bytes = Vec::from_hex("01000000ba8b9cda965dd8e536670f9ddec10e53aab14b20bacad27b913719\
24//!     0000000000190760b278fe7b8565fda3b968b918d5fd997f993b23674c0af3b6fde300b38f33a5914ce6ed5b\
25//!     1b01e32f570200000002252bf9d75c4f481ebb6278d708257d1f12beb6dd30301d26c623f789b2ba6fc0e2d3\
26//!     2adb5f8ca820731dff234a84e78ec30bce4ec69dbd562d0b2b8266bf4e5a0105").unwrap();
27//! let mb: MerkleBlock = bitcoin::consensus::deserialize(&mb_bytes).unwrap();
28//!
29//! // Authenticate and extract matched transaction ids
30//! let mut matches: Vec<Txid> = vec![];
31//! let mut index: Vec<u32> = vec![];
32//! assert!(mb.extract_matches(&mut matches, &mut index).is_ok());
33//! assert_eq!(1, matches.len());
34//! assert_eq!(
35//!     "5a4ebf66822b0b2d56bd9dc64ece0bc38ee7844a23ff1d7320a88c5fdb2ad3e2".parse::<Txid>().unwrap(),
36//!     matches[0]
37//! );
38//! assert_eq!(1, index.len());
39//! assert_eq!(1, index[0]);
40//! ```

54/// Data structure that represents a block header paired to a partial merkle tree.
55///
56/// NOTE: This assumes that the given Block has *at least* 1 transaction. If the Block has 0 txs,
57/// it will hit an assertion.
58#[derive(PartialEq, Eq, Clone, Debug)]
59pub struct MerkleBlock {
60    /// The block header
61    pub header: block::Header,
62    /// Transactions making up a partial merkle tree
63    pub txn: PartialMerkleTree,
64}
65
66impl MerkleBlock {
67    /// Create a MerkleBlock from a block, that contains proofs for specific txids.
68    ///
69    /// The `block` is a full block containing the header and transactions and `match_txids` is a
70    /// function that returns true for the ids that should be included in the partial merkle tree.
71    ///
72    /// # Examples
73    ///
74    /// ```rust
75    /// use bitcoin::hash_types::Txid;
76    /// use bitcoin::hashes::hex::FromHex;
77    /// use bitcoin::{Block, MerkleBlock};
78    ///
79    /// // Block 80000
80    /// let block_bytes = Vec::from_hex("01000000ba8b9cda965dd8e536670f9ddec10e53aab14b20bacad2\
81    ///     7b9137190000000000190760b278fe7b8565fda3b968b918d5fd997f993b23674c0af3b6fde300b38f33\
82    ///     a5914ce6ed5b1b01e32f5702010000000100000000000000000000000000000000000000000000000000\
83    ///     00000000000000ffffffff0704e6ed5b1b014effffffff0100f2052a01000000434104b68a50eaa0287e\
84    ///     ff855189f949c1c6e5f58b37c88231373d8a59809cbae83059cc6469d65c665ccfd1cfeb75c6e8e19413\
85    ///     bba7fbff9bc762419a76d87b16086eac000000000100000001a6b97044d03da79c005b20ea9c0e1a6d9d\
86    ///     c12d9f7b91a5911c9030a439eed8f5000000004948304502206e21798a42fae0e854281abd38bacd1aee\
87    ///     d3ee3738d9e1446618c4571d1090db022100e2ac980643b0b82c0e88ffdfec6b64e3e6ba35e7ba5fdd7d\
88    ///     5d6cc8d25c6b241501ffffffff0100f2052a010000001976a914404371705fa9bd789a2fcd52d2c580b6\
89    ///     5d35549d88ac00000000").unwrap();
90    /// let block: Block = bitcoin::consensus::deserialize(&block_bytes).unwrap();
91    ///
92    /// // Create a merkle block containing a single transaction
93    /// let txid = "5a4ebf66822b0b2d56bd9dc64ece0bc38ee7844a23ff1d7320a88c5fdb2ad3e2".parse::<Txid>().unwrap();
94    /// let match_txids: Vec<Txid> = vec![txid].into_iter().collect();
95    /// let mb = MerkleBlock::from_block_with_predicate(&block, |t| match_txids.contains(t));
96    ///
97    /// // Authenticate and extract matched transaction ids
98    /// let mut matches: Vec<Txid> = vec![];
99    /// let mut index: Vec<u32> = vec![];
100    /// assert!(mb.extract_matches(&mut matches, &mut index).is_ok());
101    /// assert_eq!(txid, matches[0]);
102    /// ```
103    pub fn from_block_with_predicate<F>(block: &Block, match_txids: F) -> Self
104    where
105        F: Fn(&Txid) -> bool,
106    {
107        let block_txids: Vec<_> = block.txdata.iter().map(Transaction::txid).collect();
108        Self::from_header_txids_with_predicate(&block.header, &block_txids, match_txids)
109    }
110
111    /// Create a MerkleBlock from the block's header and txids, that contain proofs for specific txids.
112    ///
113    /// The `header` is the block header, `block_txids` is the full list of txids included in the block and
114    /// `match_txids` is a function that returns true for the ids that should be included in the partial merkle tree.
115    pub fn from_header_txids_with_predicate<F>(
116        header: &block::Header,
117        block_txids: &[Txid],
118        match_txids: F,
119    ) -> Self
120    where
121        F: Fn(&Txid) -> bool,
122    {
123        let matches: Vec<bool> = block_txids.iter().map(match_txids).collect();
124
125        let pmt = PartialMerkleTree::from_txids(block_txids, &matches);
126        MerkleBlock { header: *header, txn: pmt }
127    }
128
129    /// Extract the matching txid's represented by this partial merkle tree
130    /// and their respective indices within the partial tree.
131    /// returns Ok(()) on success, or error in case of failure
132    pub fn extract_matches(
133        &self,
134        matches: &mut Vec<Txid>,
135        indexes: &mut Vec<u32>,
136    ) -> Result<(), MerkleBlockError> {
137        let merkle_root = self.txn.extract_matches(matches, indexes)?;
138
139        if merkle_root.eq(&self.header.merkle_root) {
140            Ok(())
141        } else {
142            Err(MerkleRootMismatch)
143        }
144    }
145}

163/// Data structure that represents a partial merkle tree.
164///
165/// It represents a subset of the txid's of a known block, in a way that
166/// allows recovery of the list of txid's and the merkle root, in an
167/// authenticated way.
168///
169/// The encoding works as follows: we traverse the tree in depth-first order,
170/// storing a bit for each traversed node, signifying whether the node is the
171/// parent of at least one matched leaf txid (or a matched txid itself). In
172/// case we are at the leaf level, or this bit is 0, its merkle node hash is
173/// stored, and its children are not explored further. Otherwise, no hash is
174/// stored, but we recurse into both (or the only) child branch. During
175/// decoding, the same depth-first traversal is performed, consuming bits and
176/// hashes as they written during encoding.
177///
178/// The serialization is fixed and provides a hard guarantee about the
179/// encoded size:
180///
181///   SIZE <= 10 + ceil(32.25*N)
182///
183/// Where N represents the number of leaf nodes of the partial tree. N itself
184/// is bounded by:
185///
186///   N <= total_transactions
187///   N <= 1 + matched_transactions*tree_height
188///
189/// The serialization format:
190///  - uint32     total_transactions (4 bytes)
191///  - varint     number of hashes   (1-3 bytes)
192///  - uint256[]  hashes in depth-first order (<= 32*N bytes)
193///  - varint     number of bytes of flag bits (1-3 bytes)
194///  - byte[]     flag bits, packed per 8 in a byte, least significant bit first (<= 2*N-1 bits)
195/// The size constraints follow from this.
196#[derive(PartialEq, Eq, Clone, Debug)]
197pub struct PartialMerkleTree {
198    /// The total number of transactions in the block
199    num_transactions: u32,
200    /// node-is-parent-of-matched-txid bits
201    bits: Vec<bool>,
202    /// Transaction ids and internal hashes
203    hashes: Vec<TxMerkleNode>,
204}
205
206impl PartialMerkleTree {
207    /// Returns the total number of transactions in the block.
208    pub fn num_transactions(&self) -> u32 { self.num_transactions }
209
210    /// Returns the node-is-parent-of-matched-txid bits of the partial merkle tree.
211    pub fn bits(&self) -> &Vec<bool> { &self.bits }
212
213    /// Returns the transaction ids and internal hashes of the partial merkle tree.
214    pub fn hashes(&self) -> &Vec<TxMerkleNode> { &self.hashes }
215
216    /// Construct a partial merkle tree
217    /// The `txids` are the transaction hashes of the block and the `matches` is the contains flags
218    /// wherever a tx hash should be included in the proof.
219    ///
220    /// Panics when `txids` is empty or when `matches` has a different length
221    ///
222    /// # Examples
223    ///
224    /// ```rust
225    /// use bitcoin::hash_types::Txid;
226    /// use bitcoin::hashes::hex::FromHex;
227    /// use bitcoin::merkle_tree::{MerkleBlock, PartialMerkleTree};
228    ///
229    /// // Block 80000
230    /// let txids: Vec<Txid> = [
231    ///     "c06fbab289f723c6261d3030ddb6be121f7d2508d77862bb1e484f5cd7f92b25",
232    ///     "5a4ebf66822b0b2d56bd9dc64ece0bc38ee7844a23ff1d7320a88c5fdb2ad3e2",
233    /// ]
234    /// .iter()
235    /// .map(|hex| hex.parse::<Txid>().unwrap())
236    /// .collect();
237    ///
238    /// // Select the second transaction
239    /// let matches = vec![false, true];
240    /// let tree = PartialMerkleTree::from_txids(&txids, &matches);
241    /// assert!(tree.extract_matches(&mut vec![], &mut vec![]).is_ok());
242    /// ```
243    pub fn from_txids(txids: &[Txid], matches: &[bool]) -> Self {
244        // We can never have zero txs in a merkle block, we always need the coinbase tx
245        assert_ne!(txids.len(), 0);
246        assert_eq!(txids.len(), matches.len());
247
248        let mut pmt = PartialMerkleTree {
249            num_transactions: txids.len() as u32,
250            bits: Vec::with_capacity(txids.len()),
251            hashes: vec![],
252        };
253        // calculate height of tree
254        let mut height = 0;
255        while pmt.calc_tree_width(height) > 1 {
256            height += 1;
257        }
258        // traverse the partial tree
259        pmt.traverse_and_build(height, 0, txids, matches);
260        pmt
261    }
262
263    /// Extract the matching txid's represented by this partial merkle tree
264    /// and their respective indices within the partial tree.
265    /// returns the merkle root, or error in case of failure
266    pub fn extract_matches(
267        &self,
268        matches: &mut Vec<Txid>,
269        indexes: &mut Vec<u32>,
270    ) -> Result<TxMerkleNode, MerkleBlockError> {
271        matches.clear();
272        indexes.clear();
273        // An empty set will not work
274        if self.num_transactions == 0 {
275            return Err(NoTransactions);
276        };
277        // check for excessively high numbers of transactions
278        if self.num_transactions > MAX_BLOCK_WEIGHT / MIN_TRANSACTION_WEIGHT {
279            return Err(TooManyTransactions);
280        }
281        // there can never be more hashes provided than one for every txid
282        if self.hashes.len() as u32 > self.num_transactions {
283            return Err(TooManyHashes);
284        };
285        // there must be at least one bit per node in the partial tree, and at least one node per hash
286        if self.bits.len() < self.hashes.len() {
287            return Err(NotEnoughBits);
288        };
289        // calculate height of tree
290        let mut height = 0;
291        while self.calc_tree_width(height) > 1 {
292            height += 1;
293        }
294        // traverse the partial tree
295        let mut bits_used = 0u32;
296        let mut hash_used = 0u32;
297        let hash_merkle_root =
298            self.traverse_and_extract(height, 0, &mut bits_used, &mut hash_used, matches, indexes)?;
299        // Verify that all bits were consumed (except for the padding caused by
300        // serializing it as a byte sequence)
301        if (bits_used + 7) / 8 != (self.bits.len() as u32 + 7) / 8 {
302            return Err(NotAllBitsConsumed);
303        }
304        // Verify that all hashes were consumed
305        if hash_used != self.hashes.len() as u32 {
306            return Err(NotAllHashesConsumed);
307        }
308        Ok(TxMerkleNode::from_byte_array(hash_merkle_root.to_byte_array()))
309    }
310
311    /// Helper function to efficiently calculate the number of nodes at given height
312    /// in the merkle tree
313    #[inline]
314    fn calc_tree_width(&self, height: u32) -> u32 {
315        (self.num_transactions + (1 << height) - 1) >> height
316    }
317
318    /// Calculate the hash of a node in the merkle tree (at leaf level: the txid's themselves)
319    fn calc_hash(&self, height: u32, pos: u32, txids: &[Txid]) -> TxMerkleNode {
320        if height == 0 {
321            // Hash at height 0 is the txid itself
322            TxMerkleNode::from_byte_array(txids[pos as usize].to_byte_array())
323        } else {
324            // Calculate left hash
325            let left = self.calc_hash(height - 1, pos * 2, txids);
326            // Calculate right hash if not beyond the end of the array - copy left hash otherwise
327            let right = if pos * 2 + 1 < self.calc_tree_width(height - 1) {
328                self.calc_hash(height - 1, pos * 2 + 1, txids)
329            } else {
330                left
331            };
332            // Combine subhashes
333            PartialMerkleTree::parent_hash(left, right)
334        }
335    }
336
337    /// Recursive function that traverses tree nodes, storing the data as bits and hashes
338    fn traverse_and_build(&mut self, height: u32, pos: u32, txids: &[Txid], matches: &[bool]) {
339        // Determine whether this node is the parent of at least one matched txid
340        let mut parent_of_match = false;
341        let mut p = pos << height;
342        while p < (pos + 1) << height && p < self.num_transactions {
343            parent_of_match |= matches[p as usize];
344            p += 1;
345        }
346        // Store as flag bit
347        self.bits.push(parent_of_match);
348
349        if height == 0 || !parent_of_match {
350            // If at height 0, or nothing interesting below, store hash and stop
351            let hash = self.calc_hash(height, pos, txids);
352            self.hashes.push(hash);
353        } else {
354            // Otherwise, don't store any hash, but descend into the subtrees
355            self.traverse_and_build(height - 1, pos * 2, txids, matches);
356            if pos * 2 + 1 < self.calc_tree_width(height - 1) {
357                self.traverse_and_build(height - 1, pos * 2 + 1, txids, matches);
358            }
359        }
360    }
361
362    /// Recursive function that traverses tree nodes, consuming the bits and hashes produced by
363    /// TraverseAndBuild. It returns the hash of the respective node and its respective index.
364    fn traverse_and_extract(
365        &self,
366        height: u32,
367        pos: u32,
368        bits_used: &mut u32,
369        hash_used: &mut u32,
370        matches: &mut Vec<Txid>,
371        indexes: &mut Vec<u32>,
372    ) -> Result<TxMerkleNode, MerkleBlockError> {
373        if *bits_used as usize >= self.bits.len() {
374            return Err(BitsArrayOverflow);
375        }
376        let parent_of_match = self.bits[*bits_used as usize];
377        *bits_used += 1;
378        if height == 0 || !parent_of_match {
379            // If at height 0, or nothing interesting below, use stored hash and do not descend
380            if *hash_used as usize >= self.hashes.len() {
381                return Err(HashesArrayOverflow);
382            }
383            let hash = self.hashes[*hash_used as usize];
384            *hash_used += 1;
385            if height == 0 && parent_of_match {
386                // in case of height 0, we have a matched txid
387                matches.push(Txid::from_byte_array(hash.to_byte_array()));
388                indexes.push(pos);
389            }
390            Ok(hash)
391        } else {
392            // otherwise, descend into the subtrees to extract matched txids and hashes
393            let left = self.traverse_and_extract(
394                height - 1,
395                pos * 2,
396                bits_used,
397                hash_used,
398                matches,
399                indexes,
400            )?;
401            let right;
402            if pos * 2 + 1 < self.calc_tree_width(height - 1) {
403                right = self.traverse_and_extract(
404                    height - 1,
405                    pos * 2 + 1,
406                    bits_used,
407                    hash_used,
408                    matches,
409                    indexes,
410                )?;
411                if right == left {
412                    // The left and right branches should never be identical, as the transaction
413                    // hashes covered by them must each be unique.
414                    return Err(IdenticalHashesFound);
415                }
416            } else {
417                right = left;
418            }
419            // and combine them before returning
420            Ok(PartialMerkleTree::parent_hash(left, right))
421        }
422    }
423
424    /// Helper method to produce SHA256D(left + right)
425    fn parent_hash(left: TxMerkleNode, right: TxMerkleNode) -> TxMerkleNode {
426        let mut encoder = TxMerkleNode::engine();
427        left.consensus_encode(&mut encoder).expect("engines don't error");
428        right.consensus_encode(&mut encoder).expect("engines don't error");
429        TxMerkleNode::from_engine(encoder)
430    }
431}

433impl Encodable for PartialMerkleTree {
434    fn consensus_encode<W: io::Write + ?Sized>(&self, w: &mut W) -> Result<usize, io::Error> {
435        let ret = self.num_transactions.consensus_encode(w)? + self.hashes.consensus_encode(w)?;
436        let mut bytes: Vec<u8> = vec![0; (self.bits.len() + 7) / 8];
437        for p in 0..self.bits.len() {
438            bytes[p / 8] |= (self.bits[p] as u8) << (p % 8) as u8;
439        }
440        Ok(ret + bytes.consensus_encode(w)?)
441    }
442}

444impl Decodable for PartialMerkleTree {
445    fn consensus_decode<R: io::Read + ?Sized>(r: &mut R) -> Result<Self, encode::Error> {
446        let num_transactions: u32 = Decodable::consensus_decode(r)?;
447        let hashes: Vec<TxMerkleNode> = Decodable::consensus_decode(r)?;
448
449        let bytes: Vec<u8> = Decodable::consensus_decode(r)?;
450        let mut bits: Vec<bool> = vec![false; bytes.len() * 8];
451
452        for (p, bit) in bits.iter_mut().enumerate() {
453            *bit = (bytes[p / 8] & (1 << (p % 8) as u8)) != 0;
454        }
455        Ok(PartialMerkleTree { num_transactions, hashes, bits })
456    }
457}

459/// An error when verifying the merkle block.
460#[derive(Clone, PartialEq, Eq, Debug)]
461#[non_exhaustive]
462pub enum MerkleBlockError {
463    /// Merkle root in the header doesn't match to the root calculated from partial merkle tree.
464    MerkleRootMismatch,
465    /// Partial merkle tree contains no transactions.
466    NoTransactions,
467    /// There are too many transactions.
468    TooManyTransactions,
469    /// There are too many hashes
470    TooManyHashes,
471    /// There must be at least one bit per node in the partial tree,
472    /// and at least one node per hash
473    NotEnoughBits,
474    /// Not all bits were consumed
475    NotAllBitsConsumed,
476    /// Not all hashes were consumed
477    NotAllHashesConsumed,
478    /// Overflowed the bits array
479    BitsArrayOverflow,
480    /// Overflowed the hashes array
481    HashesArrayOverflow,
482    /// The left and right branches should never be identical
483    IdenticalHashesFound,
484}
485
486impl fmt::Display for MerkleBlockError {
487    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
488        use self::MerkleBlockError::*;
489
490        match *self {
491            MerkleRootMismatch => write!(f, "merkle header root doesn't match to the root calculated from the partial merkle tree"),
492            NoTransactions => write!(f, "partial merkle tree contains no transactions"),
493            TooManyTransactions => write!(f, "too many transactions"),
494            TooManyHashes => write!(f, "proof contains more hashes than transactions"),
495            NotEnoughBits => write!(f, "proof contains less bits than hashes"),
496            NotAllBitsConsumed => write!(f, "not all bit were consumed"),
497            NotAllHashesConsumed => write!(f, "not all hashes were consumed"),
498            BitsArrayOverflow => write!(f, "overflowed the bits array"),
499            HashesArrayOverflow => write!(f, "overflowed the hashes array"),
500            IdenticalHashesFound => write!(f, "found identical transaction hashes"),
501        }
502    }
503}

505#[cfg(feature = "std")]
506#[cfg_attr(docsrs, doc(cfg(feature = "std")))]
507impl std::error::Error for MerkleBlockError {
508    fn source(&self) -> Option<&(dyn std::error::Error + 'static)> {
509        use self::MerkleBlockError::*;
510
511        match *self {
512            MerkleRootMismatch | NoTransactions | TooManyTransactions | TooManyHashes
513            | NotEnoughBits | NotAllBitsConsumed | NotAllHashesConsumed | BitsArrayOverflow
514            | HashesArrayOverflow | IdenticalHashesFound => None,
515        }
516    }
517}

565    #[cfg(feature = "rand-std")]
566    fn pmt_test(tx_count: usize) {
567        use core::cmp::min;
568
569        use crate::merkle_tree;
570
571        let mut rng = thread_rng();
572        // Create some fake tx ids
573        let tx_ids = (1..=tx_count)
574            .map(|i| format!("{:064x}", i).parse::<Txid>().unwrap())
575            .collect::<Vec<_>>();
576
577        // Calculate the merkle root and height
578        let hashes = tx_ids.iter().map(|t| t.to_raw_hash());
579        let merkle_root_1: TxMerkleNode =
580            merkle_tree::calculate_root(hashes).expect("hashes is not empty").into();
581        let mut height = 1;
582        let mut ntx = tx_count;
583        while ntx > 1 {
584            ntx = (ntx + 1) / 2;
585            height += 1;
586        }
587
588        // Check with random subsets with inclusion chances 1, 1/2, 1/4, ..., 1/128
589        for att in 1..15 {
590            let mut matches = vec![false; tx_count];
591            let mut match_txid1 = vec![];
592            for j in 0..tx_count {
593                // Generate `att / 2` random bits
594                let rand_bits = match att / 2 {
595                    0 => 0,
596                    bits => rng.gen::<u64>() >> (64 - bits),
597                };
598                let include = rand_bits == 0;
599                matches[j] = include;
600
601                if include {
602                    match_txid1.push(tx_ids[j]);
603                };
604            }
605
606            // Build the partial merkle tree
607            let pmt1 = PartialMerkleTree::from_txids(&tx_ids, &matches);
608            let serialized = serialize(&pmt1);
609
610            // Verify PartialMerkleTree's size guarantees
611            let n = min(tx_count, 1 + match_txid1.len() * height);
612            assert!(serialized.len() <= 10 + (258 * n + 7) / 8);
613
614            // Deserialize into a tester copy
615            let pmt2: PartialMerkleTree =
616                deserialize(&serialized).expect("Could not deserialize own data");
617
618            // Extract merkle root and matched txids from copy
619            let mut match_txid2: Vec<Txid> = vec![];
620            let mut indexes = vec![];
621            let merkle_root_2 = pmt2
622                .extract_matches(&mut match_txid2, &mut indexes)
623                .expect("Could not extract matches");
624
625            // Check that it has the same merkle root as the original, and a valid one
626            assert_eq!(merkle_root_1, merkle_root_2);
627            assert_ne!(merkle_root_2, TxMerkleNode::all_zeros());
628
629            // check that it contains the matched transactions (in the same order!)
630            assert_eq!(match_txid1, match_txid2);
631
632            // check that random bit flips break the authentication
633            for _ in 0..4 {
634                let mut pmt3: PartialMerkleTree = deserialize(&serialized).unwrap();
635                pmt3.damage(&mut rng);
636                let mut match_txid3 = vec![];
637                let merkle_root_3 = pmt3.extract_matches(&mut match_txid3, &mut indexes).unwrap();
638                assert_ne!(merkle_root_3, merkle_root_1);
639            }
640        }
641    }

739    #[cfg(feature = "rand-std")]
740    impl PartialMerkleTree {
741        /// Flip one bit in one of the hashes - this should break the authentication
742        fn damage(&mut self, rng: &mut ThreadRng) {
743            let n = rng.gen_range(0..self.hashes.len());
744            let bit = rng.gen::<u8>();
745            let hashes = &mut self.hashes;
746            let mut hash = hashes[n].to_byte_array();
747            hash[(bit >> 3) as usize] ^= 1 << (bit & 7);
748            hashes[n] = TxMerkleNode::from_slice(&hash).unwrap();
749        }
750    }
