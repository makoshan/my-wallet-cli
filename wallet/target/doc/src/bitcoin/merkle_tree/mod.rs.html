29/// Calculates the merkle root of a list of *hashes*, inline (in place) in `hashes`.
30///
31/// In most cases, you'll want to use [`calculate_root`] instead. Please note, calling this function
32/// trashes the data in `hashes` (i.e. the `hashes` is left in an undefined state at conclusion of
33/// this method and should not be used again afterwards).
34///
35/// # Returns
36/// - `None` if `hashes` is empty. The merkle root of an empty tree of hashes is undefined.
37/// - `Some(hash)` if `hashes` contains one element. A single hash is by definition the merkle root.
38/// - `Some(merkle_root)` if length of `hashes` is greater than one.
39pub fn calculate_root_inline<T>(hashes: &mut [T]) -> Option<T>
40where
41    T: Hash + Encodable,
42    <T as Hash>::Engine: io::Write,
43{
44    match hashes.len() {
45        0 => None,
46        1 => Some(hashes[0]),
47        _ => Some(merkle_root_r(hashes)),
48    }
49}
50
51/// Calculates the merkle root of an iterator of *hashes*.
52///
53/// # Returns
54/// - `None` if `hashes` is empty. The merkle root of an empty tree of hashes is undefined.
55/// - `Some(hash)` if `hashes` contains one element. A single hash is by definition the merkle root.
56/// - `Some(merkle_root)` if length of `hashes` is greater than one.
57pub fn calculate_root<T, I>(mut hashes: I) -> Option<T>
58where
59    T: Hash + Encodable,
60    <T as Hash>::Engine: io::Write,
61    I: Iterator<Item = T>,
62{
63    let first = hashes.next()?;
64    let second = match hashes.next() {
65        Some(second) => second,
66        None => return Some(first),
67    };
68
69    let mut hashes = iter::once(first).chain(iter::once(second)).chain(hashes);
70
71    // We need a local copy to pass to `merkle_root_r`. It's more efficient to do the first loop of
72    // processing as we make the copy instead of copying the whole iterator.
73    let (min, max) = hashes.size_hint();
74    let mut alloc = Vec::with_capacity(max.unwrap_or(min) / 2 + 1);
75
76    while let Some(hash1) = hashes.next() {
77        // If the size is odd, use the last element twice.
78        let hash2 = hashes.next().unwrap_or(hash1);
79        let mut encoder = T::engine();
80        hash1.consensus_encode(&mut encoder).expect("in-memory writers don't error");
81        hash2.consensus_encode(&mut encoder).expect("in-memory writers don't error");
82        alloc.push(T::from_engine(encoder));
83    }
84
85    Some(merkle_root_r(&mut alloc))
86}
87
88// `hashes` must contain at least one hash.
89fn merkle_root_r<T>(hashes: &mut [T]) -> T
90where
91    T: Hash + Encodable,
92    <T as Hash>::Engine: io::Write,
93{
94    if hashes.len() == 1 {
95        return hashes[0];
96    }
97
98    for idx in 0..((hashes.len() + 1) / 2) {
99        let idx1 = 2 * idx;
100        let idx2 = min(idx1 + 1, hashes.len() - 1);
101        let mut encoder = T::engine();
102        hashes[idx1].consensus_encode(&mut encoder).expect("in-memory writers don't error");
103        hashes[idx2].consensus_encode(&mut encoder).expect("in-memory writers don't error");
104        hashes[idx] = T::from_engine(encoder);
105    }
106    let half_len = hashes.len() / 2 + hashes.len() % 2;
107
108    merkle_root_r(&mut hashes[0..half_len])
109}

111#[cfg(test)]
112mod tests {
113    use super::*;
114    use crate::blockdata::block::Block;
115    use crate::consensus::encode::deserialize;
116    use crate::hashes::sha256d;
117
118    #[test]
119    fn both_merkle_root_functions_return_the_same_result() {
120        // testnet block 000000000000045e0b1660b6445b5e5c5ab63c9a4f956be7e1e69be04fa4497b
121        let segwit_block = include_bytes!("../../tests/data/testnet_block_000000000000045e0b1660b6445b5e5c5ab63c9a4f956be7e1e69be04fa4497b.raw");
122        let block: Block = deserialize(&segwit_block[..]).expect("Failed to deserialize block");
123        assert!(block.check_merkle_root()); // Sanity check.
124
125        let hashes_iter = block.txdata.iter().map(|obj| obj.txid().to_raw_hash());
126
127        let mut hashes_array: [sha256d::Hash; 15] = [Hash::all_zeros(); 15];
128        for (i, hash) in hashes_iter.clone().enumerate() {
129            hashes_array[i] = hash;
130        }
131
132        let from_iter = calculate_root(hashes_iter);
133        let from_array = calculate_root_inline(&mut hashes_array);
134        assert_eq!(from_iter, from_array);
135    }
