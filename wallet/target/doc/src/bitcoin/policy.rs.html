4//! Bitcoin policy.
5//!
6//! This module exposes some constants and functions used in the reference
7//! implementation and which, as a consequence, define some network rules.
8//!
9//! # *Warning*
10//! While the constants present in this module are very unlikely to change, they do not define
11//! Bitcoin. As such they must not be relied upon as if they were consensus rules.
12//!
13//! These values were taken from bitcoind v0.21.1 (194b9b8792d9b0798fdb570b79fa51f1d1f5ebaf).
14//!
15
16use core::cmp;
17
18use super::blockdata::constants::{MAX_BLOCK_SIGOPS_COST, WITNESS_SCALE_FACTOR};
19
20/// Maximum weight of a transaction for it to be relayed by most nodes on the network
21pub const MAX_STANDARD_TX_WEIGHT: u32 = 400_000;
22
23/// Minimum non-witness size for a standard transaction (1 segwit input + 1 P2WPKH output = 82 bytes)
24pub const MIN_STANDARD_TX_NONWITNESS_SIZE: u32 = 82;
25
26/// Maximum number of sigops in a standard tx.
27pub const MAX_STANDARD_TX_SIGOPS_COST: u32 = MAX_BLOCK_SIGOPS_COST as u32 / 5;
28
29/// The minimum incremental *feerate* (despite the name), in sats per virtual kilobyte for RBF.
30pub const DEFAULT_INCREMENTAL_RELAY_FEE: u32 = 1_000;
31
32/// The number of bytes equivalent per signature operation. Affects transaction relay through the
33/// virtual size computation.
34pub const DEFAULT_BYTES_PER_SIGOP: u32 = 20;
35
36/// The minimum feerate, in sats per kilo-virtualbyte, for defining dust. An output is considered
37/// dust if spending it under this feerate would cost more in fee.
38pub const DUST_RELAY_TX_FEE: u32 = 3_000;
39
40/// Minimum feerate, in sats per virtual kilobyte, for a transaction to be relayed by most nodes on
41/// the network.
42pub const DEFAULT_MIN_RELAY_TX_FEE: u32 = 1_000;
43
44/// Default number of hours for an unconfirmed transaction to expire in most of the network nodes'
45/// mempools.
46pub const DEFAULT_MEMPOOL_EXPIRY: u32 = 336;
47
48/// The virtual transaction size, as computed by default by bitcoind node.
49pub fn get_virtual_tx_size(weight: i64, n_sigops: i64) -> i64 {
50    (cmp::max(weight, n_sigops * DEFAULT_BYTES_PER_SIGOP as i64) + WITNESS_SCALE_FACTOR as i64 - 1)
51        / WITNESS_SCALE_FACTOR as i64
52}
