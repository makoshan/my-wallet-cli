63///
64/// Work is a measure of how difficult it is to find a hash below a given [`Target`].

109/// A 256 bit integer representing target.
110///
111/// The SHA-256 hash of a block's header must be lower than or equal to the current target for the
112/// block to be accepted by the network. The lower the target, the more difficult it is to generate
113/// a block. (See also [`Work`].)
114///
115/// ref: <https://en.bitcoin.it/wiki/Target>
116#[derive(Copy, Clone, Debug, PartialEq, Eq, PartialOrd, Ord, Hash)]
117#[cfg_attr(feature = "serde", derive(Serialize, Deserialize))]
118#[cfg_attr(feature = "serde", serde(crate = "actual_serde"))]
119pub struct Target(U256);
120
121impl Target {
122    /// When parsing nBits, Bitcoin Core converts a negative target threshold into a target of zero.
123    pub const ZERO: Target = Target(U256::ZERO);
124    /// The maximum possible target.
125    ///
126    /// This value is used to calculate difficulty, which is defined as how difficult the current
127    /// target makes it to find a block relative to how difficult it would be at the highest
128    /// possible target. Remember highest target == lowest difficulty.
129    ///
130    /// ref: <https://en.bitcoin.it/wiki/Target>
131    // In Bitcoind this is ~(u256)0 >> 32 stored as a floating-point type so it gets truncated, hence
132    // the low 208 bits are all zero.
133    pub const MAX: Self = Target(U256(0xFFFF_u128 << (208 - 128), 0));
134
135    /// The maximum possible target (see [`Target::MAX`]).
136    ///
137    /// This is provided for consistency with Rust 1.41.1, newer code should use [`Target::MAX`].
138    pub const fn max_value() -> Self { Target::MAX }
139
140    /// Computes the [`Target`] value from a compact representation.
141    ///
142    /// ref: <https://developer.bitcoin.org/reference/block_chain.html#target-nbits>
143    pub fn from_compact(c: CompactTarget) -> Target {
144        let bits = c.0;
145        // This is a floating-point "compact" encoding originally used by
146        // OpenSSL, which satoshi put into consensus code, so we're stuck
147        // with it. The exponent needs to have 3 subtracted from it, hence
148        // this goofy decoding code. 3 is due to 3 bytes in the mantissa.
149        let (mant, expt) = {
150            let unshifted_expt = bits >> 24;
151            if unshifted_expt <= 3 {
152                ((bits & 0xFFFFFF) >> (8 * (3 - unshifted_expt as usize)), 0)
153            } else {
154                (bits & 0xFFFFFF, 8 * ((bits >> 24) - 3))
155            }
156        };
157
158        // The mantissa is signed but may not be negative.
159        if mant > 0x7F_FFFF {
160            Target::ZERO
161        } else {
162            Target(U256::from(mant) << expt)
163        }
164    }
165
166    /// Computes the compact value from a [`Target`] representation.
167    ///
168    /// The compact form is by definition lossy, this means that
169    /// `t == Target::from_compact(t.to_compact_lossy())` does not always hold.
170    pub fn to_compact_lossy(self) -> CompactTarget {
171        let mut size = (self.0.bits() + 7) / 8;
172        let mut compact = if size <= 3 {
173            (self.0.low_u64() << (8 * (3 - size))) as u32
174        } else {
175            let bn = self.0 >> (8 * (size - 3));
176            bn.low_u32()
177        };
178
179        if (compact & 0x0080_0000) != 0 {
180            compact >>= 8;
181            size += 1;
182        }
183
184        CompactTarget(compact | (size << 24))
185    }

187    /// Returns true if block hash is less than or equal to this [`Target`].
188    ///
189    /// Proof-of-work validity for a block requires the hash of the block to be less than or equal
190    /// to the target.
191    #[cfg_attr(all(test, mutate), mutate)]
192    pub fn is_met_by(&self, hash: BlockHash) -> bool {
193        use crate::hashes::Hash;
194        let hash = U256::from_le_bytes(hash.to_byte_array());
195        hash <= self.0
196    }

198    /// Converts this [`Target`] to [`Work`].
199    ///
200    /// "Work" is defined as the work done to mine a block with this target value (recorded in the
201    /// block header in compact form as nBits). This is not the same as the difficulty to mine a
202    /// block with this target (see `Self::difficulty`).
203    pub fn to_work(self) -> Work { Work(self.0.inverse()) }
204
205    /// Computes the popular "difficulty" measure for mining.
206    ///
207    /// Difficulty represents how difficult the current target makes it to find a block, relative to
208    /// how difficult it would be at the highest possible target (highest target == lowest difficulty).
209    ///
210    /// For example, a difficulty of 6,695,826 means that at a given hash rate, it will, on average,
211    /// take ~6.6 million times as long to find a valid block as it would at a difficulty of 1, or
212    /// alternatively, it will take, again on average, ~6.6 million times as many hashes to find a
213    /// valid block
214    ///
215    /// # Note
216    ///
217    /// Difficulty is calculated using the following algorithm `max / current` where [max] is
218    /// defined for the Bitcoin network and `current` is the current [target] for this block. As
219    /// such, a low target implies a high difficulty. Since [`Target`] is represented as a 256 bit
220    /// integer but `difficulty()` returns only 128 bits this means for targets below approximately
221    /// `0xffff_ffff_ffff_ffff_ffff_ffff` `difficulty()` will saturate at `u128::MAX`.
222    ///
223    /// [max]: Target::max
224    /// [target]: crate::blockdata::block::Header::target
225    #[cfg_attr(all(test, mutate), mutate)]
226    pub fn difficulty(&self) -> u128 {
227        let d = Target::MAX.0 / self.0;
228        d.saturating_to_u128()
229    }
230
231    /// Computes the popular "difficulty" measure for mining and returns a float value of f64.
232    ///
233    /// See [`difficulty`] for details.
234    ///
235    /// [`difficulty`]: Target::difficulty
236    #[cfg_attr(all(test, mutate), mutate)]
237    pub fn difficulty_float(&self) -> f64 { TARGET_MAX_F64 / self.0.to_f64() }
238
239    /// Computes the minimum valid [`Target`] threshold allowed for a block in which a difficulty
240    /// adjustment occurs.
241    ///
242    /// The difficulty can only decrease or increase by a factor of 4 max on each difficulty
243    /// adjustment period.
244    pub fn min_difficulty_transition_threshold(&self) -> Self { Self(self.0 >> 2) }
245
246    /// Computes the maximum valid [`Target`] threshold allowed for a block in which a difficulty
247    /// adjustment occurs.
248    ///
249    /// The difficulty can only decrease or increase by a factor of 4 max on each difficulty
250    /// adjustment period.
251    pub fn max_difficulty_transition_threshold(&self) -> Self { Self(self.0 << 2) }
252}

302/// Big-endian 256 bit integer type.
303// (high, low): u.0 contains the high bits, u.1 contains the low bits.
304#[derive(Copy, Clone, PartialEq, Eq, PartialOrd, Ord, Hash, Default)]
305struct U256(u128, u128);

307impl U256 {
308    const MAX: U256 =
309        U256(0xffff_ffff_ffff_ffff_ffff_ffff_ffff_ffff, 0xffff_ffff_ffff_ffff_ffff_ffff_ffff_ffff);
310
311    const ZERO: U256 = U256(0, 0);
312
313    const ONE: U256 = U256(0, 1);

315    /// Creates [`U256`] from a big-endian array of `u8`s.
316    #[cfg_attr(all(test, mutate), mutate)]
317    fn from_be_bytes(a: [u8; 32]) -> U256 {
318        let (high, low) = split_in_half(a);
319        let big = u128::from_be_bytes(high);
320        let little = u128::from_be_bytes(low);
321        U256(big, little)
322    }
323
324    /// Creates a [`U256`] from a little-endian array of `u8`s.
325    #[cfg_attr(all(test, mutate), mutate)]
326    fn from_le_bytes(a: [u8; 32]) -> U256 {
327        let (high, low) = split_in_half(a);
328        let little = u128::from_le_bytes(high);
329        let big = u128::from_le_bytes(low);
330        U256(big, little)
331    }

333    /// Converts `Self` to a big-endian array of `u8`s.
334    #[cfg_attr(all(test, mutate), mutate)]
335    fn to_be_bytes(self) -> [u8; 32] {
336        let mut out = [0; 32];
337        out[..16].copy_from_slice(&self.0.to_be_bytes());
338        out[16..].copy_from_slice(&self.1.to_be_bytes());
339        out
340    }

342    /// Converts `Self` to a little-endian array of `u8`s.
343    #[cfg_attr(all(test, mutate), mutate)]
344    fn to_le_bytes(self) -> [u8; 32] {
345        let mut out = [0; 32];
346        out[..16].copy_from_slice(&self.1.to_le_bytes());
347        out[16..].copy_from_slice(&self.0.to_le_bytes());
348        out
349    }

351    /// Calculates 2^256 / (x + 1) where x is a 256 bit unsigned integer.
352    ///
353    /// 2**256 / (x + 1) == ~x / (x + 1) + 1
354    ///
355    /// (Equation shamelessly stolen from bitcoind)
356    fn inverse(&self) -> U256 {
357        // We should never have a target/work of zero so this doesn't matter
358        // that much but we define the inverse of 0 as max.
359        if self.is_zero() {
360            return U256::MAX;
361        }
362        // We define the inverse of 1 as max.
363        if self.is_one() {
364            return U256::MAX;
365        }
366        // We define the inverse of max as 1.
367        if self.is_max() {
368            return U256::ONE;
369        }
370
371        let ret = !*self / self.wrapping_inc();
372        ret.wrapping_inc()
373    }

375    #[cfg_attr(all(test, mutate), mutate)]
376    fn is_zero(&self) -> bool { self.0 == 0 && self.1 == 0 }
377
378    #[cfg_attr(all(test, mutate), mutate)]
379    fn is_one(&self) -> bool { self.0 == 0 && self.1 == 1 }
380
381    #[cfg_attr(all(test, mutate), mutate)]
382    fn is_max(&self) -> bool { self.0 == u128::max_value() && self.1 == u128::max_value() }

384    /// Returns the low 32 bits.
385    fn low_u32(&self) -> u32 { self.low_u128() as u32 }
386
387    /// Returns the low 64 bits.
388    fn low_u64(&self) -> u64 { self.low_u128() as u64 }
389
390    /// Returns the low 128 bits.
391    fn low_u128(&self) -> u128 { self.1 }

393    /// Returns `self` as a `u128` saturating to `u128::MAX` if `self` is too big.
394    // Matagen gives false positive because >= and > both return u128::MAX
395    fn saturating_to_u128(&self) -> u128 {
396        if *self > U256::from(u128::max_value()) {
397            u128::max_value()
398        } else {
399            self.low_u128()
400        }
401    }

403    /// Returns the least number of bits needed to represent the number.
404    #[cfg_attr(all(test, mutate), mutate)]
405    fn bits(&self) -> u32 {
406        if self.0 > 0 {
407            256 - self.0.leading_zeros()
408        } else {
409            128 - self.1.leading_zeros()
410        }

413    /// Wrapping multiplication by `u64`.
414    ///
415    /// # Returns
416    ///
417    /// The multiplication result along with a boolean indicating whether an arithmetic overflow
418    /// occurred. If an overflow occurred then the wrapped value is returned.
419    // mutagen false positive: binop_bit, replace `|` with `^`
420    fn mul_u64(self, rhs: u64) -> (U256, bool) {
421        let mut carry: u128 = 0;
422        let mut split_le =
423            [self.1 as u64, (self.1 >> 64) as u64, self.0 as u64, (self.0 >> 64) as u64];
424
425        for word in &mut split_le {
426            // TODO: Use `carrying_mul` when stabilized: https://github.com/rust-lang/rust/issues/85532
427            // This will not overflow, for proof see https://github.com/rust-bitcoin/rust-bitcoin/pull/1496#issuecomment-1365938572
428            let n = carry + u128::from(rhs) * u128::from(*word);
429
430            *word = n as u64; // Intentional truncation, save the low bits
431            carry = n >> 64; // and carry the high bits.
432        }
433
434        let low = u128::from(split_le[0]) | u128::from(split_le[1]) << 64;
435        let high = u128::from(split_le[2]) | u128::from(split_le[3]) << 64;
436        (Self(high, low), carry != 0)
437    }

439    /// Calculates quotient and remainder.
440    ///
441    /// # Returns
442    ///
443    /// (quotient, remainder)
444    ///
445    /// # Panics
446    ///
447    /// If `rhs` is zero.
448    #[cfg_attr(all(test, mutate), mutate)]
449    fn div_rem(self, rhs: Self) -> (Self, Self) {
450        let mut sub_copy = self;
451        let mut shift_copy = rhs;
452        let mut ret = [0u128; 2];
453
454        let my_bits = self.bits();
455        let your_bits = rhs.bits();
456
457        // Check for division by 0
458        assert!(your_bits != 0, "attempted to divide {} by zero", self);
459
460        // Early return in case we are dividing by a larger number than us
461        if my_bits < your_bits {
462            return (U256::ZERO, sub_copy);
463        }
464
465        // Bitwise long division
466        let mut shift = my_bits - your_bits;
467        shift_copy = shift_copy << shift;
468        loop {
469            if sub_copy >= shift_copy {
470                ret[1 - (shift / 128) as usize] |= 1 << (shift % 128);
471                sub_copy = sub_copy.wrapping_sub(shift_copy);
472            }
473            shift_copy = shift_copy >> 1;
474            if shift == 0 {
475                break;
476            }
477            shift -= 1;
478        }
479
480        (U256(ret[0], ret[1]), sub_copy)
481    }

483    /// Calculates `self` + `rhs`
484    ///
485    /// Returns a tuple of the addition along with a boolean indicating whether an arithmetic
486    /// overflow would occur. If an overflow would have occurred then the wrapped value is returned.
487    #[must_use = "this returns the result of the operation, without modifying the original"]
488    #[cfg_attr(all(test, mutate), mutate)]
489    fn overflowing_add(self, rhs: Self) -> (Self, bool) {
490        let mut ret = U256::ZERO;
491        let mut ret_overflow = false;
492
493        let (high, overflow) = self.0.overflowing_add(rhs.0);
494        ret.0 = high;
495        ret_overflow |= overflow;
496
497        let (low, overflow) = self.1.overflowing_add(rhs.1);
498        ret.1 = low;
499        if overflow {
500            let (high, overflow) = ret.0.overflowing_add(1);
501            ret.0 = high;
502            ret_overflow |= overflow;
503        }
504
505        (ret, ret_overflow)
506    }

508    /// Calculates `self` - `rhs`
509    ///
510    /// Returns a tuple of the subtraction along with a boolean indicating whether an arithmetic
511    /// overflow would occur. If an overflow would have occurred then the wrapped value is returned.
512    #[must_use = "this returns the result of the operation, without modifying the original"]
513    #[cfg_attr(all(test, mutate), mutate)]
514    fn overflowing_sub(self, rhs: Self) -> (Self, bool) {
515        let ret = self.wrapping_add(!rhs).wrapping_add(Self::ONE);
516        let overflow = rhs > self;
517        (ret, overflow)
518    }

520    /// Calculates the multiplication of `self` and `rhs`.
521    ///
522    /// Returns a tuple of the multiplication along with a boolean
523    /// indicating whether an arithmetic overflow would occur. If an
524    /// overflow would have occurred then the wrapped value is returned.
525    #[must_use = "this returns the result of the operation, without modifying the original"]
526    #[cfg_attr(all(test, mutate), mutate)]
527    fn overflowing_mul(self, rhs: Self) -> (Self, bool) {
528        let mut ret = U256::ZERO;
529        let mut ret_overflow = false;
530
531        for i in 0..3 {
532            let to_mul = (rhs >> (64 * i)).low_u64();
533            let (mul_res, _) = self.mul_u64(to_mul);
534            ret = ret.wrapping_add(mul_res << (64 * i));
535        }
536
537        let to_mul = (rhs >> 192).low_u64();
538        let (mul_res, overflow) = self.mul_u64(to_mul);
539        ret_overflow |= overflow;
540        let (sum, overflow) = ret.overflowing_add(mul_res);
541        ret = sum;
542        ret_overflow |= overflow;
543
544        (ret, ret_overflow)
545    }

572    /// Returns `self` incremented by 1 wrapping around at the boundary of the type.
573    #[must_use = "this returns the result of the increment, without modifying the original"]
574    #[cfg_attr(all(test, mutate), mutate)]
575    fn wrapping_inc(&self) -> U256 {
576        let mut ret = U256::ZERO;
577
578        ret.1 = self.1.wrapping_add(1);
579        if ret.1 == 0 {
580            ret.0 = self.0.wrapping_add(1);
581        } else {
582            ret.0 = self.0;
583        }
584        ret
585    }
586
587    /// Panic-free bitwise shift-left; yields `self << mask(rhs)`, where `mask` removes any
588    /// high-order bits of `rhs` that would cause the shift to exceed the bitwidth of the type.
589    ///
590    /// Note that this is *not* the same as a rotate-left; the RHS of a wrapping shift-left is
591    /// restricted to the range of the type, rather than the bits shifted out of the LHS being
592    /// returned to the other end. We do not currently support `rotate_left`.
593    #[must_use = "this returns the result of the operation, without modifying the original"]
594    #[cfg_attr(all(test, mutate), mutate)]
595    fn wrapping_shl(self, rhs: u32) -> Self {
596        let shift = rhs & 0x000000ff;
597
598        let mut ret = U256::ZERO;
599        let word_shift = shift >= 128;
600        let bit_shift = shift % 128;
601
602        if word_shift {
603            ret.0 = self.1 << bit_shift
604        } else {
605            ret.0 = self.0 << bit_shift;
606            if bit_shift > 0 {
607                ret.0 += self.1.wrapping_shr(128 - bit_shift);
608            }
609            ret.1 = self.1 << bit_shift;
610        }
611        ret
612    }
613
614    /// Panic-free bitwise shift-right; yields `self >> mask(rhs)`, where `mask` removes any
615    /// high-order bits of `rhs` that would cause the shift to exceed the bitwidth of the type.
616    ///
617    /// Note that this is *not* the same as a rotate-right; the RHS of a wrapping shift-right is
618    /// restricted to the range of the type, rather than the bits shifted out of the LHS being
619    /// returned to the other end. We do not currently support `rotate_right`.
620    #[must_use = "this returns the result of the operation, without modifying the original"]
621    #[cfg_attr(all(test, mutate), mutate)]
622    fn wrapping_shr(self, rhs: u32) -> Self {
623        let shift = rhs & 0x000000ff;
624
625        let mut ret = U256::ZERO;
626        let word_shift = shift >= 128;
627        let bit_shift = shift % 128;
628
629        if word_shift {
630            ret.1 = self.0 >> bit_shift
631        } else {
632            ret.0 = self.0 >> bit_shift;
633            ret.1 = self.1 >> bit_shift;
634            if bit_shift > 0 {
635                ret.1 += self.0.wrapping_shl(128 - bit_shift);
636            }
637        }
638        ret
639    }

641    /// Format `self` to `f` as a decimal when value is known to be non-zero.
642    fn fmt_decimal(&self, f: &mut fmt::Formatter) -> fmt::Result {
643        const DIGITS: usize = 78; // U256::MAX has 78 base 10 digits.
644        const TEN: U256 = U256(0, 10);
645
646        let mut buf = [0_u8; DIGITS];
647        let mut i = DIGITS - 1; // We loop backwards.
648        let mut cur = *self;
649
650        loop {
651            let digit = (cur % TEN).low_u128() as u8; // Cast after rem 10 is lossless.
652            buf[i] = digit + b'0';
653            cur = cur / TEN;
654            if cur.is_zero() {
655                break;
656            }
657            i -= 1;
658        }
659        let s = core::str::from_utf8(&buf[i..]).expect("digits 0-9 are valid UTF8");
660        f.pad_integral(true, "", s)
661    }

663    /// Convert self to f64.
664    #[inline]
665    fn to_f64(self) -> f64 {
666        // Reference: https://blog.m-ou.se/floats/
667        // Step 1: Get leading zeroes
668        let leading_zeroes = 256 - self.bits();
669        // Step 2: Get msb to be farthest left bit
670        let left_aligned = self.wrapping_shl(leading_zeroes);
671        // Step 3: Shift msb to fit in lower 53 bits (128-53=75) to get the mantissa
672        // * Shifting the border of the 2 u128s to line up with mantissa and dropped bits
673        let middle_aligned = left_aligned >> 75;
674        // * This is the 53 most significant bits as u128
675        let mantissa = middle_aligned.0;
676        // Step 4: Dropped bits (except for last 75 bits) are all in the second u128.
677        // Bitwise OR the rest of the bits into it, preserving the highest bit,
678        // so we take the lower 75 bits of middle_aligned.1 and mix it in. (See blog for explanation)
679        let dropped_bits = middle_aligned.1 | (left_aligned.1 & 0x7FF_FFFF_FFFF_FFFF_FFFF);
680        // Step 5: The msb of the dropped bits has been preserved, and all other bits
681        // if any were set, would be set somewhere in the other 127 bits.
682        // If msb of dropped bits is 0, it is mantissa + 0
683        // If msb of dropped bits is 1, it is mantissa + 0 only if mantissa lowest bit is 0
684        // and other bits of the dropped bits are all 0.
685        // (This is why we only care if the other non-msb dropped bits are all 0 or not,
686        // so we can just OR them to make sure any bits show up somewhere.)
687        let mantissa =
688            (mantissa + ((dropped_bits - (dropped_bits >> 127 & !mantissa)) >> 127)) as u64;
689        // Step 6: Calculate the exponent
690        // If self is 0, exponent should be 0 (special meaning) and mantissa will end up 0 too
691        // Otherwise, (255 - n) + 1022 so it simplifies to 1277 - n
692        // 1023 and 1022 are the cutoffs for the exponent having the msb next to the decimal point
693        let exponent = if self == Self::ZERO { 0 } else { 1277 - leading_zeroes as u64 };
694        // Step 7: sign bit is always 0, exponent is shifted into place
695        // Use addition instead of bitwise OR to saturate the exponent if mantissa overflows
696        f64::from_bits((exponent << 52) + mantissa)
697    }
