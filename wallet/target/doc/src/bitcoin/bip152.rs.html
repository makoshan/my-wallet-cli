21/// A BIP-152 error
22#[derive(Clone, PartialEq, Eq, Debug, Copy, PartialOrd, Ord, Hash)]
23#[non_exhaustive]
24pub enum Error {
25    /// An unknown version number was used.
26    UnknownVersion,
27    /// The prefill slice provided was invalid.
28    InvalidPrefill,
29}
30
31impl fmt::Display for Error {
32    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
33        match *self {
34            Error::UnknownVersion => write!(f, "an unknown version number was used"),
35            Error::InvalidPrefill => write!(f, "the prefill slice provided was invalid"),
36        }
37    }
38}
39
40#[cfg(feature = "std")]
41#[cfg_attr(docsrs, doc(cfg(feature = "std")))]
42impl std::error::Error for Error {
43    fn source(&self) -> Option<&(dyn std::error::Error + 'static)> {
44        use self::Error::*;
45
46        match *self {
47            UnknownVersion | InvalidPrefill => None,
48        }
49    }
50}

52/// A [PrefilledTransaction] structure is used in [HeaderAndShortIds] to
53/// provide a list of a few transactions explicitly.
54#[derive(PartialEq, Eq, Clone, Debug, PartialOrd, Ord, Hash)]
55pub struct PrefilledTransaction {
56    /// The index of the transaction in the block.
57    ///
58    /// This field is differentially encoded relative to the previous
59    /// prefilled transaction as described as follows:
60    ///
61    /// > Several uses of CompactSize below are "differentially encoded". For
62    /// > these, instead of using raw indexes, the number encoded is the
63    /// > difference between the current index and the previous index, minus one.
64    /// > For example, a first index of 0 implies a real index of 0, a second
65    /// > index of 0 thereafter refers to a real index of 1, etc.
66    pub idx: u16,
67    /// The actual transaction.
68    pub tx: Transaction,
69}
70
71impl convert::AsRef<Transaction> for PrefilledTransaction {
72    fn as_ref(&self) -> &Transaction { &self.tx }
73}
74
75impl Encodable for PrefilledTransaction {
76    #[inline]
77    fn consensus_encode<S: io::Write + ?Sized>(&self, mut s: &mut S) -> Result<usize, io::Error> {
78        Ok(VarInt(self.idx as u64).consensus_encode(&mut s)? + self.tx.consensus_encode(&mut s)?)
79    }
80}
81
82impl Decodable for PrefilledTransaction {
83    #[inline]
84    fn consensus_decode<D: io::Read + ?Sized>(
85        mut d: &mut D,
86    ) -> Result<PrefilledTransaction, encode::Error> {
87        let idx = VarInt::consensus_decode(&mut d)?.0;
88        let idx = u16::try_from(idx)
89            .map_err(|_| encode::Error::ParseFailed("BIP152 prefilled tx index out of bounds"))?;
90        let tx = Transaction::consensus_decode(&mut d)?;
91        Ok(PrefilledTransaction { idx, tx })
92    }
93}

95/// Short transaction IDs are used to represent a transaction without sending a full 256-bit hash.
96#[derive(PartialEq, Eq, Clone, Copy, Hash, Default, PartialOrd, Ord)]
97pub struct ShortId([u8; 6]);
98impl_array_newtype!(ShortId, u8, 6);
99impl_bytes_newtype!(ShortId, 6);
100
101impl ShortId {
102    /// Calculate the SipHash24 keys used to calculate short IDs.
103    pub fn calculate_siphash_keys(header: &block::Header, nonce: u64) -> (u64, u64) {
104        // 1. single-SHA256 hashing the block header with the nonce appended (in little-endian)
105        let h = {
106            let mut engine = sha256::Hash::engine();
107            header.consensus_encode(&mut engine).expect("engines don't error");
108            nonce.consensus_encode(&mut engine).expect("engines don't error");
109            sha256::Hash::from_engine(engine)
110        };
111
112        // 2. Running SipHash-2-4 with the input being the transaction ID and the keys (k0/k1)
113        // set to the first two little-endian 64-bit integers from the above hash, respectively.
114        (
115            u64::from_le_bytes(h[0..8].try_into().expect("8 byte slice")),
116            u64::from_le_bytes(h[8..16].try_into().expect("8 byte slice")),
117        )
118    }
119
120    /// Calculate the short ID with the given (w)txid and using the provided SipHash keys.
121    pub fn with_siphash_keys<T: AsRef<[u8]>>(txid: &T, siphash_keys: (u64, u64)) -> ShortId {
122        // 2. Running SipHash-2-4 with the input being the transaction ID and the keys (k0/k1)
123        // set to the first two little-endian 64-bit integers from the above hash, respectively.
124        let hash = siphash24::Hash::hash_with_keys(siphash_keys.0, siphash_keys.1, txid.as_ref());
125
126        // 3. Dropping the 2 most significant bytes from the SipHash output to make it 6 bytes.
127        let mut id = ShortId([0; 6]);
128        id.0.copy_from_slice(&hash[0..6]);
129        id
130    }
131}
132
133impl Encodable for ShortId {
134    #[inline]
135    fn consensus_encode<S: io::Write + ?Sized>(&self, s: &mut S) -> Result<usize, io::Error> {
136        self.0.consensus_encode(s)
137    }
138}
139
140impl Decodable for ShortId {
141    #[inline]
142    fn consensus_decode<D: io::Read + ?Sized>(d: &mut D) -> Result<ShortId, encode::Error> {
143        Ok(ShortId(Decodable::consensus_decode(d)?))
144    }
145}

147/// A [HeaderAndShortIds] structure is used to relay a block header, the short
148/// transactions IDs used for matching already-available transactions, and a
149/// select few transactions which we expect a peer may be missing.
150#[derive(PartialEq, Eq, Clone, Debug, PartialOrd, Ord, Hash)]
151pub struct HeaderAndShortIds {
152    /// The header of the block being provided.
153    pub header: block::Header,
154    ///  A nonce for use in short transaction ID calculations.
155    pub nonce: u64,
156    ///  The short transaction IDs calculated from the transactions
157    ///  which were not provided explicitly in prefilled_txs.
158    pub short_ids: Vec<ShortId>,
159    ///  Used to provide the coinbase transaction and a select few
160    ///  which we expect a peer may be missing.
161    pub prefilled_txs: Vec<PrefilledTransaction>,
162}
163impl_consensus_encoding!(HeaderAndShortIds, header, nonce, short_ids, prefilled_txs);
164
165impl HeaderAndShortIds {
166    /// Create a new [HeaderAndShortIds] from a full block.
167    ///
168    /// The version number must be either 1 or 2.
169    ///
170    /// The `prefill` slice indicates which transactions should be prefilled in
171    /// the block. It should contain the indexes in the block of the txs to
172    /// prefill. It must be ordered. 0 should not be included as the
173    /// coinbase tx is always prefilled.
174    ///
175    /// > Nodes SHOULD NOT use the same nonce across multiple different blocks.
176    pub fn from_block(
177        block: &Block,
178        nonce: u64,
179        version: u32,
180        mut prefill: &[usize],
181    ) -> Result<HeaderAndShortIds, Error> {
182        if version != 1 && version != 2 {
183            return Err(Error::UnknownVersion);
184        }
185
186        let siphash_keys = ShortId::calculate_siphash_keys(&block.header, nonce);
187
188        let mut prefilled = Vec::with_capacity(prefill.len() + 1); // +1 for coinbase tx
189        let mut short_ids = Vec::with_capacity(block.txdata.len() - prefill.len());
190        let mut last_prefill = 0;
191        for (idx, tx) in block.txdata.iter().enumerate() {
192            // Check if we should prefill this tx.
193            let prefill_tx = if prefill.first() == Some(&idx) {
194                prefill = &prefill[1..];
195                true
196            } else {
197                idx == 0 // Always prefill coinbase.
198            };
199
200            if prefill_tx {
201                let diff_idx = idx - last_prefill;
202                last_prefill = idx + 1;
203                prefilled.push(PrefilledTransaction {
204                    idx: diff_idx as u16,
205                    tx: match version {
206                        // >  As encoded in "tx" messages sent in response to getdata MSG_TX
207                        1 => {
208                            // strip witness for version 1
209                            let mut no_witness = tx.clone();
210                            no_witness.input.iter_mut().for_each(|i| i.witness.clear());
211                            no_witness
212                        }
213                        // > Transactions inside cmpctblock messages (both those used as direct
214                        // > announcement and those in response to getdata) and in blocktxn should
215                        // > include witness data, using the same format as responses to getdata
216                        // > MSG_WITNESS_TX, specified in BIP144.
217                        2 => tx.clone(),
218                        _ => unreachable!(),
219                    },
220                });
221            } else {
222                short_ids.push(ShortId::with_siphash_keys(
223                    &match version {
224                        1 => tx.txid().to_raw_hash(),
225                        2 => tx.wtxid().to_raw_hash(),
226                        _ => unreachable!(),
227                    },
228                    siphash_keys,
229                ));
230            }
231        }
232
233        if !prefill.is_empty() {
234            return Err(Error::InvalidPrefill);
235        }
236
237        Ok(HeaderAndShortIds {
238            header: block.header,
239            nonce,
240            // Provide coinbase prefilled.
241            prefilled_txs: prefilled,
242            short_ids,
243        })
244    }
245}

247/// A [BlockTransactionsRequest] structure is used to list transaction indexes
248/// in a block being requested.
249#[derive(PartialEq, Eq, Clone, Debug, PartialOrd, Ord, Hash)]
250pub struct BlockTransactionsRequest {
251    ///  The blockhash of the block which the transactions being requested are in.
252    pub block_hash: BlockHash,
253    ///  The indexes of the transactions being requested in the block.
254    ///
255    ///  Warning: Encoding panics with [`u64::MAX`] values. See [`BlockTransactionsRequest::consensus_encode()`]
256    pub indexes: Vec<u64>,
257}
258
259impl Encodable for BlockTransactionsRequest {
260    /// # Panics
261    ///
262    /// Panics if the index overflows [`u64::MAX`]. This happens when [`BlockTransactionsRequest::indexes`]
263    /// contains an entry with the value [`u64::MAX`] as `u64` overflows during differential encoding.
264    fn consensus_encode<S: io::Write + ?Sized>(&self, mut s: &mut S) -> Result<usize, io::Error> {
265        let mut len = self.block_hash.consensus_encode(&mut s)?;
266        // Manually encode indexes because they are differentially encoded VarInts.
267        len += VarInt(self.indexes.len() as u64).consensus_encode(&mut s)?;
268        let mut last_idx = 0;
269        for idx in &self.indexes {
270            len += VarInt(*idx - last_idx).consensus_encode(&mut s)?;
271            last_idx = *idx + 1; // can panic here
272        }
273        Ok(len)
274    }
275}
276
277impl Decodable for BlockTransactionsRequest {
278    fn consensus_decode<D: io::Read + ?Sized>(
279        mut d: &mut D,
280    ) -> Result<BlockTransactionsRequest, encode::Error> {
281        Ok(BlockTransactionsRequest {
282            block_hash: BlockHash::consensus_decode(&mut d)?,
283            indexes: {
284                // Manually decode indexes because they are differentially encoded VarInts.
285                let nb_indexes = VarInt::consensus_decode(&mut d)?.0 as usize;
286
287                // Since the number of indices ultimately represent transactions,
288                // we can limit the number of indices to the maximum number of
289                // transactions that would be allowed in a vector.
290                let byte_size = (nb_indexes)
291                    .checked_mul(mem::size_of::<Transaction>())
292                    .ok_or(encode::Error::ParseFailed("Invalid length"))?;
293                if byte_size > encode::MAX_VEC_SIZE {
294                    return Err(encode::Error::OversizedVectorAllocation {
295                        requested: byte_size,
296                        max: encode::MAX_VEC_SIZE,
297                    });
298                }
299
300                let mut indexes = Vec::with_capacity(nb_indexes);
301                let mut last_index: u64 = 0;
302                for _ in 0..nb_indexes {
303                    let differential: VarInt = Decodable::consensus_decode(&mut d)?;
304                    last_index = match last_index.checked_add(differential.0) {
305                        Some(r) => r,
306                        None => return Err(encode::Error::ParseFailed("block index overflow")),
307                    };
308                    indexes.push(last_index);
309                    last_index = match last_index.checked_add(1) {
310                        Some(r) => r,
311                        None => return Err(encode::Error::ParseFailed("block index overflow")),
312                    };
313                }
314                indexes
315            },
316        })
317    }
318}

320/// A transaction index is requested that is out of range from the
321/// corresponding block.
322#[derive(Clone, PartialEq, Eq, Debug, Copy, PartialOrd, Ord, Hash)]
323pub struct TxIndexOutOfRangeError(u64);
324
325impl fmt::Display for TxIndexOutOfRangeError {
326    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
327        write!(
328            f,
329            "a transaction index is requested that is \
330            out of range from the corresponding block: {}",
331            self.0,
332        )
333    }
334}

339/// A [BlockTransactions] structure is used to provide some of the transactions
340/// in a block, as requested.
341#[derive(PartialEq, Eq, Clone, Debug, PartialOrd, Ord, Hash)]
342pub struct BlockTransactions {
343    ///  The blockhash of the block which the transactions being provided are in.
344    pub block_hash: BlockHash,
345    ///  The transactions provided.
346    pub transactions: Vec<Transaction>,
347}
348impl_consensus_encoding!(BlockTransactions, block_hash, transactions);
349
350impl BlockTransactions {
351    /// Construct a [BlockTransactions] from a [BlockTransactionsRequest] and
352    /// the corresponsing full [Block] by providing all requested transactions.
353    pub fn from_request(
354        request: &BlockTransactionsRequest,
355        block: &Block,
356    ) -> Result<BlockTransactions, TxIndexOutOfRangeError> {
357        Ok(BlockTransactions {
358            block_hash: request.block_hash,
359            transactions: {
360                let mut txs = Vec::with_capacity(request.indexes.len());
361                for idx in &request.indexes {
362                    if *idx >= block.txdata.len() as u64 {
363                        return Err(TxIndexOutOfRangeError(*idx));
364                    }
365                    txs.push(block.txdata[*idx as usize].clone());
366                }
367                txs
368            },
369        })
370    }

446    #[test]
447    fn test_getblocktx_differential_encoding_de_and_serialization() {
448        let testcases = vec![
449            // differentially encoded VarInts, indicies
450            (vec![4, 0, 5, 1, 10], vec![0, 6, 8, 19]),
451            (vec![1, 0], vec![0]),
452            (vec![5, 0, 0, 0, 0, 0], vec![0, 1, 2, 3, 4]),
453            (vec![3, 1, 1, 1], vec![1, 3, 5]),
454            (vec![3, 0, 0, 253, 0, 1], vec![0, 1, 258]), // .., 253, 0, 1] == VarInt(256)
455        ];
456        let deser_errorcases = vec![
457            vec![2, 255, 254, 255, 255, 255, 255, 255, 255, 255, 0], // .., 255, 254, .., 255] == VarInt(u64::MAX-1)
458            vec![1, 255, 255, 255, 255, 255, 255, 255, 255, 255], // .., 255, 255, .., 255] == VarInt(u64::MAX)
459        ];
460        for testcase in testcases {
461            {
462                // test deserialization
463                let mut raw: Vec<u8> = [0u8; 32].to_vec();
464                raw.extend(testcase.0.clone());
465                let btr: BlockTransactionsRequest = deserialize(&raw.to_vec()).unwrap();
466                assert_eq!(testcase.1, btr.indexes);
467            }
468            {
469                // test serialization
470                let raw: Vec<u8> = serialize(&BlockTransactionsRequest {
471                    block_hash: Hash::all_zeros(),
472                    indexes: testcase.1,
473                });
474                let mut expected_raw: Vec<u8> = [0u8; 32].to_vec();
475                expected_raw.extend(testcase.0);
476                assert_eq!(expected_raw, raw);
477            }
478        }
479        for errorcase in deser_errorcases {
480            {
481                // test that we return Err() if deserialization fails (and don't panic)
482                let mut raw: Vec<u8> = [0u8; 32].to_vec();
483                raw.extend(errorcase);
484                assert!(deserialize::<BlockTransactionsRequest>(&raw.to_vec()).is_err());
485            }
486        }
487    }
