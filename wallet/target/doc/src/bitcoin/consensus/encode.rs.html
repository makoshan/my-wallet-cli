4//! Bitcoin consensus-encodable types.
5//!
6//! This is basically a replacement of the `Encodable` trait which does
7//! normalization of endianness etc., to ensure that the encoding matches
8//! the network consensus encoding.
9//!
10//! Essentially, anything that must go on the _disk_ or _network_ must be
11//! encoded using the `Encodable` trait, since this data must be the same for
12//! all systems. Any data going to the _user_ e.g., over JSONRPC, should use the
13//! ordinary `Encodable` trait. (This should also be the same across systems, of
14//! course, but has some critical differences from the network format e.g.,
15//! scripts come with an opcode decode, hashes are big-endian, numbers are
16//! typically big-endian decimals, etc.)
17//!

36/// Encoding error.
37#[derive(Debug)]
38#[non_exhaustive]
39pub enum Error {
40    /// And I/O error.
41    Io(io::Error),
42    /// Tried to allocate an oversized vector.
43    OversizedVectorAllocation {
44        /// The capacity requested.
45        requested: usize,
46        /// The maximum capacity.
47        max: usize,
48    },
49    /// Checksum was invalid.
50    InvalidChecksum {
51        /// The expected checksum.
52        expected: [u8; 4],
53        /// The invalid checksum.
54        actual: [u8; 4],
55    },
56    /// VarInt was encoded in a non-minimal way.
57    NonMinimalVarInt,
58    /// Parsing error.
59    ParseFailed(&'static str),
60    /// Unsupported Segwit flag.
61    UnsupportedSegwitFlag(u8),
62}
63
64impl fmt::Display for Error {
65    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
66        match *self {
67            Error::Io(ref e) => write_err!(f, "IO error"; e),
68            Error::OversizedVectorAllocation { requested: ref r, max: ref m } => write!(f,
69                "allocation of oversized vector: requested {}, maximum {}", r, m),
70            Error::InvalidChecksum { expected: ref e, actual: ref a } => write!(f,
71                "invalid checksum: expected {:x}, actual {:x}", e.as_hex(), a.as_hex()),
72            Error::NonMinimalVarInt => write!(f, "non-minimal varint"),
73            Error::ParseFailed(ref s) => write!(f, "parse failed: {}", s),
74            Error::UnsupportedSegwitFlag(ref swflag) => write!(f,
75                "unsupported segwit version: {}", swflag),
76        }
77    }
78}

82impl std::error::Error for Error {
83    fn source(&self) -> Option<&(dyn std::error::Error + 'static)> {
84        use self::Error::*;
85
86        match self {
87            Io(e) => Some(e),
88            OversizedVectorAllocation { .. }
89            | InvalidChecksum { .. }
90            | NonMinimalVarInt
91            | ParseFailed(_)
92            | UnsupportedSegwitFlag(_) => None,
93        }
94    }
95}

104/// Encodes an object into a vector.
105pub fn serialize<T: Encodable + ?Sized>(data: &T) -> Vec<u8> {
106    let mut encoder = Vec::new();
107    let len = data.consensus_encode(&mut encoder).expect("in-memory writers don't error");
108    debug_assert_eq!(len, encoder.len());
109    encoder
110}
111
112/// Encodes an object into a hex-encoded string.
113pub fn serialize_hex<T: Encodable + ?Sized>(data: &T) -> String {
114    serialize(data).to_lower_hex_string()
115}

117/// Deserializes an object from a vector, will error if said deserialization
118/// doesn't consume the entire vector.
119pub fn deserialize<T: Decodable>(data: &[u8]) -> Result<T, Error> {
120    let (rv, consumed) = deserialize_partial(data)?;
121
122    // Fail if data are not consumed entirely.
123    if consumed == data.len() {
124        Ok(rv)
125    } else {
126        Err(Error::ParseFailed("data not consumed entirely when explicitly deserializing"))
127    }
128}
129
130/// Deserializes an object from a vector, but will not report an error if said deserialization
131/// doesn't consume the entire vector.
132pub fn deserialize_partial<T: Decodable>(data: &[u8]) -> Result<(T, usize), Error> {
133    let mut decoder = Cursor::new(data);
134    let rv = Decodable::consensus_decode_from_finite_reader(&mut decoder)?;
135    let consumed = decoder.position() as usize;
136
137    Ok((rv, consumed))
138}

195macro_rules! encoder_fn {
196    ($name:ident, $val_type:ty) => {
197        #[inline]
198        fn $name(&mut self, v: $val_type) -> Result<(), io::Error> {
199            self.write_all(&v.to_le_bytes())
200        }
201    }
202}
203
204macro_rules! decoder_fn {
205    ($name:ident, $val_type:ty, $byte_len: expr) => {
206        #[inline]
207        fn $name(&mut self) -> Result<$val_type, Error> {
208            let mut val = [0; $byte_len];
209            self.read_exact(&mut val[..]).map_err(Error::Io)?;
210            Ok(<$val_type>::from_le_bytes(val))
211        }
212    }
213}

215impl<W: io::Write + ?Sized> WriteExt for W {
216    encoder_fn!(emit_u64, u64);
217    encoder_fn!(emit_u32, u32);
218    encoder_fn!(emit_u16, u16);
219    encoder_fn!(emit_i64, i64);
220    encoder_fn!(emit_i32, i32);
221    encoder_fn!(emit_i16, i16);
222
223    #[inline]
224    fn emit_i8(&mut self, v: i8) -> Result<(), io::Error> {
225        self.write_all(&[v as u8])
226    }
227    #[inline]
228    fn emit_u8(&mut self, v: u8) -> Result<(), io::Error> {
229        self.write_all(&[v])
230    }
231    #[inline]
232    fn emit_bool(&mut self, v: bool) -> Result<(), io::Error> {
233        self.write_all(&[v as u8])
234    }
235    #[inline]
236    fn emit_slice(&mut self, v: &[u8]) -> Result<(), io::Error> {
237        self.write_all(v)
238    }
239}

241impl<R: Read + ?Sized> ReadExt for R {
242    decoder_fn!(read_u64, u64, 8);
243    decoder_fn!(read_u32, u32, 4);
244    decoder_fn!(read_u16, u16, 2);
245    decoder_fn!(read_i64, i64, 8);
246    decoder_fn!(read_i32, i32, 4);
247    decoder_fn!(read_i16, i16, 2);
248
249    #[inline]
250    fn read_u8(&mut self) -> Result<u8, Error> {
251        let mut slice = [0u8; 1];
252        self.read_exact(&mut slice)?;
253        Ok(slice[0])
254    }
255    #[inline]
256    fn read_i8(&mut self) -> Result<i8, Error> {
257        let mut slice = [0u8; 1];
258        self.read_exact(&mut slice)?;
259        Ok(slice[0] as i8)
260    }
261    #[inline]
262    fn read_bool(&mut self) -> Result<bool, Error> {
263        ReadExt::read_i8(self).map(|bit| bit != 0)
264    }
265    #[inline]
266    fn read_slice(&mut self, slice: &mut [u8]) -> Result<(), Error> {
267        self.read_exact(slice).map_err(Error::Io)
268    }

271/// Maximum size, in bytes, of a vector we are allowed to decode.
272pub const MAX_VEC_SIZE: usize = 4_000_000;
273
274/// Data which can be encoded in a consensus-consistent way.
275pub trait Encodable {
276    /// Encodes an object with a well-defined format.
277    ///
278    /// # Returns
279    ///
280    /// The number of bytes written on success. The only errors returned are errors propagated from
281    /// the writer.
282    fn consensus_encode<W: io::Write + ?Sized>(&self, writer: &mut W) -> Result<usize, io::Error>;
283}
284
285/// Data which can be encoded in a consensus-consistent way.
286pub trait Decodable: Sized {
287    /// Decode `Self` from a size-limited reader.
288    ///
289    /// Like `consensus_decode` but relies on the reader being limited in the amount of data it
290    /// returns, e.g. by being wrapped in [`std::io::Take`].
291    ///
292    /// Failling to obide to this requirement might lead to memory exhaustion caused by malicious
293    /// inputs.
294    ///
295    /// Users should default to `consensus_decode`, but when data to be decoded is already in a byte
296    /// vector of a limited size, calling this function directly might be marginally faster (due to
297    /// avoiding extra checks).
298    ///
299    /// ### Rules for trait implementations
300    ///
301    /// * Simple types that that have a fixed size (own and member fields), don't have to overwrite
302    ///   this method, or be concern with it.
303    /// * Types that deserialize using externally provided length should implement it:
304    ///   * Make `consensus_decode` forward to `consensus_decode_bytes_from_finite_reader` with the
305    ///     reader wrapped by `Take`. Failure to do so, without other forms of memory exhaustion
306    ///     protection might lead to resource exhaustion vulnerability.
307    ///   * Put a max cap on things like `Vec::with_capacity` to avoid oversized allocations, and
308    ///     rely on the reader running out of data, and collections reallocating on a legitimately
309    ///     oversized input data, instead of trying to enforce arbitrary length limits.
310    /// * Types that contain other types that implement custom
311    ///   `consensus_decode_from_finite_reader`, should also implement it applying same rules, and
312    ///   in addition make sure to call `consensus_decode_from_finite_reader` on all members, to
313    ///   avoid creating redundant `Take` wrappers. Failure to do so might result only in a tiny
314    ///   performance hit.
315    #[inline]
316    fn consensus_decode_from_finite_reader<R: io::Read + ?Sized>(reader: &mut R) -> Result<Self, Error> {
317        // This method is always strictly less general than, `consensus_decode`, so it's safe and
318        // make sense to default to just calling it. This way most types, that don't care about
319        // protecting against resource exhaustion due to malicious input, can just ignore it.
320        Self::consensus_decode(reader)
321    }
322
323    /// Decode an object with a well-defined format.
324    ///
325    /// This is the method that should be implemented for a typical, fixed sized type
326    /// implementing this trait. Default implementation is wrapping the reader
327    /// in [`crate::io::Take`] to limit the input size to [`MAX_VEC_SIZE`], and forwards the call to
328    /// [`Self::consensus_decode_from_finite_reader`], which is convenient
329    /// for types that override [`Self::consensus_decode_from_finite_reader`]
330    /// instead.
331    #[inline]
332    fn consensus_decode<R: io::Read + ?Sized>(reader: &mut R) -> Result<Self, Error> {
333        Self::consensus_decode_from_finite_reader(reader.take(MAX_VEC_SIZE as u64).by_ref())
334    }
335}

373#[allow(clippy::len_without_is_empty)] // VarInt has on concept of 'is_empty'.
374impl VarInt {
375    crate::internal_macros::maybe_const_fn! {
376        /// Gets the length of this VarInt when encoded.
377        ///
378        /// *Important: this method is only `const` in Rust 1.46 or higher!*
379        ///
380        /// Returns 1 for 0..=0xFC, 3 for 0xFD..=(2^16-1), 5 for 0x10000..=(2^32-1),
381        /// and 9 otherwise.
382        #[inline]
383        pub fn len(&self) -> usize {
384            match self.0 {
385                0..=0xFC             => { 1 }
386                0xFD..=0xFFFF        => { 3 }
387                0x10000..=0xFFFFFFFF => { 5 }
388                _                    => { 9 }
389            }
390        }
391    }
392}

394impl Encodable for VarInt {
395    #[inline]
396    fn consensus_encode<W: io::Write + ?Sized>(&self, w: &mut W) -> Result<usize, io::Error> {
397        match self.0 {
398            0..=0xFC => {
399                (self.0 as u8).consensus_encode(w)?;
400                Ok(1)
401            },
402            0xFD..=0xFFFF => {
403                w.emit_u8(0xFD)?;
404                (self.0 as u16).consensus_encode(w)?;
405                Ok(3)
406            },
407            0x10000..=0xFFFFFFFF => {
408                w.emit_u8(0xFE)?;
409                (self.0 as u32).consensus_encode(w)?;
410                Ok(5)
411            },
412            _ => {
413                w.emit_u8(0xFF)?;
414                self.0.consensus_encode(w)?;
415                Ok(9)
416            },
417        }
418    }
419}

421impl Decodable for VarInt {
422    #[inline]
423    fn consensus_decode<R: io::Read + ?Sized>(r: &mut R) -> Result<Self, Error> {
424        let n = ReadExt::read_u8(r)?;
425        match n {
426            0xFF => {
427                let x = ReadExt::read_u64(r)?;
428                if x < 0x100000000 {
429                    Err(self::Error::NonMinimalVarInt)
430                } else {
431                    Ok(VarInt(x))
432                }
433            }
434            0xFE => {
435                let x = ReadExt::read_u32(r)?;
436                if x < 0x10000 {
437                    Err(self::Error::NonMinimalVarInt)
438                } else {
439                    Ok(VarInt(x as u64))
440                }
441            }
442            0xFD => {
443                let x = ReadExt::read_u16(r)?;
444                if x < 0xFD {
445                    Err(self::Error::NonMinimalVarInt)
446                } else {
447                    Ok(VarInt(x as u64))
448                }
449            }
450            n => Ok(VarInt(n as u64))
451        }
452    }
453}

557macro_rules! impl_vec {
558    ($type: ty) => {
559        impl Encodable for Vec<$type> {
560            #[inline]
561            fn consensus_encode<W: io::Write + ?Sized>(&self, w: &mut W) -> Result<usize, io::Error> {
562                let mut len = 0;
563                len += VarInt(self.len() as u64).consensus_encode(w)?;
564                for c in self.iter() {
565                    len += c.consensus_encode(w)?;
566                }
567                Ok(len)
568            }
569        }
570
571        impl Decodable for Vec<$type> {
572            #[inline]
573            fn consensus_decode_from_finite_reader<R: io::Read + ?Sized>(r: &mut R) -> Result<Self, Error> {
574                let len = VarInt::consensus_decode_from_finite_reader(r)?.0;
575                // Do not allocate upfront more items than if the sequnce of type
576                // occupied roughly quarter a block. This should never be the case
577                // for normal data, but even if that's not true - `push` will just
578                // reallocate.
579                // Note: OOM protection relies on reader eventually running out of
580                // data to feed us.
581                let max_capacity = MAX_VEC_SIZE / 4 / mem::size_of::<$type>();
582                let mut ret = Vec::with_capacity(core::cmp::min(len as usize, max_capacity));
583                for _ in 0..len {
584                    ret.push(Decodable::consensus_decode_from_finite_reader(r)?);
585                }
586                Ok(ret)
587            }
588        }
589    }

616struct ReadBytesFromFiniteReaderOpts {
617    len: usize,
618    chunk_size: usize,
619}
620
621/// Read `opts.len` bytes from reader, where `opts.len` could potentially be malicious.
622///
623/// This function relies on reader being bound in amount of data
624/// it returns for OOM protection. See [`Decodable::consensus_decode_from_finite_reader`].
625#[inline]
626fn read_bytes_from_finite_reader<D: io::Read>(mut d: D, mut opts: ReadBytesFromFiniteReaderOpts) -> Result<Vec<u8>, Error> {
627    let mut ret = vec![];
628
629    assert_ne!(opts.chunk_size, 0);
630
631    while opts.len > 0 {
632        let chunk_start = ret.len();
633        let chunk_size = core::cmp::min(opts.len, opts.chunk_size);
634        let chunk_end = chunk_start + chunk_size;
635        ret.resize(chunk_end, 0u8);
636        d.read_slice(&mut ret[chunk_start..chunk_end])?;
637        opts.len -= chunk_size;
638    }
639
640    Ok(ret)
641}

644    #[inline]
645    fn consensus_encode<W: io::Write + ?Sized>(&self, w: &mut W) -> Result<usize, io::Error> {
646        consensus_encode_with_size(self, w)
647    }
648}

653        let len = VarInt::consensus_decode(r)?.0 as usize;
654        // most real-world vec of bytes data, wouldn't be larger than 128KiB
655        read_bytes_from_finite_reader(r, ReadBytesFromFiniteReaderOpts { len, chunk_size: 128 * 1024 })
656    }

674/// Does a double-SHA256 on `data` and returns the first 4 bytes.
675fn sha2_checksum(data: &[u8]) -> [u8; 4] {
676    let checksum = <sha256d::Hash as Hash>::hash(data);
677    [checksum[0], checksum[1], checksum[2], checksum[3]]
678}

680impl Encodable for CheckedData {
681    #[inline]
682    fn consensus_encode<W: io::Write + ?Sized>(&self, w: &mut W) -> Result<usize, io::Error> {
683        (self.0.len() as u32).consensus_encode(w)?;
684        sha2_checksum(&self.0).consensus_encode(w)?;
685        w.emit_slice(&self.0)?;
686        Ok(8 + self.0.len())
687    }
688}
689
690impl Decodable for CheckedData {
691    #[inline]
692    fn consensus_decode_from_finite_reader<R: io::Read + ?Sized>(r: &mut R) -> Result<Self, Error> {
693        let len = u32::consensus_decode_from_finite_reader(r)? as usize;
694
695        let checksum = <[u8; 4]>::consensus_decode_from_finite_reader(r)?;
696        let ret = read_bytes_from_finite_reader(r, ReadBytesFromFiniteReaderOpts { len, chunk_size: MAX_VEC_SIZE })?;
697        let expected_checksum = sha2_checksum(&ret);
698        if expected_checksum != checksum {
699            Err(self::Error::InvalidChecksum {
700                expected: expected_checksum,
701                actual: checksum,
702            })
703        } else {
704            Ok(CheckedData(ret))
705        }
706    }
707}

736macro_rules! tuple_encode {
737    ($($x:ident),*) => {
738        impl <$($x: Encodable),*> Encodable for ($($x),*) {
739            #[inline]
740            #[allow(non_snake_case)]
741            fn consensus_encode<W: io::Write + ?Sized>(
742                &self,
743                w: &mut W,
744            ) -> Result<usize, io::Error> {
745                let &($(ref $x),*) = self;
746                let mut len = 0;
747                $(len += $x.consensus_encode(w)?;)*
748                Ok(len)
749            }

756                Ok(($({let $x = Decodable::consensus_decode(r)?; $x }),*))
757            }
758        }
759    };
