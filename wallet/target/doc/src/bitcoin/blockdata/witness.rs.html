21/// The Witness is the data used to unlock bitcoin since the [segwit upgrade].
22///
23/// Can be logically seen as an array of bytestrings, i.e. `Vec<Vec<u8>>`, and it is serialized on the wire
24/// in that format. You can convert between this type and `Vec<Vec<u8>>` by using [`Witness::from_slice`]
25/// and [`Witness::to_vec`].
26///
27/// For serialization and deserialization performance it is stored internally as a single `Vec`,
28/// saving some allocations.
29///
30/// [segwit upgrade]: <https://github.com/bitcoin/bips/blob/master/bip-0143.mediawiki>
31#[derive(Clone, Default, PartialEq, Eq, PartialOrd, Ord, Debug, Hash)]
32pub struct Witness {
33    /// Contains the witness `Vec<Vec<u8>>` serialization without the initial varint indicating the
34    /// number of elements (which is stored in `witness_elements`).
35    content: Vec<u8>,
36
37    /// The number of elements in the witness.
38    ///
39    /// Stored separately (instead of as a VarInt in the initial part of content) so that methods
40    /// like [`Witness::push`] don't have to shift the entire array.
41    witness_elements: usize,
42
43    /// This is the valid index pointing to the beginning of the index area. This area is 4 *
44    /// stack_size bytes at the end of the content vector which stores the indices of each item.
45    indices_start: usize,
46}
47
48/// An iterator returning individual witness elements.
49pub struct Iter<'a> {
50    inner: &'a [u8],
51    indices_start: usize,
52    current_index: usize,
53}

55impl Decodable for Witness {
56    fn consensus_decode<R: Read + ?Sized>(r: &mut R) -> Result<Self, Error> {
57        let witness_elements = VarInt::consensus_decode(r)?.0 as usize;
58        // Minimum size of witness element is 1 byte, so if the count is
59        // greater than MAX_VEC_SIZE we must return an error.
60        if witness_elements > MAX_VEC_SIZE {
61            return Err(self::Error::OversizedVectorAllocation {
62                requested: witness_elements,
63                max: MAX_VEC_SIZE,
64            });
65        }
66        if witness_elements == 0 {
67            Ok(Witness::default())
68        } else {
69            // Leave space at the head for element positions.
70            // We will rotate them to the end of the Vec later.
71            let witness_index_space = witness_elements * 4;
72            let mut cursor = witness_index_space;
73
74            // this number should be determined as high enough to cover most witness, and low enough
75            // to avoid wasting space without reallocating
76            let mut content = vec![0u8; cursor + 128];
77
78            for i in 0..witness_elements {
79                let element_size_varint = VarInt::consensus_decode(r)?;
80                let element_size_varint_len = element_size_varint.len();
81                let element_size = element_size_varint.0 as usize;
82                let required_len = cursor
83                    .checked_add(element_size)
84                    .ok_or(self::Error::OversizedVectorAllocation {
85                        requested: usize::max_value(),
86                        max: MAX_VEC_SIZE,
87                    })?
88                    .checked_add(element_size_varint_len)
89                    .ok_or(self::Error::OversizedVectorAllocation {
90                        requested: usize::max_value(),
91                        max: MAX_VEC_SIZE,
92                    })?;
93
94                if required_len > MAX_VEC_SIZE + witness_index_space {
95                    return Err(self::Error::OversizedVectorAllocation {
96                        requested: required_len,
97                        max: MAX_VEC_SIZE,
98                    });
99                }
100
101                // We will do content.rotate_left(witness_index_space) later.
102                // Encode the position's value AFTER we rotate left.
103                encode_cursor(&mut content, 0, i, cursor - witness_index_space);
104
105                resize_if_needed(&mut content, required_len);
106                element_size_varint
107                    .consensus_encode(&mut &mut content[cursor..cursor + element_size_varint_len])?;
108                cursor += element_size_varint_len;
109                r.read_exact(&mut content[cursor..cursor + element_size])?;
110                cursor += element_size;
111            }
112            content.truncate(cursor);
113            // Index space is now at the end of the Vec
114            content.rotate_left(witness_index_space);
115            Ok(Witness {
116                content,
117                witness_elements,
118                indices_start: cursor - witness_index_space,
119            })
120        }
121    }
122}

125/// Correctness Requirements: value must always fit within u32
126#[inline]
127fn encode_cursor(bytes: &mut [u8], start_of_indices: usize, index: usize, value: usize) {
128    let start = start_of_indices + index * 4;
129    let end = start + 4;
130    bytes[start..end].copy_from_slice(&u32::to_ne_bytes(value.try_into().expect("Larger than u32")));
131}
132
133#[inline]
134fn decode_cursor(bytes: &[u8], start_of_indices: usize, index: usize) -> Option<usize> {
135    let start = start_of_indices + index * 4;
136    let end = start + 4;
137    if end > bytes.len() {
138        None
139    } else {
140        Some(u32::from_ne_bytes(bytes[start..end].try_into().expect("is u32 size")) as usize)
141    }
142}
143
144fn resize_if_needed(vec: &mut Vec<u8>, required_len: usize) {
145    if required_len >= vec.len() {
146        let mut new_len = vec.len().max(1);
147        while new_len <= required_len {
148            new_len *= 2;
149        }
150        vec.resize(new_len, 0);
151    }
152}

154impl Encodable for Witness {
155    fn consensus_encode<W: Write + ?Sized>(&self, w: &mut W) -> Result<usize, io::Error> {
156        let len = VarInt(self.witness_elements as u64);
157        len.consensus_encode(w)?;
158        let content_with_indices_len = self.content.len();
159        let indices_size = self.witness_elements * 4;
160        let content_len = content_with_indices_len - indices_size;
161        w.emit_slice(&self.content[..content_len])?;
162        Ok(content_len + len.len())
163    }
164}

173    #[deprecated(since="0.30.0", note="use `Witness::from_slice()` instead")]
174    pub fn from_vec(vec: Vec<Vec<u8>>) -> Self {
175        Witness::from_slice(&vec)
176    }
177
178    /// Creates a [`Witness`] object from a slice of bytes slices where each slice is a witness item.
179    pub fn from_slice<T: AsRef<[u8]>>(slice: &[T]) -> Self {
180        let witness_elements = slice.len();
181        let index_size = witness_elements * 4;
182        let content_size = slice
183            .iter()
184            .map(|elem| elem.as_ref().len() + VarInt(elem.as_ref().len() as u64).len())
185            .sum();
186
187        let mut content = vec![0u8; content_size + index_size];
188        let mut cursor = 0usize;
189        for (i, elem) in slice.iter().enumerate() {
190            encode_cursor(&mut content, content_size, i, cursor);
191            let elem_len_varint = VarInt(elem.as_ref().len() as u64);
192            elem_len_varint
193                .consensus_encode(&mut &mut content[cursor..cursor + elem_len_varint.len()])
194                .expect("writers on vec don't errors, space granted by content_size");
195            cursor += elem_len_varint.len();
196            content[cursor..cursor + elem.as_ref().len()].copy_from_slice(elem.as_ref());
197            cursor += elem.as_ref().len();
198        }
199
200        Witness {
201            witness_elements,
202            content,
203            indices_start: content_size,
204        }
205    }

233        self.iter()

235            .sum::<usize>()

244    }

246    /// Push a new element on the witness, requires an allocation.
247    pub fn push<T: AsRef<[u8]>>(&mut self, new_element: T) {
248        self.push_slice(new_element.as_ref());
249    }
250
251    /// Push a new element slice onto the witness stack.
252    fn push_slice(&mut self, new_element: &[u8]) {
253        self.witness_elements += 1;
254        let previous_content_end = self.indices_start;
255        let element_len_varint = VarInt(new_element.len() as u64);
256        let current_content_len = self.content.len();
257        let new_item_total_len = element_len_varint.len() + new_element.len();
258        self.content
259            .resize(current_content_len + new_item_total_len + 4, 0);
260
261        self.content[previous_content_end..].rotate_right(new_item_total_len);
262        self.indices_start += new_item_total_len;
263        encode_cursor(&mut self.content, self.indices_start, self.witness_elements - 1, previous_content_end);
264
265        let end_varint = previous_content_end + element_len_varint.len();
266        element_len_varint
267            .consensus_encode(&mut &mut self.content[previous_content_end..end_varint])
268            .expect("writers on vec don't error, space granted through previous resize");
269        self.content[end_varint..end_varint + new_element.len()].copy_from_slice(new_element);
270    }
271
272    /// Pushes a DER-encoded ECDSA signature with a signature hash type as a new element on the
273    /// witness, requires an allocation.
274    pub fn push_bitcoin_signature(&mut self, signature: &ecdsa::SerializedSignature, hash_type: EcdsaSighashType) {
275        // Note that a maximal length ECDSA signature is 72 bytes, plus the sighash type makes 73
276        let mut sig = [0; 73];
277        sig[..signature.len()].copy_from_slice(signature);
278        sig[signature.len()] = hash_type as u8;
279        self.push(&sig[..signature.len() + 1]);
280    }

283    fn element_at(&self, index: usize) -> Option<&[u8]> {
284        let varint = VarInt::consensus_decode(&mut &self.content[index..]).ok()?;
285        let start = index + varint.len();
286        Some(&self.content[start..start + varint.0 as usize])
287    }
288
289    /// Returns the last element in the witness, if any.
290    pub fn last(&self) -> Option<&[u8]> {
291        if self.witness_elements == 0 {
292            None
293        } else {
294            self.nth(self.witness_elements - 1)
295        }
296    }
297
298    /// Returns the second-to-last element in the witness, if any.
299    pub fn second_to_last(&self) -> Option<&[u8]> {
300        if self.witness_elements <= 1 {
301            None
302        } else {
303            self.nth(self.witness_elements - 2)
304        }
305    }
306
307    /// Return the nth element in the witness, if any
308    pub fn nth(&self, index: usize) -> Option<&[u8]> {
309        let pos = decode_cursor(&self.content, self.indices_start, index)?;
310        self.element_at(pos)
311    }
312
313    /// Get Tapscript following BIP341 rules regarding accounting for an annex.
314    ///
315    /// This does not guarantee that this represents a P2TR [`Witness`]. It
316    /// merely gets the second to last or third to last element depending on
317    /// the first byte of the last element being equal to 0x50. See
318    /// [Script::is_v1_p2tr](crate::blockdata::script::Script::is_v1_p2tr) to
319    /// check whether this is actually a Taproot witness.
320    pub fn tapscript(&self) -> Option<&Script> {
321        let len = self.len();
322        self
323            .last()
324            .map(|last_elem| {
325                // From BIP341:
326                // If there are at least two witness elements, and the first byte of
327                // the last element is 0x50, this last element is called annex a
328                // and is removed from the witness stack.
329                if len >= 2 && last_elem.first() == Some(&TAPROOT_ANNEX_PREFIX) {
330                    // account for the extra item removed from the end
331                    3
332                } else {
333                    // otherwise script is 2nd from last
334                    2
335                }
336            })
337            .filter(|&script_pos_from_last| len >= script_pos_from_last)
338            .and_then(|script_pos_from_last| {
339                self.nth(len - script_pos_from_last)
340            })
341            .map(Script::from_bytes)
342    }
343}

353impl<'a> Iterator for Iter<'a> {
354    type Item = &'a [u8];
355
356    fn next(&mut self) -> Option<Self::Item> {
357        let index = decode_cursor(self.inner, self.indices_start, self.current_index)?;
358        let varint = VarInt::consensus_decode(&mut &self.inner[index..]).ok()?;
359        let start = index + varint.len();
360        let end = start + varint.0 as usize;
361        let slice = &self.inner[start..end];
362        self.current_index += 1;
363        Some(slice)
364    }
365
366    fn size_hint(&self) -> (usize, Option<usize>) {
367        let total_count = (self.inner.len() - self.indices_start) / 4;
368        let remaining = total_count - self.current_index;
369        (remaining, Some(remaining))
370    }
371}

384// Serde keep backward compatibility with old Vec<Vec<u8>> format
385#[cfg(feature = "serde")]
386impl serde::Serialize for Witness {
387    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
388    where
389        S: serde::Serializer,
390    {
391        use serde::ser::SerializeSeq;
392
393        let human_readable = serializer.is_human_readable();
394        let mut seq = serializer.serialize_seq(Some(self.witness_elements))?;
395
396        for elem in self.iter() {
397            if human_readable {
398                seq.serialize_element(&crate::serde_utils::SerializeBytesAsHex(elem))?;
399            } else {
400                seq.serialize_element(&elem)?;
401            }
402        }
403        seq.end()
404    }
405}

407#[cfg(feature = "serde")]
408impl<'de> serde::Deserialize<'de> for Witness {
409    fn deserialize<D>(deserializer: D) -> Result<Self, D::Error>
410    where
411        D: serde::Deserializer<'de>,
412    {
413        struct Visitor;         // Human-readable visitor.
414        impl<'de> serde::de::Visitor<'de> for Visitor
415        {
416            type Value = Witness;
417
418            fn expecting(&self, f: &mut core::fmt::Formatter) -> core::fmt::Result {
419                write!(f, "a sequence of hex arrays")
420            }
421
422            fn visit_seq<A: serde::de::SeqAccess<'de>>(self, mut a: A) -> Result<Self::Value, A::Error>
423            {
424                use crate::hashes::hex::FromHex;
425                use crate::hashes::hex::Error::*;
426                use serde::de::{self, Unexpected};
427
428                let mut ret = match a.size_hint() {
429                    Some(len) => Vec::with_capacity(len),
430                    None => Vec::new(),
431                };
432
433                while let Some(elem) = a.next_element::<String>()? {
434                    let vec = Vec::<u8>::from_hex(&elem).map_err(|e| {
435                        match e {
436                            InvalidChar(b) => {
437                                match core::char::from_u32(b.into()) {
438                                    Some(c) => de::Error::invalid_value(Unexpected::Char(c), &"a valid hex character"),
439                                    None => de::Error::invalid_value(Unexpected::Unsigned(b.into()), &"a valid hex character")
440                                }
441                            }
442                            OddLengthString(len) => de::Error::invalid_length(len, &"an even length string"),
443                            InvalidLength(expected, got) => {
444                                let exp = format!("expected length: {}", expected);
445                                de::Error::invalid_length(got, &exp.as_str())
446                            }
447                        }
448                    })?;
449                    ret.push(vec);
450                }
451                Ok(Witness::from_slice(&ret))
452            }
453        }
454
455        if deserializer.is_human_readable() {
456            deserializer.deserialize_seq(Visitor)
457        } else {
458            let vec: Vec<Vec<u8>> = serde::Deserialize::deserialize(deserializer)?;
459            Ok(Witness::from_slice(&vec))
460        }
461    }
462}
