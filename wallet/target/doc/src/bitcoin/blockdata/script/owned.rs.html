18/// An owned, growable script.
19///
20/// `ScriptBuf` is the most common script type that has the ownership over the contents of the
21/// script. It has a close relationship with its borrowed counterpart, [`Script`].
22///
23/// Just as other similar types, this implements [`Deref`], so [deref coercions] apply. Also note
24/// that all the safety/validity restrictions that apply to [`Script`] apply to `ScriptBuf` as well.
25///
26/// [deref coercions]: https://doc.rust-lang.org/std/ops/trait.Deref.html#more-on-deref-coercion
27#[derive(Default, Clone, PartialOrd, Ord, PartialEq, Eq, Hash)]
28pub struct ScriptBuf(pub(in crate::blockdata::script) Vec<u8>);
29
30impl ScriptBuf {
31    /// Creates a new empty script.
32    pub fn new() -> Self {
33        ScriptBuf(Vec::new())
34    }
35
36    /// Creates a new empty script with pre-allocated capacity.
37    pub fn with_capacity(capacity: usize) -> Self {
38        ScriptBuf(Vec::with_capacity(capacity))
39    }
40
41    /// Pre-allocates at least `additional_len` bytes if needed.
42    ///
43    /// Reserves capacity for at least `additional_len` more bytes to be inserted in the given
44    /// script. The script may reserve more space to speculatively avoid frequent reallocations.
45    /// After calling `reserve`, capacity will be greater than or equal to
46    /// `self.len() + additional_len`. Does nothing if capacity is already sufficient.
47    ///
48    /// # Panics
49    ///
50    /// Panics if the new capacity exceeds `isize::MAX bytes`.
51    pub fn reserve(&mut self, additional_len: usize) {
52        self.0.reserve(additional_len);
53    }
54
55    /// Pre-allocates exactly `additional_len` bytes if needed.
56    ///
57    /// Unlike `reserve`, this will not deliberately over-allocate to speculatively avoid frequent
58    /// allocations. After calling `reserve_exact`, capacity will be greater than or equal to
59    /// `self.len() + additional`. Does nothing if the capacity is already sufficient.
60    ///
61    /// Note that the allocator may give the collection more space than it requests. Therefore,
62    /// capacity can not be relied upon to be precisely minimal. Prefer [`reserve`](Self::reserve)
63    /// if future insertions are expected.
64    ///
65    /// # Panics
66    ///
67    /// Panics if the new capacity exceeds `isize::MAX bytes`.
68    pub fn reserve_exact(&mut self, additional_len: usize) {
69        self.0.reserve_exact(additional_len);
70    }

87    /// Generates P2PK-type of scriptPubkey.
88    pub fn new_p2pk(pubkey: &PublicKey) -> Self {
89        Builder::new()
90            .push_key(pubkey)
91            .push_opcode(OP_CHECKSIG)

95    /// Generates P2PKH-type of scriptPubkey.
96    pub fn new_p2pkh(pubkey_hash: &PubkeyHash) -> Self {
97        Builder::new()
98            .push_opcode(OP_DUP)
99            .push_opcode(OP_HASH160)
100            .push_slice(pubkey_hash)
101            .push_opcode(OP_EQUALVERIFY)
102            .push_opcode(OP_CHECKSIG)
103            .into_script()
104    }

106    /// Generates P2SH-type of scriptPubkey with a given hash of the redeem script.
107    pub fn new_p2sh(script_hash: &ScriptHash) -> Self {
108        Builder::new()
109            .push_opcode(OP_HASH160)
110            .push_slice(script_hash)
111            .push_opcode(OP_EQUAL)
112            .into_script()
113    }

115    /// Generates P2WPKH-type of scriptPubkey.
116    pub fn new_v0_p2wpkh(pubkey_hash: &WPubkeyHash) -> Self {
117        // pubkey hash is 20 bytes long, so it's safe to use `new_witness_program_unchecked` (Segwitv0)
118        ScriptBuf::new_witness_program_unchecked(WitnessVersion::V0, pubkey_hash)
119    }

121    /// Generates P2WSH-type of scriptPubkey with a given hash of the redeem script.
122    pub fn new_v0_p2wsh(script_hash: &WScriptHash) -> Self {
123        // script hash is 32 bytes long, so it's safe to use `new_witness_program_unchecked` (Segwitv0)
124        ScriptBuf::new_witness_program_unchecked(WitnessVersion::V0, script_hash)
125    }

127    /// Generates P2TR for script spending path using an internal public key and some optional
128    /// script tree merkle root.
129    pub fn new_v1_p2tr<C: Verification>(secp: &Secp256k1<C>, internal_key: UntweakedPublicKey, merkle_root: Option<TapNodeHash>) -> Self {
130        let (output_key, _) = internal_key.tap_tweak(secp, merkle_root);
131        // output key is 32 bytes long, so it's safe to use `new_witness_program_unchecked` (Segwitv1)
132        ScriptBuf::new_witness_program_unchecked(WitnessVersion::V1, output_key.serialize())
133    }
134
135    /// Generates P2TR for key spending path for a known [`TweakedPublicKey`].
136    pub fn new_v1_p2tr_tweaked(output_key: TweakedPublicKey) -> Self {
137        // output key is 32 bytes long, so it's safe to use `new_witness_program_unchecked` (Segwitv1)
138        ScriptBuf::new_witness_program_unchecked(WitnessVersion::V1, output_key.serialize())
139    }

141    /// Generates P2WSH-type of scriptPubkey with a given [`WitnessProgram`].
142    pub fn new_witness_program(witness_program: &WitnessProgram) -> Self {
143        Builder::new()
144            .push_opcode(witness_program.version().into())
145            .push_slice(witness_program.program())
146            .into_script()
147    }
148
149    /// Generates P2WSH-type of scriptPubkey with a given [`WitnessVersion`] and the program bytes.
150    /// Does not do any checks on version or program length.
151    ///
152    /// Convenience method used by `new_v0_p2wpkh`, `new_v0_p2wsh`, `new_v1_p2tr`, and
153    /// `new_v1_p2tr_tweaked`.
154    fn new_witness_program_unchecked<T: AsRef<PushBytes>>(version: WitnessVersion, program: T) -> Self {
155        let program = program.as_ref();
156        debug_assert!(program.len() >= 2 && program.len() <= 40);
157        // In segwit v0, the program must be 20 or 32 bytes long.
158        debug_assert!(version != WitnessVersion::V0 || program.len() == 20 || program.len() == 32);
159        Builder::new()
160            .push_opcode(version.into())
161            .push_slice(program)
162            .into_script()
163    }
164
165    /// Generates OP_RETURN-type of scriptPubkey for the given data.
166    pub fn new_op_return<T: AsRef<PushBytes>>(data: &T) -> Self {
167        Builder::new()
168            .push_opcode(OP_RETURN)
169            .push_slice(data)
170            .into_script()
171    }

198    /// Returns the script code used for spending a P2WPKH output if this script is a script pubkey
199    /// for a P2WPKH output. The `scriptCode` is described in [BIP143].
200    ///
201    /// [BIP143]: <https://github.com/bitcoin/bips/blob/99701f68a88ce33b2d0838eb84e115cef505b4c2/bip-0143.mediawiki>
202    pub fn p2wpkh_script_code(&self) -> Option<ScriptBuf> {
203        self.v0_p2wpkh().map(|wpkh| {
204            Builder::new()
205                .push_opcode(OP_DUP)
206                .push_opcode(OP_HASH160)
207                // The `self` script is 0x00, 0x14, <pubkey_hash>
208                .push_slice(wpkh)
209                .push_opcode(OP_EQUALVERIFY)
210                .push_opcode(OP_CHECKSIG)
211                .into_script()
212        })
213    }

215    /// Adds a single opcode to the script.
216    pub fn push_opcode(&mut self, data: opcodes::All) {
217        self.0.push(data.to_u8());
218    }
219
220    /// Adds instructions to push some arbitrary data onto the stack.
221    pub fn push_slice<T: AsRef<PushBytes>>(&mut self, data: T) {
222        let data = data.as_ref();
223        self.reserve(Self::reserved_len_for_slice(data.len()));
224        self.push_slice_no_opt(data);
225    }
226
227    /// Pushes the slice without reserving
228    fn push_slice_no_opt(&mut self, data: &PushBytes) {
229        // Start with a PUSH opcode
230        match data.len() as u64 {
231            n if n < opcodes::Ordinary::OP_PUSHDATA1 as u64 => { self.0.push(n as u8); },
232            n if n < 0x100 => {
233                self.0.push(opcodes::Ordinary::OP_PUSHDATA1.to_u8());
234                self.0.push(n as u8);
235            },
236            n if n < 0x10000 => {
237                self.0.push(opcodes::Ordinary::OP_PUSHDATA2.to_u8());
238                self.0.push((n % 0x100) as u8);
239                self.0.push((n / 0x100) as u8);
240            },
241            n if n < 0x100000000 => {
242                self.0.push(opcodes::Ordinary::OP_PUSHDATA4.to_u8());
243                self.0.push((n % 0x100) as u8);
244                self.0.push(((n / 0x100) % 0x100) as u8);
245                self.0.push(((n / 0x10000) % 0x100) as u8);
246                self.0.push((n / 0x1000000) as u8);
247            }
248            _ => panic!("tried to put a 4bn+ sized object into a script!")
249        }
250        // Then push the raw bytes
251        self.0.extend_from_slice(data.as_bytes());
252    }
253
254    /// Computes the sum of `len` and the lenght of an appropriate push opcode.
255    pub(in crate::blockdata::script) fn reserved_len_for_slice(len: usize) -> usize {
256        len + match len {
257            0..=0x4b => 1,
258            0x4c..=0xff => 2,
259            0x100..=0xffff => 3,
260            // we don't care about oversized, the other fn will panic anyway
261            _ => 5,
262        }
263    }
264
265    /// Add a single instruction to the script.
266    ///
267    /// ## Panics
268    ///
269    /// The method panics if the instruction is a data push with length greater or equal to
270    /// 0x100000000.
271    pub fn push_instruction(&mut self, instruction: Instruction<'_>) {
272        match instruction {
273            Instruction::Op(opcode) => self.push_opcode(opcode),
274            Instruction::PushBytes(bytes) => self.push_slice(bytes),
275        }
276    }
277
278    /// Like push_instruction, but avoids calling `reserve` to not re-check the length.
279    pub fn push_instruction_no_opt(&mut self, instruction: Instruction<'_>) {
280        match instruction {
281            Instruction::Op(opcode) => self.push_opcode(opcode),
282            Instruction::PushBytes(bytes) => self.push_slice_no_opt(bytes),
283        }
284    }

286    /// Adds an `OP_VERIFY` to the script or replaces the last opcode with VERIFY form.
287    ///
288    /// Some opcodes such as `OP_CHECKSIG` have a verify variant that works as if `VERIFY` was
289    /// in the script right after. To save space this function appends `VERIFY` only if
290    /// the most-recently-added opcode *does not* have an alternate `VERIFY` form. If it does
291    /// the last opcode is replaced. E.g., `OP_CHECKSIG` will become `OP_CHECKSIGVERIFY`.
292    ///
293    /// Note that existing `OP_*VERIFY` opcodes do not lead to the instruction being ignored
294    /// because `OP_VERIFY` consumes an item from the stack so ignoring them would change the
295    /// semantics.
296    ///
297    /// This function needs to iterate over the script to find the last instruction. Prefer
298    /// `Builder` if you're creating the script from scratch or if you want to push `OP_VERIFY`
299    /// multiple times.
300    pub fn scan_and_push_verify(&mut self) {
301        self.push_verify(self.last_opcode());
302    }
303
304    /// Adds an `OP_VERIFY` to the script or changes the most-recently-added opcode to `VERIFY`
305    /// alternative.
306    ///
307    /// See the public fn [`Self::scan_and_push_verify`] to learn more.
308    pub(in crate::blockdata::script) fn push_verify(&mut self, last_opcode: Option<opcodes::All>) {
309        match opcode_to_verify(last_opcode) {
310            Some(opcode) => {
311                self.0.pop();
312                self.push_opcode(opcode);
313            },
314            None => self.push_opcode(OP_VERIFY),
315        }
316    }

318    /// Converts this `ScriptBuf` into a [boxed](Box) [`Script`].
319    ///
320    /// This method reallocates if the capacity is greater than lenght of the script but should not
321    /// when they are equal. If you know beforehand that you need to create a script of exact size
322    /// use [`reserve_exact`](Self::reserve_exact) before adding data to the script so that the
323    /// reallocation can be avoided.
324    #[must_use = "`self` will be dropped if the result is not used"]
325    #[inline]
326    pub fn into_boxed_script(self) -> Box<Script> {
327        // Copied from PathBuf::into_boxed_path
328        let rw = Box::into_raw(self.0.into_boxed_slice()) as *mut Script;
329        unsafe { Box::from_raw(rw) }
330    }

333impl<'a> core::iter::FromIterator<Instruction<'a>> for ScriptBuf {
334    fn from_iter<T>(iter: T) -> Self where T: IntoIterator<Item = Instruction<'a>> {
335        let mut script = ScriptBuf::new();
336        script.extend(iter);
337        script
338    }
339}
340
341impl<'a> Extend<Instruction<'a>> for ScriptBuf {
342    fn extend<T>(&mut self, iter: T) where T: IntoIterator<Item = Instruction<'a>> {
343        let iter = iter.into_iter();
344        // Most of Bitcoin scripts have only a few opcodes, so we can avoid reallocations in many
345        // cases.
346        if iter.size_hint().1.map(|max| max < 6).unwrap_or(false) {
347            let mut iter = iter.fuse();
348            // `MaybeUninit` might be faster but we don't want to introduce more `unsafe` than
349            // required.
350            let mut head = [None; 5];
351            let mut total_size = 0;
352            for (head, instr) in head.iter_mut().zip(&mut iter) {
353                total_size += instr.script_serialized_len();
354                *head = Some(instr);
355            }
356            // Incorrect impl of `size_hint` breaks `Iterator` contract so we're free to panic.
357            assert!(iter.next().is_none(), "Buggy implementation of `Iterator` on {} returns invalid upper bound", core::any::type_name::<T::IntoIter>());
358            self.reserve(total_size);
359            for instr in head.iter().cloned().flatten() {
360                self.push_instruction_no_opt(instr);
361            }
362        } else {
363            for instr in iter {
364                self.push_instruction(instr);
365            }
366        }
367    }
