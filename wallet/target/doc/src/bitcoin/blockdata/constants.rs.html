65/// Constructs and returns the coinbase (and only) transaction of the Bitcoin genesis block.
66fn bitcoin_genesis_tx() -> Transaction {
67    // Base
68    let mut ret = Transaction {
69        version: 1,
70        lock_time: absolute::LockTime::ZERO,
71        input: vec![],
72        output: vec![],
73    };
74
75    // Inputs
76    let in_script = script::Builder::new().push_int(486604799)
77                                          .push_int_non_minimal(4)
78                                          .push_slice(b"The Times 03/Jan/2009 Chancellor on brink of second bailout for banks")
79                                          .into_script();
80    ret.input.push(TxIn {
81        previous_output: OutPoint::null(),
82        script_sig: in_script,
83        sequence: Sequence::MAX,
84        witness: Witness::default(),
85    });
86
87    // Outputs
88    let script_bytes = hex!("04678afdb0fe5548271967f1a67130b7105cd6a828e03909a67962e0ea1f61deb649f6bc3f4cef38c4f35504e51ec112de5c384df7ba0b8d578a4c702b6bf11d5f");
89    let out_script = script::Builder::new()
90        .push_slice(script_bytes)
91        .push_opcode(OP_CHECKSIG)
92        .into_script();
93    ret.output.push(TxOut {
94        value: 50 * COIN_VALUE,
95        script_pubkey: out_script
96    });
97
98    // end
99    ret
100}

102/// Constructs and returns the genesis block.
103pub fn genesis_block(network: Network) -> Block {
104    let txdata = vec![bitcoin_genesis_tx()];
105    let hash: sha256d::Hash = txdata[0].txid().into();
106    let merkle_root = hash.into();
107    match network {
108        Network::Bitcoin => {
109            Block {
110                header: block::Header {
111                    version: block::Version::ONE,
112                    prev_blockhash: Hash::all_zeros(),
113                    merkle_root,
114                    time: 1231006505,
115                    bits: CompactTarget::from_consensus(0x1d00ffff),
116                    nonce: 2083236893
117                },
118                txdata,
119            }
120        }
121        Network::Testnet => {
122            Block {
123                header: block::Header {
124                    version: block::Version::ONE,
125                    prev_blockhash: Hash::all_zeros(),
126                    merkle_root,
127                    time: 1296688602,
128                    bits: CompactTarget::from_consensus(0x1d00ffff),
129                    nonce: 414098458
130                },
131                txdata,
132            }
133        }
134        Network::Signet => {
135            Block {
136                header: block::Header {
137                    version: block::Version::ONE,
138                    prev_blockhash: Hash::all_zeros(),
139                    merkle_root,
140                    time: 1598918400,
141                    bits: CompactTarget::from_consensus(0x1e0377ae),
142                    nonce: 52613770
143                },
144                txdata,
145            }
146        }
147        Network::Regtest => {
148            Block {
149                header: block::Header {
150                    version: block::Version::ONE,
151                    prev_blockhash: Hash::all_zeros(),
152                    merkle_root,
153                    time: 1296688602,
154                    bits: CompactTarget::from_consensus(0x207fffff),
155                    nonce: 2
156                },
157                txdata,
158            }
159        }
160    }
161}

163/// The uniquely identifying hash of the target blockchain.
164#[derive(Copy, Clone, PartialEq, Eq, PartialOrd, Ord, Hash)]
165pub struct ChainHash([u8; 32]);
166impl_array_newtype!(ChainHash, u8, 32);
167impl_bytes_newtype!(ChainHash, 32);
168
169impl ChainHash {
170    // Mainnet value can be verified at https://github.com/lightning/bolts/blob/master/00-introduction.md
171    /// `ChainHash` for mainnet bitcoin.
172    pub const BITCOIN: Self = Self([111, 226, 140, 10, 182, 241, 179, 114, 193, 166, 162, 70, 174, 99, 247, 79, 147, 30, 131, 101, 225, 90, 8, 156, 104, 214, 25, 0, 0, 0, 0, 0]);
173    /// `ChainHash` for testnet bitcoin.

175    /// `ChainHash` for signet bitcoin.
176    pub const SIGNET: Self = Self([246, 30, 238, 59, 99, 163, 128, 164, 119, 160, 99, 175, 50, 178, 187, 201, 124, 159, 249, 240, 31, 44, 66, 37, 233, 115, 152, 129, 8, 0, 0, 0]);
177    /// `ChainHash` for regtest bitcoin.
178    pub const REGTEST: Self = Self([6, 34, 110, 70, 17, 26, 11, 89, 202, 175, 18, 96, 67, 235, 91, 191, 40, 195, 79, 58, 94, 51, 42, 31, 199, 178, 183, 60, 241, 136, 145, 15]);
179
180    /// Returns the hash of the `network` genesis block for use as a chain hash.
181    ///
182    /// See [BOLT 0](https://github.com/lightning/bolts/blob/ffeece3dab1c52efdb9b53ae476539320fa44938/00-introduction.md#chain_hash)
183    /// for specification.
184    pub const fn using_genesis_block(network: Network) -> Self {
185        let hashes = [Self::BITCOIN, Self::TESTNET, Self::SIGNET, Self::REGTEST];
186        hashes[network as usize]
187    }

257    // The *_chain_hash tests are sanity/regression tests, they verify that the const byte array
258    // representing the genesis block is the same as that created by hashing the genesis block.
259    fn chain_hash_and_genesis_block(network: Network) {
260        use crate::hashes::sha256;
261
262        // The genesis block hash is a double-sha256 and it is displayed backwards.
263        let genesis_hash = genesis_block(network).block_hash();
264        // We abuse the sha256 hash here so we get a LowerHex impl that does not print the hex backwards.
265        let hash = sha256::Hash::from_slice(genesis_hash.as_byte_array()).unwrap();
266        let want = format!("{:02x}", hash);
267
268        let chain_hash = ChainHash::using_genesis_block(network);
269        let got = format!("{:02x}", chain_hash);
270
271        // Compare strings because the spec specifically states how the chain hash must encode to hex.
272        assert_eq!(got, want);
273
274        match network {
275            Network::Bitcoin => {},
276            Network::Testnet => {},
277            Network::Signet => {},
278            Network::Regtest => {},
279            // Update ChainHash::using_genesis_block and chain_hash_genesis_block with new variants.
280        }
281    }
282
283    macro_rules! chain_hash_genesis_block {
284        ($($test_name:ident, $network:expr);* $(;)*) => {
285            $(
286                #[test]
287                fn $test_name() {
288                    chain_hash_and_genesis_block($network);
289                }
290            )*
291        }
292    }
