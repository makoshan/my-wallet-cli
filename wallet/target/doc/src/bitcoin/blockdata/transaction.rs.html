4//! Bitcoin transactions.
5//!
6//! A transaction describes a transfer of money. It consumes previously-unspent
7//! transaction outputs and produces new ones, satisfying the condition to spend
8//! the old outputs (typically a digital signature with a specific key must be
9//! provided) and defining the condition to spend the new ones. The use of digital
10//! signatures ensures that coins cannot be spent by unauthorized parties.
11//!
12//! This module provides the structures and functions needed to support transactions.
13//!

43/// A reference to a transaction output.
44///
45/// ### Bitcoin Core References
46///
47/// * [COutPoint definition](https://github.com/bitcoin/bitcoin/blob/345457b542b6a980ccfbc868af0970a6f91d1b82/src/primitives/transaction.h#L26)
48#[derive(Copy, Clone, Debug, Eq, Hash, PartialEq, PartialOrd, Ord)]
49pub struct OutPoint {
50    /// The referenced transaction's txid.
51    pub txid: Txid,
52    /// The index of the referenced output in its transaction's vout.
53    pub vout: u32,
54}
55#[cfg(feature = "serde")]
56crate::serde_utils::serde_struct_human_string_impl!(OutPoint, "an OutPoint", txid, vout);
57
58impl OutPoint {
59    /// Creates a new [`OutPoint`].
60    #[inline]
61    pub fn new(txid: Txid, vout: u32) -> OutPoint {
62        OutPoint { txid, vout }
63    }
64
65    /// Creates a "null" `OutPoint`.
66    ///
67    /// This value is used for coinbase transactions because they don't have any previous outputs.
68    #[inline]
69    pub fn null() -> OutPoint {
70        OutPoint {
71            txid: Hash::all_zeros(),
72            vout: u32::max_value(),
73        }
74    }
75
76    /// Checks if an `OutPoint` is "null".
77    ///
78    /// # Examples
79    ///
80    /// ```rust
81    /// use bitcoin::constants::genesis_block;
82    /// use bitcoin::network::constants::Network;
83    ///
84    /// let block = genesis_block(Network::Bitcoin);
85    /// let tx = &block.txdata[0];
86    ///
87    /// // Coinbase transactions don't have any previous output.
88    /// assert!(tx.input[0].previous_output.is_null());
89    /// ```
90    #[inline]
91    pub fn is_null(&self) -> bool {
92        *self == OutPoint::null()
93    }
94}

108/// An error in parsing an OutPoint.
109#[derive(Clone, PartialEq, Eq, Debug)]
110#[non_exhaustive]
111pub enum ParseOutPointError {
112    /// Error in TXID part.
113    Txid(hashes::hex::Error),
114    /// Error in vout part.
115    Vout(crate::error::ParseIntError),
116    /// Error in general format.
117    Format,
118    /// Size exceeds max.
119    TooLong,
120    /// Vout part is not strictly numeric without leading zeroes.
121    VoutNotCanonical,
122}
123
124impl fmt::Display for ParseOutPointError {
125    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
126        match *self {
127            ParseOutPointError::Txid(ref e) => write_err!(f, "error parsing TXID"; e),
128            ParseOutPointError::Vout(ref e) => write_err!(f, "error parsing vout"; e),
129            ParseOutPointError::Format => write!(f, "OutPoint not in <txid>:<vout> format"),
130            ParseOutPointError::TooLong => write!(f, "vout should be at most 10 digits"),
131            ParseOutPointError::VoutNotCanonical => write!(f, "no leading zeroes or + allowed in vout part"),
132        }
133    }
134}

138impl std::error::Error for ParseOutPointError {
139    fn source(&self) -> Option<&(dyn std::error::Error + 'static)> {
140        use self::ParseOutPointError::*;
141
142        match self {
143            Txid(e) => Some(e),
144            Vout(e) => Some(e),
145            Format | TooLong | VoutNotCanonical => None,
146        }
147    }

150/// Parses a string-encoded transaction index (vout).
151///
152/// Does not permit leading zeroes or non-digit characters.
153fn parse_vout(s: &str) -> Result<u32, ParseOutPointError> {
154    if s.len() > 1 {
155        let first = s.chars().next().unwrap();
156        if first == '0' || first == '+' {
157            return Err(ParseOutPointError::VoutNotCanonical);
158        }
159    }
160    crate::parse::int(s).map_err(ParseOutPointError::Vout)
161}
162
163impl core::str::FromStr for OutPoint {
164    type Err = ParseOutPointError;
165
166    fn from_str(s: &str) -> Result<Self, Self::Err> {
167        if s.len() > 75 { // 64 + 1 + 10
168            return Err(ParseOutPointError::TooLong);
169        }
170        let find = s.find(':');
171        if find.is_none() || find != s.rfind(':') {
172            return Err(ParseOutPointError::Format);
173        }
174        let colon = find.unwrap();
175        if colon == 0 || colon == s.len() - 1 {
176            return Err(ParseOutPointError::Format);
177        }
178        Ok(OutPoint {
179            txid: s[..colon].parse().map_err(ParseOutPointError::Txid)?,
180            vout: parse_vout(&s[colon+1..])?,
181        })
182    }
183}

185/// Bitcoin transaction input.
186///
187/// It contains the location of the previous transaction's output,
188/// that it spends and set of scripts that satisfy its spending
189/// conditions.
190///
191/// ### Bitcoin Core References
192///
193/// * [CTxIn definition](https://github.com/bitcoin/bitcoin/blob/345457b542b6a980ccfbc868af0970a6f91d1b82/src/primitives/transaction.h#L65)
194#[derive(Clone, PartialEq, Eq, PartialOrd, Ord, Debug, Hash)]
195#[cfg_attr(feature = "serde", derive(Serialize, Deserialize))]
196#[cfg_attr(feature = "serde", serde(crate = "actual_serde"))]
197pub struct TxIn {
198    /// The reference to the previous output that is being used an an input.
199    pub previous_output: OutPoint,
200    /// The script which pushes values on the stack which will cause
201    /// the referenced output's script to be accepted.
202    pub script_sig: ScriptBuf,
203    /// The sequence number, which suggests to miners which of two
204    /// conflicting transactions should be preferred, or 0xFFFFFFFF
205    /// to ignore this feature. This is generally never used since
206    /// the miner behaviour cannot be enforced.
207    pub sequence: Sequence,
208    /// Witness data: an array of byte-arrays.
209    /// Note that this field is *not* (de)serialized with the rest of the TxIn in
210    /// Encodable/Decodable, as it is (de)serialized at the end of the full
211    /// Transaction. It *is* (de)serialized with the rest of the TxIn in other
212    /// (de)serialization routines.
213    pub witness: Witness
214}
215
216impl TxIn {
217    /// Returns true if this input enables the [`absolute::LockTime`] (aka `nLockTime`) of its
218    /// [`Transaction`].
219    ///
220    /// `nLockTime` is enabled if *any* input enables it. See [`Transaction::is_lock_time_enabled`]
221    ///  to check the overall state. If none of the inputs enables it, the lock time value is simply
222    ///  ignored. If this returns false and OP_CHECKLOCKTIMEVERIFY is used in the redeem script with
223    ///  this input then the script execution will fail [BIP-0065].
224    ///
225    /// [BIP-65](https://github.com/bitcoin/bips/blob/master/bip-0065.mediawiki)
226    pub fn enables_lock_time(&self) -> bool {
227        self.sequence != Sequence::MAX
228    }
229
230    /// The weight of the TxIn when it's included in a legacy transaction (i.e., a transaction
231    /// having only legacy inputs).
232    ///
233    /// The witness weight is ignored here even when the witness is non-empty.
234    /// If you want the witness to be taken into account, use `TxIn::segwit_weight` instead.
235    ///
236    /// Keep in mind that when adding a TxIn to a transaction, the total weight of the transaction
237    /// might increase more than `TxIn::legacy_weight`. This happens when the new input added causes
238    /// the input length `VarInt` to increase its encoding length.
239    pub fn legacy_weight(&self) -> usize {
240        let script_sig_size = self.script_sig.len();
241        // Size in vbytes:
242        // previous_output (36) + script_sig varint len + script_sig push + sequence (4)
243        // We then multiply by 4 to convert to WU
244        (36 + VarInt(script_sig_size as u64).len() + script_sig_size + 4) * 4
245    }
246
247    /// The weight of the TxIn when it's included in a segwit transaction (i.e., a transcation
248    /// having at least one segwit input).
249    ///
250    /// This always takes into account the witness, even when empty, in which
251    /// case 1WU for the witness length varint (`00`) is included.
252    ///
253    /// Keep in mind that when adding a TxIn to a transaction, the total weight of the transaction
254    /// might increase more than `TxIn::segwit_weight`. This happens when:
255    /// - the new input added causes the input length `VarInt` to increase its encoding length
256    /// - the new input is the first segwit input added - this will add an additional 2WU to the
257    ///   transaction weight to take into account the segwit marker
258    pub fn segwit_weight(&self) -> usize {
259        self.legacy_weight() + self.witness.serialized_len()
260    }
261}

263impl Default for TxIn {
264    fn default() -> TxIn {
265        TxIn {
266            previous_output: OutPoint::default(),
267            script_sig: ScriptBuf::new(),
268            sequence: Sequence::MAX,
269            witness: Witness::default(),
270        }
271    }

274/// Bitcoin transaction input sequence number.
275///
276/// The sequence field is used for:
277/// - Indicating whether absolute lock-time (specified in `lock_time` field of [`Transaction`])
278///   is enabled.
279/// - Indicating and encoding [BIP-68] relative lock-times.
280/// - Indicating whether a transcation opts-in to [BIP-125] replace-by-fee.
281///
282/// Note that transactions spending an output with `OP_CHECKLOCKTIMEVERIFY`MUST NOT use
283/// `Sequence::MAX` for the corresponding input. [BIP-65]
284///
285/// [BIP-65]: <https://github.com/bitcoin/bips/blob/master/bip-0065.mediawiki>
286/// [BIP-68]: <https://github.com/bitcoin/bips/blob/master/bip-0068.mediawiki>
287/// [BIP-125]: <https://github.com/bitcoin/bips/blob/master/bip-0125.mediawiki>
288#[derive(Copy, Clone, PartialEq, Eq, PartialOrd, Ord, Debug, Hash)]
289#[cfg_attr(feature = "serde", derive(Serialize, Deserialize))]
290#[cfg_attr(feature = "serde", serde(crate = "actual_serde"))]
291pub struct Sequence(pub u32);
292
293impl Sequence {
294    /// The maximum allowable sequence number.
295    ///
296    /// This sequence number disables absolute lock time and replace-by-fee.
297    pub const MAX: Self = Sequence(0xFFFFFFFF);
298    /// Zero value sequence.
299    ///
300    /// This sequence number enables replace-by-fee and absolute lock time.
301    pub const ZERO: Self = Sequence(0);
302    /// The sequence number that enables absolute lock time but disables replace-by-fee
303    /// and relative lock time.
304    pub const ENABLE_LOCKTIME_NO_RBF: Self = Sequence::MIN_NO_RBF;
305    /// The sequence number that enables replace-by-fee and absolute lock time but
306    /// disables relative lock time.
307    pub const ENABLE_RBF_NO_LOCKTIME: Self = Sequence(0xFFFFFFFD);
308
309    /// The lowest sequence number that does not opt-in for replace-by-fee.
310    ///
311    /// A transaction is considered to have opted in to replacement of itself
312    /// if any of it's inputs have a `Sequence` number less than this value
313    /// (Explicit Signalling [BIP-125]).
314    ///
315    /// [BIP-125]: <https://github.com/bitcoin/bips/blob/master/bip-0125.mediawiki]>
316    const MIN_NO_RBF: Self = Sequence(0xFFFFFFFE);
317    /// BIP-68 relative lock time disable flag mask.
318    const LOCK_TIME_DISABLE_FLAG_MASK: u32 = 0x80000000;
319    /// BIP-68 relative lock time type flag mask.
320    const LOCK_TYPE_MASK: u32 = 0x00400000;
321
322    /// The maximum allowable sequence number.
323    ///
324    /// This is provided for consistency with Rust 1.41.1, newer code should use [`Sequence::MAX`].
325    pub const fn max_value() -> Self { Self::MAX }
326
327    /// Returns `true` if the sequence number enables absolute lock-time ([`Transaction::lock_time`]).
328    #[inline]
329    pub fn enables_absolute_lock_time(&self) -> bool {
330        *self != Sequence::MAX
331    }
332
333    /// Returns `true` if the sequence number indicates that the transaction is finalized.
334    ///
335    /// Instead of this method please consider using `!enables_absolute_lock_time` because it
336    /// is equivalent and improves readability for those not steeped in Bitcoin folklore.
337    ///
338    /// ## Historical note
339    ///
340    /// The term 'final' is an archaic Bitcoin term, it may have come about because the sequence
341    /// number in the original Bitcoin code was intended to be incremented in order to replace a
342    /// transaction, so once the sequence number got to `u64::MAX` it could no longer be increased,
343    /// hence it was 'final'.
344    ///
345    ///
346    /// Some other references to the term:
347    /// - `CTxIn::SEQUENCE_FINAL` in the Bitcoin Core code.
348    /// - [BIP-112]: "BIP 68 prevents a non-final transaction from being selected for inclusion in a
349    ///   block until the corresponding input has reached the specified age"
350    ///
351    /// [BIP-112]: <https://github.com/bitcoin/bips/blob/master/bip-0065.mediawiki>
352    #[inline]
353    pub fn is_final(&self) -> bool {
354        !self.enables_absolute_lock_time()
355    }
356
357    /// Returns true if the transaction opted-in to BIP125 replace-by-fee.
358    ///
359    /// Replace by fee is signaled by the sequence being less than 0xfffffffe which is checked by
360    /// this method. Note, this is the highest "non-final" value (see [`Sequence::is_final`]).
361    #[inline]
362    pub fn is_rbf(&self) -> bool {
363        *self < Sequence::MIN_NO_RBF
364    }
365
366    /// Returns `true` if the sequence has a relative lock-time.
367    #[inline]
368    pub fn is_relative_lock_time(&self) -> bool {
369        self.0 & Sequence::LOCK_TIME_DISABLE_FLAG_MASK == 0
370    }
371
372    /// Returns `true` if the sequence number encodes a block based relative lock-time.
373    #[inline]
374    pub fn is_height_locked(&self) -> bool {
375        self.is_relative_lock_time() & (self.0 & Sequence::LOCK_TYPE_MASK == 0)
376    }
377
378    /// Returns `true` if the sequene number encodes a time interval based relative lock-time.
379    #[inline]
380    pub fn is_time_locked(&self) -> bool {
381        self.is_relative_lock_time() & (self.0 & Sequence::LOCK_TYPE_MASK > 0)
382    }
383
384    /// Creates a relative lock-time using block height.
385    #[inline]
386    pub fn from_height(height: u16) -> Self {
387        Sequence(u32::from(height))
388    }
389
390    /// Creates a relative lock-time using time intervals where each interval is equivalent
391    /// to 512 seconds.
392    ///
393    /// Encoding finer granularity of time for relative lock-times is not supported in Bitcoin
394    #[inline]
395    pub fn from_512_second_intervals(intervals: u16) -> Self {
396        Sequence(u32::from(intervals) | Sequence::LOCK_TYPE_MASK)
397    }
398
399    /// Creates a relative lock-time from seconds, converting the seconds into 512 second
400    /// interval with floor division.
401    ///
402    /// Will return an error if the input cannot be encoded in 16 bits.
403    #[inline]
404    pub fn from_seconds_floor(seconds: u32) -> Result<Self, relative::Error> {
405        if let Ok(interval) = u16::try_from(seconds / 512) {
406            Ok(Sequence::from_512_second_intervals(interval))
407        } else {
408            Err(relative::Error::IntegerOverflow(seconds))
409        }
410    }
411
412    /// Creates a relative lock-time from seconds, converting the seconds into 512 second
413    /// interval with ceiling division.
414    ///
415    /// Will return an error if the input cannot be encoded in 16 bits.
416    #[inline]
417    pub fn from_seconds_ceil(seconds: u32) -> Result<Self, relative::Error> {
418        if let Ok(interval) = u16::try_from((seconds + 511) / 512) {
419            Ok(Sequence::from_512_second_intervals(interval))
420        } else {
421            Err(relative::Error::IntegerOverflow(seconds))
422        }
423    }
424
425    /// Creates a sequence from a u32 value.
426    #[inline]
427    pub fn from_consensus(n: u32) -> Self {
428        Sequence(n)
429    }
430
431    /// Returns the inner 32bit integer value of Sequence.
432    #[inline]
433    pub fn to_consensus_u32(self) -> u32 {
434        self.0
435    }
436
437    /// Creates a [`relative::LockTime`] from this [`Sequence`] number.
438    #[inline]
439    pub fn to_relative_lock_time(&self) -> Option<relative::LockTime> {
440        use crate::locktime::relative::{LockTime, Height, Time};
441
442        if !self.is_relative_lock_time() {
443            return None;
444        }
445
446        let lock_value = self.low_u16();
447
448        if self.is_time_locked() {
449            Some(LockTime::from(Time::from_512_second_intervals(lock_value)))
450        } else {
451            Some(LockTime::from(Height::from(lock_value)))
452        }
453    }
454
455    /// Returns the low 16 bits from sequence number.
456    ///
457    /// BIP-68 only uses the low 16 bits for relative lock value.
458    fn low_u16(&self) -> u16 {
459        self.0 as u16
460    }
461}

505/// Bitcoin transaction output.
506///
507/// Defines new coins to be created as a result of the transaction,
508/// along with spending conditions ("script", aka "output script"),
509/// which an input spending it must satisfy.
510///
511/// An output that is not yet spent by an input is called Unspent Transaction Output ("UTXO").
512///
513/// ### Bitcoin Core References
514///
515/// * [CTxOut definition](https://github.com/bitcoin/bitcoin/blob/345457b542b6a980ccfbc868af0970a6f91d1b82/src/primitives/transaction.h#L148)
516#[derive(Clone, PartialEq, Eq, PartialOrd, Ord, Debug, Hash)]
517#[cfg_attr(feature = "serde", derive(Serialize, Deserialize))]
518#[cfg_attr(feature = "serde", serde(crate = "actual_serde"))]
519pub struct TxOut {
520    /// The value of the output, in satoshis.
521    pub value: u64,
522    /// The script which must be satisfied for the output to be spent.
523    pub script_pubkey: ScriptBuf
524}
525
526impl TxOut {
527    /// The weight of the txout in witness units
528    ///
529    /// Keep in mind that when adding a TxOut to a transaction, the total weight of the transaction
530    /// might increase more than `TxOut::weight`. This happens when the new output added causes
531    /// the output length `VarInt` to increase its encoding length.
532    pub fn weight(&self) -> usize {
533        let script_len = self.script_pubkey.len();
534        // In vbytes:
535        // value (8) + script varint len + script push
536        // Then we multiply by 4 to convert to WU
537        (8 + VarInt(script_len as u64).len() + script_len) * 4
538    }
539
540    /// Creates a `TxOut` with given script and the smallest possible `value` that is **not** dust
541    /// per current Core policy.
542    ///
543    /// The current dust fee rate is 3 sat/vB.
544    pub fn minimal_non_dust(script_pubkey: ScriptBuf) -> Self {
545        let len = script_pubkey.len() + VarInt(script_pubkey.len() as u64).len() + 8;
546        let len = len + if script_pubkey.is_witness_program() {
547            32 + 4 + 1 + (107 / 4) + 4
548        } else {
549            32 + 4 + 1 + 107 + 4
550        };
551        let dust_amount = (len as u64) * 3;
552
553        TxOut {
554            value: dust_amount + 1, // minimal non-dust amount is one higher than dust amount
555            script_pubkey,
556        }
557    }
558}

562    fn default() -> TxOut {
563        TxOut { value: 0xffffffffffffffff, script_pubkey: ScriptBuf::new() }
564    }

631/// Bitcoin transaction.
632///
633/// An authenticated movement of coins.
634///
635/// See [Bitcoin Wiki: Transaction][wiki-transaction] for more information.
636///
637/// [wiki-transaction]: https://en.bitcoin.it/wiki/Transaction
638///
639/// ### Bitcoin Core References
640///
641/// * [CTtransaction definition](https://github.com/bitcoin/bitcoin/blob/345457b542b6a980ccfbc868af0970a6f91d1b82/src/primitives/transaction.h#L279)
642///
643/// ### Serialization notes
644///
645/// If any inputs have nonempty witnesses, the entire transaction is serialized
646/// in the post-BIP141 Segwit format which includes a list of witnesses. If all
647/// inputs have empty witnesses, the transaction is serialized in the pre-BIP141
648/// format.
649///
650/// There is one major exception to this: to avoid deserialization ambiguity,
651/// if the transaction has no inputs, it is serialized in the BIP141 style. Be
652/// aware that this differs from the transaction format in PSBT, which _never_
653/// uses BIP141. (Ordinarily there is no conflict, since in PSBT transactions
654/// are always unsigned and therefore their inputs have empty witnesses.)
655///
656/// The specific ambiguity is that Segwit uses the flag bytes `0001` where an old
657/// serializer would read the number of transaction inputs. The old serializer
658/// would interpret this as "no inputs, one output", which means the transaction
659/// is invalid, and simply reject it. Segwit further specifies that this encoding
660/// should *only* be used when some input has a nonempty witness; that is,
661/// witness-less transactions should be encoded in the traditional format.
662///
663/// However, in protocols where transactions may legitimately have 0 inputs, e.g.
664/// when parties are cooperatively funding a transaction, the "00 means Segwit"
665/// heuristic does not work. Since Segwit requires such a transaction be encoded
666/// in the original transaction format (since it has no inputs and therefore
667/// no input witnesses), a traditionally encoded transaction may have the `0001`
668/// Segwit flag in it, which confuses most Segwit parsers including the one in
669/// Bitcoin Core.
670///
671/// We therefore deviate from the spec by always using the Segwit witness encoding
672/// for 0-input transactions, which results in unambiguously parseable transactions.
673///
674/// ### A note on ordering
675///
676/// This type implements `Ord`, even though it contains a locktime, which is not
677/// itself `Ord`. This was done to simplify applications that may need to hold
678/// transactions inside a sorted container. We have ordered the locktimes based
679/// on their representation as a `u32`, which is not a semantically meaningful
680/// order, and therefore the ordering on `Transaction` itself is not semantically
681/// meaningful either.
682///
683/// The ordering is, however, consistent with the ordering present in this library
684/// before this change, so users should not notice any breakage (here) when
685/// transitioning from 0.29 to 0.30.
686#[derive(Clone, PartialEq, Eq, Debug, Hash)]
687#[cfg_attr(feature = "serde", derive(Serialize, Deserialize))]
688#[cfg_attr(feature = "serde", serde(crate = "actual_serde"))]
689pub struct Transaction {
690    /// The protocol version, is currently expected to be 1 or 2 (BIP 68).
691    pub version: i32,
692    /// Block height or timestamp. Transaction cannot be included in a block until this height/time.
693    ///
694    /// ### Relevant BIPs
695    ///
696    /// * [BIP-65 OP_CHECKLOCKTIMEVERIFY](https://github.com/bitcoin/bips/blob/master/bip-0065.mediawiki)
697    /// * [BIP-113 Median time-past as endpoint for lock-time calculations](https://github.com/bitcoin/bips/blob/master/bip-0113.mediawiki)
698    pub lock_time: absolute::LockTime,
699    /// List of transaction inputs.
700    pub input: Vec<TxIn>,
701    /// List of transaction outputs.
702    pub output: Vec<TxOut>,
703}

714            .then(self.input.cmp(&other.input))

716    }

928    /// Internal utility function for size/weight functions.
929    fn scaled_size(&self, scale_factor: usize) -> usize {
930        let mut input_weight = 0;
931        let mut inputs_with_witnesses = 0;
932        for input in &self.input {
933            input_weight += scale_factor*(32 + 4 + 4 + // outpoint (32+4) + nSequence
934                VarInt(input.script_sig.len() as u64).len() +
935                input.script_sig.len());
936            if !input.witness.is_empty() {
937                inputs_with_witnesses += 1;
938                input_weight += input.witness.serialized_len();
939            }
940        }
941        let mut output_size = 0;
942        for output in &self.output {
943            output_size += 8 + // value
944                VarInt(output.script_pubkey.len() as u64).len() +
945                output.script_pubkey.len();
946        }
947        let non_input_size =
948        // version:
949        4 +
950        // count varints:
951        VarInt(self.input.len() as u64).len() +
952        VarInt(self.output.len() as u64).len() +
953        output_size +
954        // lock_time
955        4;
956        if inputs_with_witnesses == 0 {
957            non_input_size * scale_factor + input_weight
958        } else {
959            non_input_size * scale_factor + input_weight + self.input.len() - inputs_with_witnesses + 2
960        }

1099impl Encodable for Transaction {
1100    fn consensus_encode<W: io::Write + ?Sized>(&self, w: &mut W) -> Result<usize, io::Error> {
1101        let mut len = 0;
1102        len += self.version.consensus_encode(w)?;
1103        // To avoid serialization ambiguity, no inputs means we use BIP141 serialization (see
1104        // `Transaction` docs for full explanation).
1105        let mut have_witness = self.input.is_empty();
1106        for input in &self.input {
1107            if !input.witness.is_empty() {
1108                have_witness = true;
1109                break;
1110            }
1111        }
1112        if !have_witness {
1113            len += self.input.consensus_encode(w)?;
1114            len += self.output.consensus_encode(w)?;
1115        } else {
1116            len += 0u8.consensus_encode(w)?;
1117            len += 1u8.consensus_encode(w)?;
1118            len += self.input.consensus_encode(w)?;
1119            len += self.output.consensus_encode(w)?;
1120            for input in &self.input {
1121                len += input.witness.consensus_encode(w)?;
1122            }
1123        }
1124        len += self.lock_time.consensus_encode(w)?;
1125        Ok(len)
1126    }

1154                }
1155                // We don't support anything else
1156                x => Err(encode::Error::UnsupportedSegwitFlag(x)),

1159        } else {

1194/// Predicts the weight of a to-be-constructed transaction.
1195///
1196/// This function computes the weight of a transaction which is not fully known. All that is needed
1197/// is the lengths of scripts and witness elements.
1198///
1199/// # Arguments
1200///
1201/// * `inputs` - an iterator which returns `InputWeightPrediction` for each input of the
1202///   to-be-constructed transaction.
1203/// * `output_script_lens` - an iterator which returns the length of `script_pubkey` of each output
1204///   of the to-be-constructed transaction.
1205///
1206/// Note that lengths of the scripts and witness elements must be non-serialized, IOW *without* the
1207/// preceding compact size. The lenght of preceding compact size is computed and added inside the
1208/// function for convenience.
1209///
1210/// If you  have the transaction already constructed (except for signatures) with a dummy value for
1211/// fee output you can use the return value of [`Transaction::script_pubkey_lens`] method directly
1212/// as the second argument.
1213///
1214/// # Usage
1215///
1216/// When signing a transaction one doesn't know the signature before knowing the transaction fee and
1217/// the transaction fee is not known before knowing the transaction size which is not known before
1218/// knowing the signature. This apparent dependency cycle can be broken by knowing the length of the
1219/// signature without knowing the contents of the signature e.g., we know all Schnorr signatures
1220/// are 64 bytes long.
1221///
1222/// Additionally, some protocols may require calculating the amounts before knowing various parts
1223/// of the transaction (assuming their length is known).
1224///
1225/// # Notes on integer overflow
1226///
1227/// Overflows are intentionally not checked because one of the following holds:
1228///
1229/// * The transaction is valid (obeys the block size limit) and the code feeds correct values to
1230///   this function - no overflow can happen.
1231/// * The transaction will be so large it doesn't fit in the memory - overflow will happen but
1232///   then the transaction will fail to construct and even if one serialized it on disk directly
1233///   it'd be invalid anyway so overflow doesn't matter.
1234/// * The values fed into this function are inconsistent with the actual lengths the transaction
1235///   will have - the code is already broken and checking overflows doesn't help. Unfortunately
1236///   this probably cannot be avoided.
1237pub fn predict_weight<I, O>(inputs: I, output_script_lens: O) -> Weight
1238    where I: IntoIterator<Item = InputWeightPrediction>,
1239          O: IntoIterator<Item = usize>,
1240{
1241    let (input_count, partial_input_weight, inputs_with_witnesses) = inputs.into_iter()
1242        .fold((0, 0, 0), |(count, partial_input_weight, inputs_with_witnesses), prediction| {
1243            (count + 1, partial_input_weight + prediction.script_size * 4 + prediction.witness_size, inputs_with_witnesses + (prediction.witness_size > 0) as usize)
1244        });
1245    let (output_count, output_scripts_size) = output_script_lens.into_iter()
1246        .fold((0, 0), |(output_count, total_scripts_size), script_len| {
1247                let script_size = script_len + VarInt(script_len as u64).len();
1248                (output_count + 1, total_scripts_size + script_size)
1249        });
1250    predict_weight_internal(input_count, partial_input_weight, inputs_with_witnesses, output_count, output_scripts_size)
1251}
1252
1253crate::internal_macros::maybe_const_fn! {
1254    fn predict_weight_internal(input_count: usize, partial_input_weight: usize, inputs_with_witnesses: usize, output_count: usize, output_scripts_size: usize) -> Weight {
1255        let input_weight = partial_input_weight + input_count * 4 * (32 + 4 + 4);
1256        let output_size = 8 * output_count + output_scripts_size;
1257        let non_input_size =
1258            // version:
1259            4 +
1260            // count varints:
1261            VarInt(input_count as u64).len() +
1262            VarInt(output_count as u64).len() +
1263            output_size +
1264            // lock_time
1265            4;
1266        let weight = if inputs_with_witnesses == 0 {
1267            non_input_size * 4 + input_weight
1268        } else {
1269            non_input_size * 4 + input_weight + input_count - inputs_with_witnesses + 2
1270        };
1271        Weight::from_wu(weight as u64)
1272    }
1273}

1275/// Predicts the weight of a to-be-constructed transaction in const context.
1276///
1277/// *Important: only available in Rust 1.46+*
1278///
1279/// This is a `const` version of [`predict_weight`] which only allows slices due to current Rust
1280/// limitations around `const fn`. Because of these limitations it may be less efficient than
1281/// `predict_weight` and thus is intended to be only used in `const` context.
1282///
1283/// Please see the documentation of `predict_weight` to learn more about this function.
1284#[cfg(rust_v_1_46)]
1285pub const fn predict_weight_from_slices(inputs: &[InputWeightPrediction], output_script_lens: &[usize]) -> Weight {
1286    let mut partial_input_weight = 0;
1287    let mut inputs_with_witnesses = 0;
1288
1289    // for loops not supported in const fn
1290    let mut i = 0;
1291    while i < inputs.len() {
1292        let prediction = inputs[i];
1293        partial_input_weight += prediction.script_size * 4 + prediction.witness_size;
1294        inputs_with_witnesses += (prediction.witness_size > 0) as usize;
1295        i += 1;
1296    }
1297
1298    let mut output_scripts_size = 0;
1299
1300    i = 0;
1301    while i < output_script_lens.len() {
1302        let script_len = output_script_lens[i];
1303        output_scripts_size += script_len + VarInt(script_len as u64).len();
1304        i += 1;
1305    }
1306
1307    predict_weight_internal(inputs.len(), partial_input_weight, inputs_with_witnesses, output_script_lens.len(), output_scripts_size)
1308}

1310/// Weight prediction of an individual input.
1311///
1312/// This helper type collects information about an input to be used in [`predict_weight`] function.
1313/// It can only be created using the [`new`](InputWeightPrediction::new) function.
1314#[derive(Copy, Clone, Debug)]
1315pub struct InputWeightPrediction {
1316    script_size: usize,
1317    witness_size: usize,
1318}
1319
1320impl InputWeightPrediction {
1321    /// Input weight prediction corresponding to spending of P2WPKH output with the largest possible
1322    /// DER-encoded signature.
1323    ///
1324    /// If the input in your transaction uses P2WPKH you can use this instead of
1325    /// [`InputWeightPrediction::new`].
1326    ///
1327    /// This is useful when you **do not** use [signature grinding] and want to ensure you are not
1328    /// under-paying. See [`ground_p2wpkh`](Self::ground_p2wpkh) if you do use signature grinding.
1329    ///
1330    /// [signature grinding]: https://bitcoin.stackexchange.com/questions/111660/what-is-signature-grinding
1331    pub const P2WPKH_MAX: Self = InputWeightPrediction { script_size: 0, witness_size: 1 + 1 + 73 + 1 + 33 };
1332
1333    /// Input weight prediction corresponding to spending of taproot output using the key and
1334    /// default sighash.
1335    ///
1336    /// If the input in your transaction uses Taproot key spend you can use this instead of
1337    /// [`InputWeightPrediction::new`].
1338    pub const P2TR_KEY_DEFAULT_SIGHASH: Self = InputWeightPrediction { script_size: 0, witness_size: 1 + 1 + 64 };
1339
1340    /// Input weight prediction corresponding to spending of taproot output using the key and
1341    /// **non**-default sighash.
1342    ///
1343    /// If the input in your transaction uses Taproot key spend you can use this instead of
1344    /// [`InputWeightPrediction::new`].
1345    pub const P2TR_KEY_NON_DEFAULT_SIGHASH: Self = InputWeightPrediction { script_size: 0, witness_size: 1 + 1 + 65 };
1346
1347    /// Input weight prediction corresponding to spending of P2WPKH output using [signature
1348    /// grinding].
1349    ///
1350    /// If the input in your transaction uses P2WPKH and you use signature grinding you can use this
1351    /// instead of [`InputWeightPrediction::new`]. See [`P2WPKH_MAX`](Self::P2WPKH_MAX) if you don't
1352    /// use signature grinding.
1353    ///
1354    /// Note: `bytes_to_grind` is usually `1` because of exponential cost of higher values.
1355    ///
1356    /// # Panics
1357    ///
1358    /// The funcion panics in const context and debug builds if `bytes_to_grind` is higher than 62.
1359    ///
1360    /// [signature grinding]: https://bitcoin.stackexchange.com/questions/111660/what-is-signature-grinding
1361    pub const fn ground_p2wpkh(bytes_to_grind: usize) -> Self {
1362        // Written to trigger const/debug panic for unreasonably high values.
1363        let der_signature_size = 10 + (62 - bytes_to_grind);
1364        let witness_size = 1 // length of element count varint
1365            + 1 // length of element size varint (max signature length is 73B)
1366            + der_signature_size
1367            + 1 // sighash flag
1368            + 1 // length of element size varint
1369            + 33; // length of (always-compressed) public key
1370        InputWeightPrediction { script_size: 0, witness_size }
1371    }
1372
1373    /// Computes the prediction for a single input.
1374    pub fn new<T>(input_script_len: usize, witness_element_lengths: T) -> Self
1375        where T :IntoIterator, T::Item:Borrow<usize>,
1376    {
1377        let (count, total_size) = witness_element_lengths.into_iter()
1378            .fold((0, 0), |(count, total_size), elem_len| {
1379                let elem_len = *elem_len.borrow();
1380                let elem_size = elem_len + VarInt(elem_len as u64).len();
1381                (count + 1, total_size + elem_size)
1382            });
1383        let witness_size = if count > 0 {
1384            total_size + VarInt(count as u64).len()
1385        } else {
1386            0
1387        };
1388        let script_size = input_script_len + VarInt(input_script_len as u64).len();
1389
1390        InputWeightPrediction {
1391            script_size,
1392            witness_size,
1393        }
1394    }

1396    /// Computes the prediction for a single input in `const` context.
1397    ///
1398    /// *Important: only available in Rust 1.46+*
1399    ///
1400    /// This is a `const` version of [`new`](Self::new) which only allows slices due to current Rust
1401    /// limitations around `const fn`. Because of these limitations it may be less efficient than
1402    /// `new` and thus is intended to be only used in `const` context.
1403    #[cfg(rust_v_1_46)]
1404    pub const fn from_slice(input_script_len: usize, witness_element_lengths: &[usize]) -> Self {
1405        let mut i = 0;
1406        let mut total_size = 0;
1407        // for loops not supported in const fn
1408        while i < witness_element_lengths.len() {
1409            let elem_len = witness_element_lengths[i];
1410            let elem_size = elem_len + VarInt(elem_len as u64).len();
1411            total_size += elem_size;
1412            i += 1;
1413        }
1414        let witness_size = if !witness_element_lengths.is_empty() {
1415            total_size + VarInt(witness_element_lengths.len() as u64).len()
1416        } else {
1417            0
1418        };
1419        let script_size = input_script_len + VarInt(input_script_len as u64).len();
1420
1421        InputWeightPrediction {
1422            script_size,
1423            witness_size,
1424        }
1425    }
