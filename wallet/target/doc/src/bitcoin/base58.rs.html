38/// Decodes a base58-encoded string into a byte vector.
39#[deprecated(since = "0.30.0", note = "Use base58::decode() instead")]
40pub fn from(data: &str) -> Result<Vec<u8>, Error> { decode(data) }
41
42/// Decodes a base58-encoded string into a byte vector.
43pub fn decode(data: &str) -> Result<Vec<u8>, Error> {
44    // 11/15 is just over log_256(58)
45    let mut scratch = vec![0u8; 1 + data.len() * 11 / 15];
46    // Build in base 256
47    for d58 in data.bytes() {
48        // Compute "X = X * 58 + next_digit" in base 256
49        if d58 as usize >= BASE58_DIGITS.len() {
50            return Err(Error::BadByte(d58));
51        }
52        let mut carry = match BASE58_DIGITS[d58 as usize] {
53            Some(d58) => d58 as u32,
54            None => {
55                return Err(Error::BadByte(d58));
56            }
57        };
58        for d256 in scratch.iter_mut().rev() {
59            carry += *d256 as u32 * 58;
60            *d256 = carry as u8;
61            carry /= 256;
62        }
63        assert_eq!(carry, 0);
64    }
65
66    // Copy leading zeroes directly
67    let mut ret: Vec<u8> = data.bytes().take_while(|&x| x == BASE58_CHARS[0]).map(|_| 0).collect();
68    // Copy rest of string
69    ret.extend(scratch.into_iter().skip_while(|&x| x == 0));
70    Ok(ret)
71}

73/// Decodes a base58check-encoded string into a byte vector verifying the checksum.
74#[deprecated(since = "0.30.0", note = "Use base58::decode_check() instead")]
75pub fn from_check(data: &str) -> Result<Vec<u8>, Error> { decode_check(data) }

77/// Decodes a base58check-encoded string into a byte vector verifying the checksum.
78pub fn decode_check(data: &str) -> Result<Vec<u8>, Error> {
79    let mut ret: Vec<u8> = decode(data)?;
80    if ret.len() < 4 {
81        return Err(Error::TooShort(ret.len()));
82    }
83    let check_start = ret.len() - 4;
84
85    let hash_check =
86        sha256d::Hash::hash(&ret[..check_start])[..4].try_into().expect("4 byte slice");
87    let data_check = ret[check_start..].try_into().expect("4 byte slice");
88
89    let expected = u32::from_le_bytes(hash_check);
90    let actual = u32::from_le_bytes(data_check);
91
92    if expected != actual {
93        return Err(Error::BadChecksum(expected, actual));
94    }
95
96    ret.truncate(check_start);
97    Ok(ret)
98}

108///

111pub fn check_encode_slice(data: &[u8]) -> String { encode_check(data) }

113/// Encodes `data` as a base58 string including the checksum.
114///
115/// The checksum is the first four bytes of the sha256d of the data, concatenated onto the end.
116pub fn encode_check(data: &[u8]) -> String {
117    let checksum = sha256d::Hash::hash(data);
118    encode_iter(data.iter().cloned().chain(checksum[0..4].iter().cloned()))
119}

129/// Encodes a slice as base58, including the checksum, into a formatter.
130///
131/// The checksum is the first four bytes of the sha256d of the data, concatenated onto the end.
132pub fn encode_check_to_fmt(fmt: &mut fmt::Formatter, data: &[u8]) -> fmt::Result {
133    let checksum = sha256d::Hash::hash(data);
134    let iter = data.iter().cloned().chain(checksum[0..4].iter().cloned());
135    format_iter(fmt, iter)
136}

138fn encode_iter<I>(data: I) -> String
139where
140    I: Iterator<Item = u8> + Clone,
141{
142    let mut ret = String::new();
143    format_iter(&mut ret, data).expect("writing into string shouldn't fail");
144    ret
145}
146
147fn format_iter<I, W>(writer: &mut W, data: I) -> Result<(), fmt::Error>
148where
149    I: Iterator<Item = u8> + Clone,
150    W: fmt::Write,
151{
152    let mut ret = SmallVec::new();
153
154    let mut leading_zero_count = 0;
155    let mut leading_zeroes = true;
156    // Build string in little endian with 0-58 in place of characters...
157    for d256 in data {
158        let mut carry = d256 as usize;
159        if leading_zeroes && carry == 0 {
160            leading_zero_count += 1;
161        } else {
162            leading_zeroes = false;
163        }
164
165        for ch in ret.iter_mut() {
166            let new_ch = *ch as usize * 256 + carry;
167            *ch = (new_ch % 58) as u8;
168            carry = new_ch / 58;
169        }
170        while carry > 0 {
171            ret.push((carry % 58) as u8);
172            carry /= 58;
173        }
174    }
175
176    // ... then reverse it and convert to chars
177    for _ in 0..leading_zero_count {
178        ret.push(0);
179    }
180
181    for ch in ret.iter().rev() {
182        writer.write_char(BASE58_CHARS[*ch as usize] as char)?;
183    }
184
185    Ok(())
186}

188/// Vector-like object that holds the first 100 elements on the stack. If more space is needed it
189/// will be allocated on the heap.
190struct SmallVec<T> {
191    len: usize,
192    stack: [T; 100],
193    heap: Vec<T>,
194}
195
196impl<T: Default + Copy> SmallVec<T> {
197    fn new() -> SmallVec<T> { SmallVec { len: 0, stack: [T::default(); 100], heap: Vec::new() } }
198
199    fn push(&mut self, val: T) {
200        if self.len < 100 {
201            self.stack[self.len] = val;
202            self.len += 1;
203        } else {
204            self.heap.push(val);
205        }
206    }
207
208    fn iter(&self) -> iter::Chain<slice::Iter<T>, slice::Iter<T>> {
209        // If len<100 then we just append an empty vec
210        self.stack[0..self.len].iter().chain(self.heap.iter())
211    }
212
213    fn iter_mut(&mut self) -> iter::Chain<slice::IterMut<T>, slice::IterMut<T>> {
214        // If len<100 then we just append an empty vec
215        self.stack[0..self.len].iter_mut().chain(self.heap.iter_mut())
216    }
217}

219/// An error that might occur during base58 decoding.
220#[derive(Debug, PartialEq, Eq, PartialOrd, Ord, Hash, Clone)]
221#[non_exhaustive]
222pub enum Error {
223    /// Invalid character encountered.
224    BadByte(u8),
225    /// Checksum was not correct (expected, actual).
226    BadChecksum(u32, u32),
227    /// The length (in bytes) of the object was not correct.
228    ///
229    /// Note that if the length is excessively long the provided length may be an estimate (and the
230    /// checksum step may be skipped).
231    InvalidLength(usize),
232    /// Extended Key version byte(s) were not recognized.
233    InvalidExtendedKeyVersion([u8; 4]),
234    /// Address version byte were not recognized.
235    InvalidAddressVersion(u8),
236    /// Checked data was less than 4 bytes.
237    TooShort(usize),
238}
239
240impl fmt::Display for Error {
241    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
242        match *self {
243            Error::BadByte(b) => write!(f, "invalid base58 character {:#x}", b),
244            Error::BadChecksum(exp, actual) =>
245                write!(f, "base58ck checksum {:#x} does not match expected {:#x}", actual, exp),
246            Error::InvalidLength(ell) => write!(f, "length {} invalid for this base58 type", ell),
247            Error::InvalidExtendedKeyVersion(ref v) =>
248                write!(f, "extended key version {:#04x?} is invalid for this base58 type", v),
249            Error::InvalidAddressVersion(ref v) =>
250                write!(f, "address version {} is invalid for this base58 type", v),
251            Error::TooShort(_) => write!(f, "base58ck data not even long enough for a checksum"),
252        }
253    }
254}

256#[cfg(feature = "std")]
257#[cfg_attr(docsrs, doc(cfg(feature = "std")))]
258impl std::error::Error for Error {
259    fn source(&self) -> Option<&(dyn std::error::Error + 'static)> {
260        use self::Error::*;
261
262        match self {
263            BadByte(_)
264            | BadChecksum(_, _)
265            | InvalidLength(_)
266            | InvalidExtendedKeyVersion(_)
267            | InvalidAddressVersion(_)
268            | TooShort(_) => None,
269        }
270    }
271}

328    #[test]
329    fn test_base58_roundtrip() {
330        let s = "xprv9wTYmMFdV23N2TdNG573QoEsfRrWKQgWeibmLntzniatZvR9BmLnvSxqu53Kw1UmYPxLgboyZQaXwTCg8MSY3H2EU4pWcQDnRnrVA1xe8fs";
331        let v: Vec<u8> = decode_check(s).unwrap();
332        assert_eq!(encode_check(&v[..]), s);
333        assert_eq!(decode_check(&encode_check(&v[..])).ok(), Some(v));
334
335        // Check that empty slice passes roundtrip.
336        assert_eq!(decode_check(&encode_check(&[])), Ok(vec![]));
337        // Check that `len > 4` is enforced.
338        assert_eq!(decode_check(&encode(&[1, 2, 3])), Err(Error::TooShort(3)));
339    }
