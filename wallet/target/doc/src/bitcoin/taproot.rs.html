62impl TapTweakHash {
63    /// Creates a new BIP341 [`TapTweakHash`] from key and tweak. Produces `H_taptweak(P||R)` where
64    /// `P` is the internal key and `R` is the merkle root.
65    pub fn from_key_and_tweak(
66        internal_key: UntweakedPublicKey,
67        merkle_root: Option<TapNodeHash>,
68    ) -> TapTweakHash {
69        let mut eng = TapTweakHash::engine();
70        // always hash the key
71        eng.input(&internal_key.serialize());
72        if let Some(h) = merkle_root {
73            eng.input(h.as_ref());
74        } else {
75            // nothing to hash
76        }
77        TapTweakHash::from_engine(eng)
78    }

83        Scalar::from_be_bytes(self.to_byte_array()).expect("hash value greater than curve order")

113    fn combine_node_hashes(a: TapNodeHash, b: TapNodeHash) -> (TapNodeHash, bool) {
114        let mut eng = TapNodeHash::engine();
115        if a < b {
116            eng.input(a.as_ref());
117            eng.input(b.as_ref());
118        } else {
119            eng.input(b.as_ref());
120            eng.input(a.as_ref());
121        };
122        (TapNodeHash::from_engine(eng), a < b)
123    }

167/// Represents taproot spending information.
168///
169/// Taproot output corresponds to a combination of a single public key condition (known as the
170/// internal key), and zero or more general conditions encoded in scripts organized in the form of a
171/// binary tree.
172///
173/// Taproot can be spent by either:
174/// - Spending using the key path i.e., with secret key corresponding to the tweaked `output_key`.
175/// - By satisfying any of the scripts in the script spend path. Each script can be satisfied by
176///   providing a witness stack consisting of the script's inputs, plus the script itself and the
177///   control block.
178///
179/// If one or more of the spending conditions consist of just a single key (after aggregation), the
180/// most likely key should be made the internal key.
181/// See [BIP341](https://github.com/bitcoin/bips/blob/master/bip-0341.mediawiki) for more details on
182/// choosing internal keys for a taproot application.
183///
184/// Note: This library currently does not support
185/// [annex](https://github.com/bitcoin/bips/blob/master/bip-0341.mediawiki#cite_note-5).
186#[derive(Debug, Clone, PartialEq, Eq, PartialOrd, Ord, Hash)]
187pub struct TaprootSpendInfo {
188    /// The BIP341 internal key.
189    internal_key: UntweakedPublicKey,
190    /// The merkle root of the script tree (None if there are no scripts).
191    merkle_root: Option<TapNodeHash>,
192    /// The sign final output pubkey as per BIP 341.
193    output_key_parity: secp256k1::Parity,
194    /// The tweaked output key.
195    output_key: TweakedPublicKey,
196    /// Map from (script, leaf_version) to (sets of) [`TaprootMerkleBranch`]. More than one control
197    /// block for a given script is only possible if it appears in multiple branches of the tree. In
198    /// all cases, keeping one should be enough for spending funds, but we keep all of the paths so
199    /// that a full tree can be constructed again from spending data if required.
200    script_map: ScriptMerkleProofMap,
201}

222    /// the `merkle_root` if there is no script path.
223    ///
224    /// *Note*: As per BIP341
225    ///
226    /// When the merkle root is [`None`], the output key commits to an unspendable script path
227    /// instead of having no script path. This is achieved by computing the output key point as
228    /// `Q = P + int(hashTapTweak(bytes(P)))G`. See also [`TaprootSpendInfo::tap_tweak`].

239            internal_key,

265    /// Computes the [`TaprootSpendInfo`] from `internal_key` and `node`.
266    ///
267    /// This is useful when you want to manually build a taproot tree without using
268    /// [`TaprootBuilder`].
269    pub fn from_node_info<C: secp256k1::Verification>(
270        secp: &Secp256k1<C>,
271        internal_key: UntweakedPublicKey,
272        node: NodeInfo,
273    ) -> TaprootSpendInfo {
274        // Create as if it is a key spend path with the given merkle root
275        let root_hash = Some(node.hash);
276        let mut info = TaprootSpendInfo::new_key_spend(secp, internal_key, root_hash);
277
278        for leaves in node.leaves {
279            match leaves.leaf {
280                TapLeaf::Hidden(_) => {
281                    // We don't store any information about hidden nodes in TaprootSpendInfo.
282                }
283                TapLeaf::Script(script, ver) => {
284                    let key = (script, ver);
285                    let value = leaves.merkle_branch;
286                    match info.script_map.get_mut(&key) {
287                        None => {
288                            let mut set = BTreeSet::new();
289                            set.insert(value);
290                            info.script_map.insert(key, set);
291                        }
292                        Some(set) => {
293                            set.insert(value);
294                        }
295                    }
296                }
297            }
298        }
299        info
300    }

305    /// Constructs a [`ControlBlock`] for particular script with the given version.
306    ///
307    /// # Returns
308    ///
309    /// - If there are multiple control blocks possible, returns the shortest one.
310    /// - If the script is not contained in the [`TaprootSpendInfo`], returns `None`.
311    pub fn control_block(&self, script_ver: &(ScriptBuf, LeafVersion)) -> Option<ControlBlock> {
312        let merkle_branch_set = self.script_map.get(script_ver)?;
313        // Choose the smallest one amongst the multiple script maps
314        let smallest = merkle_branch_set
315            .iter()
316            .min_by(|x, y| x.0.len().cmp(&y.0.len()))
317            .expect("Invariant: ScriptBuf map key must contain non-empty set value");
318        Some(ControlBlock {
319            internal_key: self.internal_key,
320            output_key_parity: self.output_key_parity,
321            leaf_version: script_ver.1,
322            merkle_branch: smallest.clone(),
323        })
324    }

335/// Builder for building taproot iteratively. Users can specify tap leaf or omitted/hidden branches
336/// in a depth-first search (DFS) walk order to construct this tree.
337///
338/// See Wikipedia for more details on [DFS](https://en.wikipedia.org/wiki/Depth-first_search).
339// Similar to Taproot Builder in Bitcoin Core.
340#[derive(Debug, Clone, PartialEq, Eq, PartialOrd, Ord, Hash)]
341pub struct TaprootBuilder {
342    // The following doc-comment is from Bitcoin Core, but modified for Rust. It describes the
343    // current state of the builder for a given tree.
344    //
345    // For each level in the tree, one NodeInfo object may be present. Branch at index 0 is
346    // information about the root; further values are for deeper subtrees being explored.
347    //
348    // During the construction of Taptree, for every right branch taken to reach the position we're
349    // currently working on, there will be a `(Some(_))` entry in branch corresponding to the left
350    // branch at that level.
351    //
352    // For example, imagine this tree:     - N0 -
353    //                                    /      \
354    //                                   N1      N2
355    //                                  /  \    /  \
356    //                                 A    B  C   N3
357    //                                            /  \
358    //                                           D    E
359    //
360    // Initially, branch is empty. After processing leaf A, it would become {None, None, A}. When
361    // processing leaf B, an entry at level 2 already exists, and it would thus be combined with it
362    // to produce a level 1 entry, resulting in {None, N1}. Adding C and D takes us to {None, N1, C}
363    // and {None, N1, C, D} respectively. When E is processed, it is combined with D, and then C,
364    // and then N1, to produce the root, resulting in {N0}.
365    //
366    // This structure allows processing with just O(log n) overhead if the leaves are computed on
367    // the fly.
368    //
369    // As an invariant, there can never be None entries at the end. There can also not be more than
370    // 128 entries (as that would mean more than 128 levels in the tree). The depth of newly added
371    // entries will always be at least equal to the current size of branch (otherwise it does not
372    // correspond to a depth-first traversal of a tree). A branch is only empty if no entries have
373    // ever be processed. A branch having length 1 corresponds to being done.
374    branch: Vec<Option<NodeInfo>>,
375}

388    /// Creates a new [`TaprootSpendInfo`] from a list of scripts (with default script version) and
389    /// weights of satisfaction for that script.
390    ///
391    /// The weights represent the probability of each branch being taken. If probabilities/weights
392    /// for each condition are known, constructing the tree as a Huffman Tree is the optimal way to
393    /// minimize average case satisfaction cost. This function takes as input an iterator of
394    /// `tuple(u32, ScriptBuf)` where `u32` represents the satisfaction weights of the branch. For
395    /// example, [(3, S1), (2, S2), (5, S3)] would construct a [`TapTree`] that has optimal
396    /// satisfaction weight when probability for S1 is 30%, S2 is 20% and S3 is 50%.
397    ///
398    /// # Errors:
399    ///
400    /// - When the optimal Huffman Tree has a depth more than 128.
401    /// - If the provided list of script weights is empty.
402    ///
403    /// # Edge Cases:
404    ///
405    /// If the script weight calculations overflow, a sub-optimal tree may be generated. This should
406    /// not happen unless you are dealing with billions of branches with weights close to 2^32.
407    ///
408    /// [`TapTree`]: crate::taproot::TapTree
409    pub fn with_huffman_tree<I>(script_weights: I) -> Result<Self, TaprootBuilderError>
410    where
411        I: IntoIterator<Item = (u32, ScriptBuf)>,
412    {
413        let mut node_weights = BinaryHeap::<(Reverse<u32>, NodeInfo)>::new();
414        for (p, leaf) in script_weights {
415            node_weights
416                .push((Reverse(p), NodeInfo::new_leaf_with_ver(leaf, LeafVersion::TapScript)));
417        }
418        if node_weights.is_empty() {
419            return Err(TaprootBuilderError::EmptyTree);
420        }
421        while node_weights.len() > 1 {
422            // Combine the last two elements and insert a new node
423            let (p1, s1) = node_weights.pop().expect("len must be at least two");
424            let (p2, s2) = node_weights.pop().expect("len must be at least two");
425            // Insert the sum of first two in the tree as a new node
426            // N.B.: p1 + p2 can not practically saturate as you would need to have 2**32 max u32s
427            // from the input to overflow. However, saturating is a reasonable behavior here as
428            // huffman tree construction would treat all such elements as "very likely".
429            let p = Reverse(p1.0.saturating_add(p2.0));
430            node_weights.push((p, NodeInfo::combine(s1, s2)?));
431        }
432        // Every iteration of the loop reduces the node_weights.len() by exactly 1
433        // Therefore, the loop will eventually terminate with exactly 1 element
434        debug_assert_eq!(node_weights.len(), 1);
435        let node = node_weights.pop().expect("huffman tree algorithm is broken").1;
436        Ok(TaprootBuilder { branch: vec![Some(node)] })
437    }

473    /// Converts the builder into a [`NodeInfo`] if the builder is a full tree with possibly
474    /// hidden nodes
475    ///
476    /// # Errors:
477    ///
478    /// [`IncompleteBuilder::NotFinalized`] if the builder is not finalized. The builder
479    /// can be restored by calling [`IncompleteBuilder::into_builder`]
480    pub fn try_into_node_info(mut self) -> Result<NodeInfo, IncompleteBuilder> {
481        if self.branch().len() != 1 {
482            return Err(IncompleteBuilder::NotFinalized(self));
483        }
484        Ok(self
485            .branch
486            .pop()
487            .expect("length checked above")
488            .expect("invariant guarantees node info exists"))
489    }

491    /// Converts the builder into a [`TapTree`] if the builder is a full tree and
492    /// does not contain any hidden nodes
493    pub fn try_into_taptree(self) -> Result<TapTree, IncompleteBuilder> {
494        let node = self.try_into_node_info()?;
495        if node.has_hidden_nodes {
496            // Reconstruct the builder as it was if it has hidden nodes
497            return Err(IncompleteBuilder::HiddenParts(TaprootBuilder {
498                branch: vec![Some(node)],
499            }));
500        }
501        Ok(TapTree(node))
502    }

509    /// Creates a [`TaprootSpendInfo`] with the given internal key.
510    ///
511    /// Returns the unmodified builder as Err if the builder is not finalizable.
512    /// See also [`TaprootBuilder::is_finalizable`]
513    pub fn finalize<C: secp256k1::Verification>(
514        mut self,
515        secp: &Secp256k1<C>,
516        internal_key: UntweakedPublicKey,
517    ) -> Result<TaprootSpendInfo, TaprootBuilder> {
518        match self.branch.len() {
519            0 => Ok(TaprootSpendInfo::new_key_spend(secp, internal_key, None)),
520            1 =>
521                if let Some(Some(node)) = self.branch.pop() {
522                    Ok(TaprootSpendInfo::from_node_info(secp, internal_key, node))
523                } else {
524                    unreachable!("Size checked above. Builder guarantees the last element is Some")
525                },
526            _ => Err(self),
527        }
528    }
529
530    pub(crate) fn branch(&self) -> &[Option<NodeInfo>] { &self.branch }
531
532    /// Inserts a leaf at `depth`.
533    fn insert(mut self, mut node: NodeInfo, mut depth: u8) -> Result<Self, TaprootBuilderError> {
534        // early error on invalid depth. Though this will be checked later
535        // while constructing TaprootMerkelBranch
536        if depth as usize > TAPROOT_CONTROL_MAX_NODE_COUNT {
537            return Err(TaprootBuilderError::InvalidMerkleTreeDepth(depth as usize));
538        }
539        // We cannot insert a leaf at a lower depth while a deeper branch is unfinished. Doing
540        // so would mean the add_leaf/add_hidden invocations do not correspond to a DFS traversal of a
541        // binary tree.
542        if (depth as usize + 1) < self.branch.len() {
543            return Err(TaprootBuilderError::NodeNotInDfsOrder);
544        }
545
546        while self.branch.len() == depth as usize + 1 {
547            let child = match self.branch.pop() {
548                None => unreachable!("Len of branch checked to be >= 1"),
549                Some(Some(child)) => child,
550                // Needs an explicit push to add the None that we just popped.
551                // Cannot use .last() because of borrow checker issues.
552                Some(None) => {
553                    self.branch.push(None);
554                    break;
555                } // Cannot combine further
556            };
557            if depth == 0 {
558                // We are trying to combine two nodes at root level.
559                // Can't propagate further up than the root
560                return Err(TaprootBuilderError::OverCompleteTree);
561            }
562            node = NodeInfo::combine(node, child)?;
563            // Propagate to combine nodes at a lower depth
564            depth -= 1;
565        }
566
567        if self.branch.len() < depth as usize + 1 {
568            // add enough nodes so that we can insert node at depth `depth`
569            let num_extra_nodes = depth as usize + 1 - self.branch.len();
570            self.branch.extend((0..num_extra_nodes).map(|_| None));
571        }
572        // Push the last node to the branch
573        self.branch[depth as usize] = Some(node);
574        Ok(self)
575    }
576}

777/// Represents the node information in taproot tree. In contrast to [`TapTree`], this
778/// is allowed to have hidden leaves as children.
779///
780/// Helper type used in merkle tree construction allowing one to build sparse merkle trees. The node
781/// represents part of the tree that has information about all of its descendants.
782/// See how [`TaprootBuilder`] works for more details.
783///
784/// You can use [`TaprootSpendInfo::from_node_info`] to a get a [`TaprootSpendInfo`] from the merkle
785/// root [`NodeInfo`].
786#[derive(Debug, Clone, PartialOrd, Ord)]
787pub struct NodeInfo {
788    /// Merkle hash for this node.
789    pub(crate) hash: TapNodeHash,
790    /// Information about leaves inside this node.
791    pub(crate) leaves: Vec<LeafNode>,
792    /// Tracks information on hidden nodes below this node.
793    pub(crate) has_hidden_nodes: bool,
794}
795
796impl PartialEq for NodeInfo {
797    fn eq(&self, other: &Self) -> bool { self.hash.eq(&other.hash) }
798}
799
800impl core::hash::Hash for NodeInfo {
801    fn hash<H: core::hash::Hasher>(&self, state: &mut H) { self.hash.hash(state) }
802}
803
804impl Eq for NodeInfo {}
805
806impl NodeInfo {
807    /// Creates a new [`NodeInfo`] with omitted/hidden info.
808    pub fn new_hidden_node(hash: TapNodeHash) -> Self {
809        Self { hash, leaves: vec![], has_hidden_nodes: true }
810    }
811
812    /// Creates a new leaf [`NodeInfo`] with given [`ScriptBuf`] and [`LeafVersion`].
813    pub fn new_leaf_with_ver(script: ScriptBuf, ver: LeafVersion) -> Self {
814        Self {
815            hash: TapNodeHash::from_script(&script, ver),
816            leaves: vec![LeafNode::new_script(script, ver)],
817            has_hidden_nodes: false,
818        }
819    }
820
821    /// Combines two [`NodeInfo`] to create a new parent.
822    pub fn combine(a: Self, b: Self) -> Result<Self, TaprootBuilderError> {
823        let mut all_leaves = Vec::with_capacity(a.leaves.len() + b.leaves.len());
824        let (hash, left_first) = TapNodeHash::combine_node_hashes(a.hash, b.hash);
825        let (a, b) = if left_first { (a, b) } else { (b, a) };
826        for mut a_leaf in a.leaves {
827            a_leaf.merkle_branch.push(b.hash)?; // add hashing partner
828            all_leaves.push(a_leaf);
829        }
830        for mut b_leaf in b.leaves {
831            b_leaf.merkle_branch.push(a.hash)?; // add hashing partner
832            all_leaves.push(b_leaf);
833        }
834        Ok(Self {
835            hash,
836            leaves: all_leaves,
837            has_hidden_nodes: a.has_hidden_nodes || b.has_hidden_nodes,
838        })
839    }
840
841    /// Creates an iterator over all leaves (including hidden leaves) in the tree.
842    pub fn leaf_nodes(&self) -> LeafNodes { LeafNodes { leaf_iter: self.leaves.iter() } }
843}

850    }

870#[cfg(feature = "serde")]
871#[cfg_attr(docsrs, doc(cfg(feature = "serde")))]
872impl<'de> serde::Deserialize<'de> for NodeInfo {
873    fn deserialize<D>(deserializer: D) -> Result<Self, D::Error>
874    where
875        D: serde::Deserializer<'de>,
876    {
877        struct SeqVisitor;
878        impl<'de> serde::de::Visitor<'de> for SeqVisitor {
879            type Value = NodeInfo;
880
881            fn expecting(&self, formatter: &mut fmt::Formatter) -> fmt::Result {
882                formatter.write_str("Taproot tree in DFS walk order")
883            }
884
885            fn visit_seq<A>(self, mut seq: A) -> Result<Self::Value, A::Error>
886            where
887                A: serde::de::SeqAccess<'de>,
888            {
889                let size = seq
890                    .size_hint()
891                    .map(|x| core::mem::size_of::<usize>() * 8 - x.leading_zeros() as usize)
892                    .map(|x| x / 2) // Each leaf is serialized as two elements.
893                    .unwrap_or(0)
894                    .min(TAPROOT_CONTROL_MAX_NODE_COUNT); // no more than 128 nodes
895                let mut builder = TaprootBuilder::with_capacity(size);
896                while let Some(depth) = seq.next_element()? {
897                    let tap_leaf: TapLeaf = seq
898                        .next_element()?
899                        .ok_or_else(|| serde::de::Error::custom("Missing tap_leaf"))?;
900                    match tap_leaf {
901                        TapLeaf::Script(script, ver) => {
902                            builder =
903                                builder.add_leaf_with_ver(depth, script, ver).map_err(|e| {
904                                    serde::de::Error::custom(format!("Leaf insertion error: {}", e))
905                                })?;
906                        }
907                        TapLeaf::Hidden(h) => {
908                            builder = builder.add_hidden_node(depth, h).map_err(|e| {
909                                serde::de::Error::custom(format!(
910                                    "Hidden node insertion error: {}",
911                                    e
912                                ))
913                            })?;
914                        }
915                    }
916                }
917                NodeInfo::try_from(builder).map_err(|e| {
918                    serde::de::Error::custom(format!("Incomplete taproot tree: {}", e))
919                })
920            }
921        }
922
923        deserializer.deserialize_seq(SeqVisitor)
924    }
925}

1056/// The merkle proof for inclusion of a tree in a taptree hash.
1057#[derive(Debug, Clone, PartialEq, Eq, PartialOrd, Ord, Hash, Default)]
1058#[cfg_attr(feature = "serde", derive(Serialize, Deserialize))]
1059#[cfg_attr(feature = "serde", serde(crate = "actual_serde"))]
1060#[cfg_attr(feature = "serde", serde(into = "Vec<TapNodeHash>"))]
1061#[cfg_attr(feature = "serde", serde(try_from = "Vec<TapNodeHash>"))]
1062pub struct TaprootMerkleBranch(Vec<TapNodeHash>);
1063
1064impl TaprootMerkleBranch {
1065    /// Returns a reference to the inner vector of hashes.
1066    pub fn as_inner(&self) -> &[TapNodeHash] { &self.0 }
1067
1068    /// Returns the number of nodes in this merkle proof.
1069    pub fn len(&self) -> usize { self.0.len() }
1070
1071    /// Checks if this merkle proof is empty.
1072    pub fn is_empty(&self) -> bool { self.0.is_empty() }
1073
1074    /// Decodes bytes from control block.
1075    #[deprecated(since = "0.30.0", note = "Use decode instead")]
1076    pub fn from_slice(sl: &[u8]) -> Result<Self, TaprootError> { Self::decode(sl) }
1077
1078    /// Decodes bytes from control block.
1079    ///
1080    /// This reads the branch as encoded in the control block: the concatenated 32B byte chunks -
1081    /// one for each hash.
1082    ///
1083    /// # Errors
1084    ///
1085    /// The function returns an error if the the number of bytes is not an integer multiple of 32 or
1086    /// if the number of hashes exceeds 128.
1087    pub fn decode(sl: &[u8]) -> Result<Self, TaprootError> {
1088        if sl.len() % TAPROOT_CONTROL_NODE_SIZE != 0 {
1089            Err(TaprootError::InvalidMerkleBranchSize(sl.len()))
1090        } else if sl.len() > TAPROOT_CONTROL_NODE_SIZE * TAPROOT_CONTROL_MAX_NODE_COUNT {
1091            Err(TaprootError::InvalidMerkleTreeDepth(sl.len() / TAPROOT_CONTROL_NODE_SIZE))
1092        } else {
1093            let inner = sl
1094                .chunks_exact(TAPROOT_CONTROL_NODE_SIZE)
1095                .map(|chunk| {
1096                    TapNodeHash::from_slice(chunk)
1097                        .expect("chunks_exact always returns the correct size")
1098                })
1099                .collect();
1100
1101            Ok(TaprootMerkleBranch(inner))
1102        }
1103    }
1104
1105    /// Creates a merkle proof from list of hashes.

1197/// Control block data structure used in Tapscript satisfaction.
1198#[derive(Debug, Clone, PartialEq, Eq, PartialOrd, Ord, Hash)]
1199#[cfg_attr(feature = "serde", derive(Serialize, Deserialize))]
1200#[cfg_attr(feature = "serde", serde(crate = "actual_serde"))]
1201pub struct ControlBlock {
1202    /// The tapleaf version.
1203    pub leaf_version: LeafVersion,
1204    /// The parity of the output key (NOT THE INTERNAL KEY WHICH IS ALWAYS XONLY).
1205    pub output_key_parity: secp256k1::Parity,
1206    /// The internal key.
1207    pub internal_key: UntweakedPublicKey,
1208    /// The merkle proof of a script associated with this leaf.
1209    pub merkle_branch: TaprootMerkleBranch,
1210}
1211
1212impl ControlBlock {
1213    /// Constructs a `ControlBlock` from slice.
1214    #[deprecated(since = "0.30.0", note = "Use decode instead")]
1215    pub fn from_slice(sl: &[u8]) -> Result<ControlBlock, TaprootError> { Self::decode(sl) }
1216
1217    /// Decodes bytes representing a `ControlBlock`.
1218    ///
1219    /// This is an extra witness element that provides the proof that taproot script pubkey is
1220    /// correctly computed with some specified leaf hash. This is the last element in taproot
1221    /// witness when spending a output via script path.
1222    ///
1223    /// # Errors
1224    ///
1225    /// - [`TaprootError::InvalidControlBlockSize`] if `sl` is not of size 1 + 32 + 32N for any N >= 0.
1226    /// - [`TaprootError::InvalidParity`] if first byte of `sl` is not a valid output key parity.
1227    /// - [`TaprootError::InvalidTaprootLeafVersion`] if first byte of `sl` is not a valid leaf version.
1228    /// - [`TaprootError::InvalidInternalKey`] if internal key is invalid (first 32 bytes after the parity byte).
1229    /// - [`TaprootError::InvalidMerkleTreeDepth`] if merkle tree is too deep (more than 128 levels).
1230    pub fn decode(sl: &[u8]) -> Result<ControlBlock, TaprootError> {
1231        if sl.len() < TAPROOT_CONTROL_BASE_SIZE
1232            || (sl.len() - TAPROOT_CONTROL_BASE_SIZE) % TAPROOT_CONTROL_NODE_SIZE != 0
1233        {
1234            return Err(TaprootError::InvalidControlBlockSize(sl.len()));
1235        }
1236        let output_key_parity =
1237            secp256k1::Parity::from_i32((sl[0] & 1) as i32).map_err(TaprootError::InvalidParity)?;
1238        let leaf_version = LeafVersion::from_consensus(sl[0] & TAPROOT_LEAF_MASK)?;
1239        let internal_key = UntweakedPublicKey::from_slice(&sl[1..TAPROOT_CONTROL_BASE_SIZE])
1240            .map_err(TaprootError::InvalidInternalKey)?;
1241        let merkle_branch = TaprootMerkleBranch::decode(&sl[TAPROOT_CONTROL_BASE_SIZE..])?;
1242        Ok(ControlBlock { leaf_version, output_key_parity, internal_key, merkle_branch })
1243    }
1244
1245    /// Returns the size of control block. Faster and more efficient than calling
1246    /// `Self::serialize().len()`. Can be handy for fee estimation.
1247    pub fn size(&self) -> usize {
1248        TAPROOT_CONTROL_BASE_SIZE + TAPROOT_CONTROL_NODE_SIZE * self.merkle_branch.as_inner().len()
1249    }
1250
1251    /// Serializes to a writer.
1252    ///
1253    /// # Returns
1254    ///
1255    /// The number of bytes written to the writer.
1256    pub fn encode<Write: io::Write>(&self, mut writer: Write) -> io::Result<usize> {
1257        let first_byte: u8 =
1258            i32::from(self.output_key_parity) as u8 | self.leaf_version.to_consensus();
1259        writer.write_all(&[first_byte])?;
1260        writer.write_all(&self.internal_key.serialize())?;
1261        self.merkle_branch.encode(&mut writer)?;
1262        Ok(self.size())
1263    }
1264
1265    /// Serializes the control block.
1266    ///
1267    /// This would be required when using [`ControlBlock`] as a witness element while spending an
1268    /// output via script path. This serialization does not include the [`crate::VarInt`] prefix that would
1269    /// be applied when encoding this element as a witness.
1270    pub fn serialize(&self) -> Vec<u8> {
1271        let mut buf = Vec::with_capacity(self.size());
1272        self.encode(&mut buf).expect("writers don't error");
1273        buf
1274    }
1275
1276    /// Verifies that a control block is correct proof for a given output key and script.
1277    ///
1278    /// Only checks that script is contained inside the taptree described by output key. Full
1279    /// verification must also execute the script with witness data.
1280    pub fn verify_taproot_commitment<C: secp256k1::Verification>(
1281        &self,
1282        secp: &Secp256k1<C>,
1283        output_key: XOnlyPublicKey,
1284        script: &Script,
1285    ) -> bool {
1286        // compute the script hash
1287        // Initially the curr_hash is the leaf hash
1288        let mut curr_hash = TapNodeHash::from_script(script, self.leaf_version);
1289        // Verify the proof
1290        for elem in self.merkle_branch.as_inner() {
1291            // Recalculate the curr hash as parent hash
1292            curr_hash = TapNodeHash::from_node_hashes(curr_hash, *elem);
1293        }
1294        // compute the taptweak
1295        let tweak =
1296            TapTweakHash::from_key_and_tweak(self.internal_key, Some(curr_hash)).to_scalar();
1297        self.internal_key.tweak_add_check(secp, &output_key, self.output_key_parity, tweak)
1298    }
1299}
