36/// Serializer for command string
37#[derive(PartialEq, Eq, Clone, Debug)]
38pub struct CommandString(Cow<'static, str>);
39
40impl CommandString {
41    /// Converts `&'static str` to `CommandString`
42    ///
43    /// This is more efficient for string literals than non-static conversions because it avoids
44    /// allocation.
45    ///
46    /// # Errors
47    ///
48    /// Returns an error if, and only if, the string is
49    /// larger than 12 characters in length.
50    pub fn try_from_static(s: &'static str) -> Result<CommandString, CommandStringError> {
51        Self::try_from_static_cow(s.into())
52    }
53
54    fn try_from_static_cow(cow: Cow<'static, str>) -> Result<CommandString, CommandStringError> {
55        if cow.len() > 12 {
56            Err(CommandStringError { cow })
57        } else {
58            Ok(CommandString(cow))
59        }
60    }
61}
62
63impl TryFrom<String> for CommandString {
64    type Error = CommandStringError;
65
66    fn try_from(value: String) -> Result<Self, Self::Error> {
67        Self::try_from_static_cow(value.into())
68    }
69}
70
71impl TryFrom<Box<str>> for CommandString {
72    type Error = CommandStringError;
73
74    fn try_from(value: Box<str>) -> Result<Self, Self::Error> {
75        Self::try_from_static_cow(String::from(value).into())
76    }
77}
78
79impl<'a> TryFrom<&'a str> for CommandString {
80    type Error = CommandStringError;
81
82    fn try_from(value: &'a str) -> Result<Self, Self::Error> {
83        Self::try_from_static_cow(value.to_owned().into())
84    }
85}

87impl core::str::FromStr for CommandString {
88    type Err = CommandStringError;
89
90    fn from_str(s: &str) -> Result<Self, Self::Err> {
91        Self::try_from_static_cow(s.to_owned().into())
92    }
93}

97}

103impl Encodable for CommandString {
104    #[inline]
105    fn consensus_encode<W: io::Write + ?Sized>(&self, w: &mut W) -> Result<usize, io::Error> {
106        let mut rawbytes = [0u8; 12];
107        let strbytes = self.0.as_bytes();
108        debug_assert!(strbytes.len() <= 12);
109        rawbytes[..strbytes.len()].copy_from_slice(strbytes);
110        rawbytes.consensus_encode(w)
111    }
112}
113
114impl Decodable for CommandString {
115    #[inline]
116    fn consensus_decode<R: io::Read + ?Sized>(r: &mut R) -> Result<Self, encode::Error> {
117        let rawbytes: [u8; 12] = Decodable::consensus_decode(r)?;
118        let rv = iter::FromIterator::from_iter(rawbytes.iter().filter_map(|&u| {
119            if u > 0 {
120                Some(u as char)
121            } else {
122                None
123            }
124        }));
125        Ok(CommandString(rv))
126    }
127}

129/// Error returned when a command string is invalid.
130///
131/// This is currently returned for command strings longer than 12.
132#[derive(Clone, Debug)]
133pub struct CommandStringError {
134    cow: Cow<'static, str>,
135}
136
137impl fmt::Display for CommandStringError {
138    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
139        write!(
140            f,
141            "the command string '{}' has length {} which is larger than 12",
142            self.cow,
143            self.cow.len()
144        )
145    }
146}

159/// A Network message payload. Proper documentation is available on at
160/// [Bitcoin Wiki: Protocol Specification](https://en.bitcoin.it/wiki/Protocol_specification)
161#[derive(Clone, PartialEq, Eq, Debug)]
162pub enum NetworkMessage {
163    /// `version`
164    Version(message_network::VersionMessage),
165    /// `verack`
166    Verack,
167    /// `addr`
168    Addr(Vec<(u32, Address)>),
169    /// `inv`
170    Inv(Vec<message_blockdata::Inventory>),
171    /// `getdata`
172    GetData(Vec<message_blockdata::Inventory>),
173    /// `notfound`
174    NotFound(Vec<message_blockdata::Inventory>),
175    /// `getblocks`
176    GetBlocks(message_blockdata::GetBlocksMessage),
177    /// `getheaders`
178    GetHeaders(message_blockdata::GetHeadersMessage),
179    /// `mempool`
180    MemPool,
181    /// tx
182    Tx(transaction::Transaction),
183    /// `block`
184    Block(block::Block),
185    /// `headers`
186    Headers(Vec<block::Header>),
187    /// `sendheaders`
188    SendHeaders,
189    /// `getaddr`
190    GetAddr,
191    // TODO: checkorder,
192    // TODO: submitorder,
193    // TODO: reply,
194    /// `ping`
195    Ping(u64),
196    /// `pong`
197    Pong(u64),
198    /// `merkleblock`
199    MerkleBlock(MerkleBlock),
200    /// BIP 37 `filterload`
201    FilterLoad(message_bloom::FilterLoad),
202    /// BIP 37 `filteradd`
203    FilterAdd(message_bloom::FilterAdd),
204    /// BIP 37 `filterclear`
205    FilterClear,
206    /// BIP157 getcfilters
207    GetCFilters(message_filter::GetCFilters),
208    /// BIP157 cfilter
209    CFilter(message_filter::CFilter),
210    /// BIP157 getcfheaders
211    GetCFHeaders(message_filter::GetCFHeaders),
212    /// BIP157 cfheaders
213    CFHeaders(message_filter::CFHeaders),
214    /// BIP157 getcfcheckpt
215    GetCFCheckpt(message_filter::GetCFCheckpt),
216    /// BIP157 cfcheckpt
217    CFCheckpt(message_filter::CFCheckpt),
218    /// BIP152 sendcmpct
219    SendCmpct(message_compact_blocks::SendCmpct),
220    /// BIP152 cmpctblock
221    CmpctBlock(message_compact_blocks::CmpctBlock),
222    /// BIP152 getblocktxn
223    GetBlockTxn(message_compact_blocks::GetBlockTxn),
224    /// BIP152 blocktxn
225    BlockTxn(message_compact_blocks::BlockTxn),
226    /// `alert`
227    Alert(Vec<u8>),
228    /// `reject`
229    Reject(message_network::Reject),
230    /// `feefilter`
231    FeeFilter(i64),
232    /// `wtxidrelay`
233    WtxidRelay,
234    /// `addrv2`
235    AddrV2(Vec<AddrV2Message>),
236    /// `sendaddrv2`
237    SendAddrV2,
238
239    /// Any other message.
240    Unknown {
241        /// The command of this message.
242        command: CommandString,
243        /// The payload of this message.
244        payload: Vec<u8>,
245    },
246}

248impl NetworkMessage {
249    /// Return the message command as a static string reference.
250    ///
251    /// This returns `"unknown"` for [NetworkMessage::Unknown],
252    /// regardless of the actual command in the unknown message.
253    /// Use the [Self::command] method to get the command for unknown messages.
254    pub fn cmd(&self) -> &'static str {
255        match *self {
256            NetworkMessage::Version(_) => "version",
257            NetworkMessage::Verack => "verack",
258            NetworkMessage::Addr(_) => "addr",
259            NetworkMessage::Inv(_) => "inv",
260            NetworkMessage::GetData(_) => "getdata",
261            NetworkMessage::NotFound(_) => "notfound",
262            NetworkMessage::GetBlocks(_) => "getblocks",
263            NetworkMessage::GetHeaders(_) => "getheaders",
264            NetworkMessage::MemPool => "mempool",
265            NetworkMessage::Tx(_) => "tx",
266            NetworkMessage::Block(_) => "block",
267            NetworkMessage::Headers(_) => "headers",
268            NetworkMessage::SendHeaders => "sendheaders",
269            NetworkMessage::GetAddr => "getaddr",
270            NetworkMessage::Ping(_) => "ping",
271            NetworkMessage::Pong(_) => "pong",
272            NetworkMessage::MerkleBlock(_) => "merkleblock",
273            NetworkMessage::FilterLoad(_) => "filterload",
274            NetworkMessage::FilterAdd(_) => "filteradd",
275            NetworkMessage::FilterClear => "filterclear",
276            NetworkMessage::GetCFilters(_) => "getcfilters",
277            NetworkMessage::CFilter(_) => "cfilter",
278            NetworkMessage::GetCFHeaders(_) => "getcfheaders",
279            NetworkMessage::CFHeaders(_) => "cfheaders",
280            NetworkMessage::GetCFCheckpt(_) => "getcfcheckpt",
281            NetworkMessage::CFCheckpt(_) => "cfcheckpt",
282            NetworkMessage::SendCmpct(_) => "sendcmpct",
283            NetworkMessage::CmpctBlock(_) => "cmpctblock",
284            NetworkMessage::GetBlockTxn(_) => "getblocktxn",
285            NetworkMessage::BlockTxn(_) => "blocktxn",
286            NetworkMessage::Alert(_) => "alert",
287            NetworkMessage::Reject(_) => "reject",
288            NetworkMessage::FeeFilter(_) => "feefilter",
289            NetworkMessage::WtxidRelay => "wtxidrelay",
290            NetworkMessage::AddrV2(_) => "addrv2",
291            NetworkMessage::SendAddrV2 => "sendaddrv2",
292            NetworkMessage::Unknown { .. } => "unknown",
293        }
294    }
295
296    /// Return the CommandString for the message command.
297    pub fn command(&self) -> CommandString {
298        match *self {
299            NetworkMessage::Unknown { command: ref c, .. } => c.clone(),
300            _ => CommandString::try_from_static(self.cmd()).expect("cmd returns valid commands"),
301        }
302    }
303}

317struct HeaderSerializationWrapper<'a>(&'a Vec<block::Header>);
318
319impl<'a> Encodable for HeaderSerializationWrapper<'a> {
320    #[inline]
321    fn consensus_encode<W: io::Write + ?Sized>(&self, w: &mut W) -> Result<usize, io::Error> {
322        let mut len = 0;
323        len += VarInt(self.0.len() as u64).consensus_encode(w)?;
324        for header in self.0.iter() {
325            len += header.consensus_encode(w)?;
326            len += 0u8.consensus_encode(w)?;
327        }
328        Ok(len)
329    }
330}
331
332impl Encodable for RawNetworkMessage {
333    fn consensus_encode<W: io::Write + ?Sized>(&self, w: &mut W) -> Result<usize, io::Error> {
334        let mut len = 0;
335        len += self.magic.consensus_encode(w)?;
336        len += self.command().consensus_encode(w)?;
337        len += CheckedData(match self.payload {
338            NetworkMessage::Version(ref dat) => serialize(dat),
339            NetworkMessage::Addr(ref dat) => serialize(dat),
340            NetworkMessage::Inv(ref dat) => serialize(dat),
341            NetworkMessage::GetData(ref dat) => serialize(dat),
342            NetworkMessage::NotFound(ref dat) => serialize(dat),
343            NetworkMessage::GetBlocks(ref dat) => serialize(dat),
344            NetworkMessage::GetHeaders(ref dat) => serialize(dat),
345            NetworkMessage::Tx(ref dat) => serialize(dat),
346            NetworkMessage::Block(ref dat) => serialize(dat),
347            NetworkMessage::Headers(ref dat) => serialize(&HeaderSerializationWrapper(dat)),
348            NetworkMessage::Ping(ref dat) => serialize(dat),
349            NetworkMessage::Pong(ref dat) => serialize(dat),
350            NetworkMessage::MerkleBlock(ref dat) => serialize(dat),
351            NetworkMessage::FilterLoad(ref dat) => serialize(dat),
352            NetworkMessage::FilterAdd(ref dat) => serialize(dat),
353            NetworkMessage::GetCFilters(ref dat) => serialize(dat),
354            NetworkMessage::CFilter(ref dat) => serialize(dat),
355            NetworkMessage::GetCFHeaders(ref dat) => serialize(dat),
356            NetworkMessage::CFHeaders(ref dat) => serialize(dat),
357            NetworkMessage::GetCFCheckpt(ref dat) => serialize(dat),
358            NetworkMessage::CFCheckpt(ref dat) => serialize(dat),
359            NetworkMessage::SendCmpct(ref dat) => serialize(dat),
360            NetworkMessage::CmpctBlock(ref dat) => serialize(dat),
361            NetworkMessage::GetBlockTxn(ref dat) => serialize(dat),
362            NetworkMessage::BlockTxn(ref dat) => serialize(dat),
363            NetworkMessage::Alert(ref dat) => serialize(dat),
364            NetworkMessage::Reject(ref dat) => serialize(dat),
365            NetworkMessage::FeeFilter(ref data) => serialize(data),
366            NetworkMessage::AddrV2(ref dat) => serialize(dat),
367            NetworkMessage::Verack
368            | NetworkMessage::SendHeaders
369            | NetworkMessage::MemPool
370            | NetworkMessage::GetAddr
371            | NetworkMessage::WtxidRelay
372            | NetworkMessage::FilterClear
373            | NetworkMessage::SendAddrV2 => vec![],
374            NetworkMessage::Unknown { payload: ref data, .. } => serialize(data),
375        })
376        .consensus_encode(w)?;
377        Ok(len)
378    }
379}

381struct HeaderDeserializationWrapper(Vec<block::Header>);
382
383impl Decodable for HeaderDeserializationWrapper {
384    #[inline]
385    fn consensus_decode_from_finite_reader<R: io::Read + ?Sized>(
386        r: &mut R,
387    ) -> Result<Self, encode::Error> {
388        let len = VarInt::consensus_decode(r)?.0;
389        // should be above usual number of items to avoid
390        // allocation
391        let mut ret = Vec::with_capacity(core::cmp::min(1024 * 16, len as usize));
392        for _ in 0..len {
393            ret.push(Decodable::consensus_decode(r)?);
394            if u8::consensus_decode(r)? != 0u8 {
395                return Err(encode::Error::ParseFailed(
396                    "Headers message should not contain transactions",
397                ));
398            }
399        }
400        Ok(HeaderDeserializationWrapper(ret))
401    }
402
403    #[inline]
404    fn consensus_decode<R: io::Read + ?Sized>(r: &mut R) -> Result<Self, encode::Error> {
405        Self::consensus_decode_from_finite_reader(r.take(MAX_MSG_SIZE as u64).by_ref())
406    }
407}

409impl Decodable for RawNetworkMessage {
410    fn consensus_decode_from_finite_reader<R: io::Read + ?Sized>(
411        r: &mut R,
412    ) -> Result<Self, encode::Error> {
413        let magic = Decodable::consensus_decode_from_finite_reader(r)?;
414        let cmd = CommandString::consensus_decode_from_finite_reader(r)?;
415        let raw_payload = CheckedData::consensus_decode_from_finite_reader(r)?.0;
416
417        let mut mem_d = io::Cursor::new(raw_payload);
418        let payload = match &cmd.0[..] {
419            "version" =>
420                NetworkMessage::Version(Decodable::consensus_decode_from_finite_reader(&mut mem_d)?),
421            "verack" => NetworkMessage::Verack,
422            "addr" =>
423                NetworkMessage::Addr(Decodable::consensus_decode_from_finite_reader(&mut mem_d)?),
424            "inv" =>
425                NetworkMessage::Inv(Decodable::consensus_decode_from_finite_reader(&mut mem_d)?),
426            "getdata" =>
427                NetworkMessage::GetData(Decodable::consensus_decode_from_finite_reader(&mut mem_d)?),
428            "notfound" => NetworkMessage::NotFound(Decodable::consensus_decode_from_finite_reader(
429                &mut mem_d,
430            )?),
431            "getblocks" => NetworkMessage::GetBlocks(
432                Decodable::consensus_decode_from_finite_reader(&mut mem_d)?,
433            ),
434            "getheaders" => NetworkMessage::GetHeaders(

654        assert_eq!(
655            CommandString::try_from_static("AndrewAndrew").unwrap().as_ref(),
656            "AndrewAndrew"
657        );
658        assert!(CommandString::try_from_static("AndrewAndrewA").is_err());
659
660        // Test serializing.
661        let cs = CommandString("Andrew".into());
662        assert_eq!(serialize(&cs), vec![0x41u8, 0x6e, 0x64, 0x72, 0x65, 0x77, 0, 0, 0, 0, 0, 0]);
663
664        // Test deserializing
665        let cs: Result<CommandString, _> =
666            deserialize(&[0x41u8, 0x6e, 0x64, 0x72, 0x65, 0x77, 0, 0, 0, 0, 0, 0]);
667        assert!(cs.is_ok());
668        assert_eq!(cs.as_ref().unwrap().to_string(), "Andrew".to_owned());
669        assert_eq!(cs.unwrap(), CommandString::try_from_static("Andrew").unwrap());
670

672            deserialize(&[0x41u8, 0x6e, 0x64, 0x72, 0x65, 0x77, 0, 0, 0, 0, 0]);
673        assert!(short_cs.is_err());
674    }
