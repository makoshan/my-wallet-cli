18/// A message which can be sent on the Bitcoin network
19#[derive(Clone, PartialEq, Eq, Hash)]
20pub struct Address {
21    /// Services provided by the peer whose address this is
22    pub services: ServiceFlags,
23    /// Network byte-order ipv6 address, or ipv4-mapped ipv6 address
24    pub address: [u16; 8],
25    /// Network port
26    pub port: u16,
27}
28
29const ONION: [u16; 3] = [0xFD87, 0xD87E, 0xEB43];
30
31impl Address {
32    /// Create an address message for a socket
33    pub fn new(socket: &SocketAddr, services: ServiceFlags) -> Address {
34        let (address, port) = match *socket {
35            SocketAddr::V4(addr) => (addr.ip().to_ipv6_mapped().segments(), addr.port()),
36            SocketAddr::V6(addr) => (addr.ip().segments(), addr.port()),
37        };
38        Address { address, port, services }
39    }
40
41    /// Extract socket address from an [Address] message.
42    /// This will return [io::Error] [io::ErrorKind::AddrNotAvailable]
43    /// if the message contains a Tor address.
44    pub fn socket_addr(&self) -> Result<SocketAddr, io::Error> {
45        let addr = &self.address;
46        if addr[0..3] == ONION {
47            return Err(io::Error::from(io::ErrorKind::AddrNotAvailable));
48        }
49        let ipv6 =
50            Ipv6Addr::new(addr[0], addr[1], addr[2], addr[3], addr[4], addr[5], addr[6], addr[7]);
51        if let Some(ipv4) = ipv6.to_ipv4() {
52            Ok(SocketAddr::V4(SocketAddrV4::new(ipv4, self.port)))
53        } else {
54            Ok(SocketAddr::V6(SocketAddrV6::new(ipv6, self.port, 0, 0)))
55        }
56    }
57}
58
59impl Encodable for Address {
60    #[inline]
61    fn consensus_encode<W: io::Write + ?Sized>(&self, w: &mut W) -> Result<usize, io::Error> {
62        let mut len = self.services.consensus_encode(w)?;
63
64        for word in &self.address {
65            w.write_all(&word.to_be_bytes())?;
66            len += 2;
67        }
68
69        w.write_all(&self.port.to_be_bytes())?;
70        len += 2;
71
72        Ok(len)
73    }
74}
75
76impl Decodable for Address {
77    #[inline]
78    fn consensus_decode<R: io::Read + ?Sized>(r: &mut R) -> Result<Self, encode::Error> {
79        Ok(Address {
80            services: Decodable::consensus_decode(r)?,
81            address: read_be_address(r)?,
82            port: u16::swap_bytes(Decodable::consensus_decode(r)?),
83        })
84    }
85}

87/// Read a big-endian address from reader.
88fn read_be_address<R: io::Read + ?Sized>(r: &mut R) -> Result<[u16; 8], encode::Error> {
89    let mut address = [0u16; 8];
90    let mut buf = [0u8; 2];
91
92    for word in &mut address {
93        io::Read::read_exact(r, &mut buf)?;
94        *word = u16::from_be_bytes(buf)
95    }
96    Ok(address)
97}

99impl fmt::Debug for Address {
100    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
101        let ipv6 = Ipv6Addr::from(self.address);
102
103        match ipv6.to_ipv4() {
104            Some(addr) => write!(
105                f,
106                "Address {{services: {}, address: {}, port: {}}}",
107                self.services, addr, self.port
108            ),
109            None => write!(
110                f,
111                "Address {{services: {}, address: {}, port: {}}}",
112                self.services, ipv6, self.port
113            ),
114        }
115    }
116}

125/// Supported networks for use in BIP155 addrv2 message
126#[derive(Clone, PartialEq, Eq, Hash, Debug)]
127pub enum AddrV2 {
128    /// IPV4
129    Ipv4(Ipv4Addr),
130    /// IPV6
131    Ipv6(Ipv6Addr),
132    /// TORV2
133    TorV2([u8; 10]),
134    /// TORV3
135    TorV3([u8; 32]),
136    /// I2P
137    I2p([u8; 32]),
138    /// CJDNS
139    Cjdns(Ipv6Addr),
140    /// Unknown
141    Unknown(u8, Vec<u8>),
142}
143
144impl Encodable for AddrV2 {
145    fn consensus_encode<W: io::Write + ?Sized>(&self, e: &mut W) -> Result<usize, io::Error> {
146        fn encode_addr<W: io::Write + ?Sized>(
147            w: &mut W,
148            network: u8,
149            bytes: &[u8],
150        ) -> Result<usize, io::Error> {
151            let len = network.consensus_encode(w)?
152                + VarInt(bytes.len() as u64).consensus_encode(w)?
153                + bytes.len();
154            w.emit_slice(bytes)?;
155            Ok(len)
156        }
157        Ok(match *self {
158            AddrV2::Ipv4(ref addr) => encode_addr(e, 1, &addr.octets())?,
159            AddrV2::Ipv6(ref addr) => encode_addr(e, 2, &addr.octets())?,
160            AddrV2::TorV2(ref bytes) => encode_addr(e, 3, bytes)?,
161            AddrV2::TorV3(ref bytes) => encode_addr(e, 4, bytes)?,
162            AddrV2::I2p(ref bytes) => encode_addr(e, 5, bytes)?,
163            AddrV2::Cjdns(ref addr) => encode_addr(e, 6, &addr.octets())?,
164            AddrV2::Unknown(network, ref bytes) => encode_addr(e, network, bytes)?,
165        })
166    }
167}
168
169impl Decodable for AddrV2 {
170    fn consensus_decode<R: io::Read + ?Sized>(r: &mut R) -> Result<Self, encode::Error> {
171        let network_id = u8::consensus_decode(r)?;
172        let len = VarInt::consensus_decode(r)?.0;
173        if len > 512 {
174            return Err(encode::Error::ParseFailed("IP must be <= 512 bytes"));
175        }
176        Ok(match network_id {
177            1 => {
178                if len != 4 {
179                    return Err(encode::Error::ParseFailed("Invalid IPv4 address"));
180                }
181                let addr: [u8; 4] = Decodable::consensus_decode(r)?;
182                AddrV2::Ipv4(Ipv4Addr::new(addr[0], addr[1], addr[2], addr[3]))
183            }
184            2 => {
185                if len != 16 {
186                    return Err(encode::Error::ParseFailed("Invalid IPv6 address"));
187                }
188                let addr: [u16; 8] = read_be_address(r)?;
189                if addr[0..3] == ONION {
190                    return Err(encode::Error::ParseFailed(
191                        "OnionCat address sent with IPv6 network id",
192                    ));
193                }
194                if addr[0..6] == [0x0000, 0x0000, 0x0000, 0x0000, 0x0000, 0xFFFF] {
195                    return Err(encode::Error::ParseFailed(
196                        "IPV4 wrapped address sent with IPv6 network id",
197                    ));
198                }
199                AddrV2::Ipv6(Ipv6Addr::new(
200                    addr[0], addr[1], addr[2], addr[3], addr[4], addr[5], addr[6], addr[7],
201                ))
202            }
203            3 => {
204                if len != 10 {
205                    return Err(encode::Error::ParseFailed("Invalid TorV2 address"));
206                }
207                let id = Decodable::consensus_decode(r)?;
208                AddrV2::TorV2(id)
209            }
210            4 => {
211                if len != 32 {
212                    return Err(encode::Error::ParseFailed("Invalid TorV3 address"));
213                }
214                let pubkey = Decodable::consensus_decode(r)?;
215                AddrV2::TorV3(pubkey)
216            }
217            5 => {
218                if len != 32 {
219                    return Err(encode::Error::ParseFailed("Invalid I2P address"));
220                }
221                let hash = Decodable::consensus_decode(r)?;
222                AddrV2::I2p(hash)
223            }
224            6 => {
225                if len != 16 {
226                    return Err(encode::Error::ParseFailed("Invalid CJDNS address"));
227                }
228                let addr: [u16; 8] = read_be_address(r)?;
229                // check the first byte for the CJDNS marker
230                if addr[0] != u16::from_be_bytes([0xFC, 0x00]) {
231                    return Err(encode::Error::ParseFailed("Invalid CJDNS address"));
232                }
233                AddrV2::Cjdns(Ipv6Addr::new(
234                    addr[0], addr[1], addr[2], addr[3], addr[4], addr[5], addr[6], addr[7],
235                ))
236            }
237            _ => {
238                // len already checked above to be <= 512
239                let mut addr = vec![0u8; len as usize];
240                r.read_slice(&mut addr)?;
241                AddrV2::Unknown(network_id, addr)
242            }
243        })
244    }
245}

247/// Address received from BIP155 addrv2 message
248#[derive(Clone, PartialEq, Eq, Hash, Debug)]
249pub struct AddrV2Message {
250    /// Time that this node was last seen as connected to the network
251    pub time: u32,
252    /// Service bits
253    pub services: ServiceFlags,
254    /// Network ID + Network Address
255    pub addr: AddrV2,
256    /// Network port, 0 if not applicable
257    pub port: u16,
258}
259
260impl AddrV2Message {
261    /// Extract socket address from an [AddrV2Message] message.
262    /// This will return [io::Error] [io::ErrorKind::AddrNotAvailable]
263    /// if the address type can't be converted into a [SocketAddr].
264    pub fn socket_addr(&self) -> Result<SocketAddr, io::Error> {
265        match self.addr {
266            AddrV2::Ipv4(addr) => Ok(SocketAddr::V4(SocketAddrV4::new(addr, self.port))),
267            AddrV2::Ipv6(addr) => Ok(SocketAddr::V6(SocketAddrV6::new(addr, self.port, 0, 0))),
268            _ => Err(io::Error::from(io::ErrorKind::AddrNotAvailable)),
269        }
270    }
271}
272
273impl Encodable for AddrV2Message {
274    fn consensus_encode<W: io::Write + ?Sized>(&self, w: &mut W) -> Result<usize, io::Error> {
275        let mut len = 0;
276        len += self.time.consensus_encode(w)?;
277        len += VarInt(self.services.to_u64()).consensus_encode(w)?;
278        len += self.addr.consensus_encode(w)?;
279
280        w.write_all(&self.port.to_be_bytes())?;
281        len += 2; // port u16 is two bytes.
282
283        Ok(len)
284    }
285}
286
287impl Decodable for AddrV2Message {
288    fn consensus_decode<R: io::Read + ?Sized>(r: &mut R) -> Result<Self, encode::Error> {
289        Ok(AddrV2Message {
290            time: Decodable::consensus_decode(r)?,
291            services: ServiceFlags::from(VarInt::consensus_decode(r)?.0),
292            addr: Decodable::consensus_decode(r)?,
293            port: u16::swap_bytes(Decodable::consensus_decode(r)?),
294        })
295    }
296}
297
298impl ToSocketAddrs for AddrV2Message {
299    type Iter = iter::Once<SocketAddr>;
300    fn to_socket_addrs(&self) -> Result<Self::Iter, io::Error> {
301        Ok(iter::once(self.socket_addr()?))
302    }
303}

439    #[test]
440    fn deserialize_addrv2_test() {
441        // Taken from https://github.com/bitcoin/bitcoin/blob/12a1c3ad1a43634d2a98717e49e3f02c4acea2fe/src/test/net_tests.cpp#L386
442
443        // Valid IPv4.
444        let ip: AddrV2 = deserialize(&hex!("010401020304")).unwrap();
445        assert_eq!(ip, AddrV2::Ipv4(Ipv4Addr::new(1, 2, 3, 4)));
446
447        // Invalid IPv4, valid length but address itself is shorter.
448        deserialize::<AddrV2>(&hex!("01040102")).unwrap_err();
449
450        // Invalid IPv4, with bogus length.
451        assert!(deserialize::<AddrV2>(&hex!("010501020304")).is_err());
452
453        // Invalid IPv4, with extreme length.
454        assert!(deserialize::<AddrV2>(&hex!("01fd010201020304")).is_err());
455
456        // Valid IPv6.
457        let ip: AddrV2 = deserialize(&hex!("02100102030405060708090a0b0c0d0e0f10")).unwrap();
458        assert_eq!(
459            ip,
460            AddrV2::Ipv6(Ipv6Addr::from_str("102:304:506:708:90a:b0c:d0e:f10").unwrap())
461        );
462
463        // Invalid IPv6, with bogus length.
464        assert!(deserialize::<AddrV2>(&hex!("020400")).is_err());
465
466        // Invalid IPv6, contains embedded IPv4.
467        assert!(deserialize::<AddrV2>(&hex!("021000000000000000000000ffff01020304")).is_err());
468
469        // Invalid IPv6, contains embedded TORv2.
470        assert!(deserialize::<AddrV2>(&hex!("0210fd87d87eeb430102030405060708090a")).is_err());
471
472        // Valid TORv2.
473        let ip: AddrV2 = deserialize(&hex!("030af1f2f3f4f5f6f7f8f9fa")).unwrap();
474        assert_eq!(ip, AddrV2::TorV2(FromHex::from_hex("f1f2f3f4f5f6f7f8f9fa").unwrap()));
475
476        // Invalid TORv2, with bogus length.
477        assert!(deserialize::<AddrV2>(&hex!("030700")).is_err());
478
479        // Valid TORv3.
480        let ip: AddrV2 = deserialize(&hex!(
481            "042079bcc625184b05194975c28b66b66b0469f7f6556fb1ac3189a79b40dda32f1f"
482        ))
483        .unwrap();
484        assert_eq!(
485            ip,
486            AddrV2::TorV3(
487                FromHex::from_hex(
488                    "79bcc625184b05194975c28b66b66b0469f7f6556fb1ac3189a79b40dda32f1f"
489                )
490                .unwrap()
491            )
492        );
493
494        // Invalid TORv3, with bogus length.
495        assert!(deserialize::<AddrV2>(&hex!("040000")).is_err());
496
497        // Valid I2P.
498        let ip: AddrV2 = deserialize(&hex!(
499            "0520a2894dabaec08c0051a481a6dac88b64f98232ae42d4b6fd2fa81952dfe36a87"
500        ))
501        .unwrap();
502        assert_eq!(
503            ip,
504            AddrV2::I2p(
505                FromHex::from_hex(
506                    "a2894dabaec08c0051a481a6dac88b64f98232ae42d4b6fd2fa81952dfe36a87"
507                )
508                .unwrap()
509            )
510        );
511
512        // Invalid I2P, with bogus length.
513        assert!(deserialize::<AddrV2>(&hex!("050300")).is_err());
514
515        // Valid CJDNS.
516        let ip: AddrV2 = deserialize(&hex!("0610fc000001000200030004000500060007")).unwrap();
517        assert_eq!(ip, AddrV2::Cjdns(Ipv6Addr::from_str("fc00:1:2:3:4:5:6:7").unwrap()));
518
519        // Invalid CJDNS, incorrect marker
520        assert!(deserialize::<AddrV2>(&hex!("0610fd000001000200030004000500060007")).is_err());
521
522        // Invalid CJDNS, with bogus length.
523        assert!(deserialize::<AddrV2>(&hex!("060100")).is_err());
524
525        // Unknown, with extreme length.
526        assert!(deserialize::<AddrV2>(&hex!("aafe0000000201020304050607")).is_err());
527
528        // Unknown, with reasonable length.
529        let ip: AddrV2 = deserialize(&hex!("aa0401020304")).unwrap();
530        assert_eq!(ip, AddrV2::Unknown(170, hex!("01020304")));
531
532        // Unknown, with zero length.
533        let ip: AddrV2 = deserialize(&hex!("aa00")).unwrap();
534        assert_eq!(ip, AddrV2::Unknown(170, vec![]));
535    }
